 3def warshall_floyd(G: List[List[Tuple[int, int]]], INF: Union[int, float]=float('inf')) -> List[List[Union[int, float]]]:
 4  """重み付き隣接リスト ``G`` に対し、全点対最短経路を返します。
 5
 6  :math:`O(n^3)` です。
 7
 8  Args:
 9      G (List[List[Tuple[int, int]]]): 重み付き隣接リストです。
10      INF (Union[int, float], optional): 無限大です。
11
12  Returns:
13      List[List[Union[int, float]]]: dist[a][b] -> a to b
14  """
15  n = len(G)
16  dist = [[INF]*n for _ in range(n)]
17  for v in range(n):
18    dist_v_ = dist[v]
19    for x, c in G[v]:
20      dist_v_[x] = c
21    dist_v_[v] = 0
22  for k in range(n):
23    dist_k_ = dist[k]
24    for i in range(n):
25      dist_i_ = dist[i]
26      dist_i_k_ = dist_i_[k]
27      if dist_i_k_ == INF: continue
28      for j, dist_k_j_ in enumerate(dist_k_):
29        if dist_i_[j] > dist_i_k_ + dist_k_j_:
30          dist_i_[j] = dist_i_k_ + dist_k_j_
31  return dist
