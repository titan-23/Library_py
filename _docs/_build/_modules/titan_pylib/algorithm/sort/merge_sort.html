 7def merge_sort(
 8    a: Iterable[T], key: Callable[[T, T], bool] = lambda s, t: s < t
 9) -> list[T]:
10    """マージソートです。安定なはずです。
11    最悪 :math:`O(n\\log{n})` 時間です。
12
13    Args:
14        a (Iterable[T]): ソートする列です。
15        key (Callable[[T, T], bool], optional): 比較関数 `key` にしたがって比較演算をします。
16                                                (第1引数)<=(第2引数) のとき、 ``True`` を返すようにしてください。
17    """
18
19    def _sort(l: int, r: int) -> None:
20        if r - l <= 1:
21            return
22        if r - l == 2:
23            if not key(a[l], a[l + 1]):
24                a[l], a[l + 1] = a[l + 1], a[l]
25            return
26        mid = (l + r) // 2
27        _sort(l, mid)
28        _sort(mid, r)
29        i, j = l, mid
30        indx = 0
31        while i < mid and j < r:
32            if key(a[i], a[j]):
33                buff[indx] = a[i]
34                i += 1
35            else:
36                buff[indx] = a[j]
37                j += 1
38            indx += 1
39        for k in range(i, mid):
40            a[l + indx + k - i] = a[k]
41        for k in range(l, l + indx):
42            a[k] = buff[k - l]
43
44    a = list(a)
45    buff = [0] * len(a)
46    _sort(0, len(a))
47    return a
