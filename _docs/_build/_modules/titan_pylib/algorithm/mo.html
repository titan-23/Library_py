  5class Mo():
  6  """長さ `n` の列、クエリ数 `q` に対する `Mo's algorithm` です。
  7  :math:`O(\\frac{n}{\\sqrt{q}})` です。
  8
  9  Args:
 10    n (int): 列の長さです。
 11    q (int): クエリの数です。
 12
 13  制約:
 14    :math:`0 \\leq n, 0 \\leq q`
 15  """
 16
 17  def __init__(self, n: int, q: int):
 18    assert 0 <= n and 0 <= q, f'ValueError: {n=} {q=}'
 19    self.n = n
 20    self.q = q
 21    self.bucket_size = ceil(sqrt(3)*n/sqrt(2*q)) if q > 0 else n
 22    if self.bucket_size == 0:
 23      self.bucket_size = 1
 24    self.bit = max(n, q).bit_length()
 25    self.msk = (1 << self.bit) - 1
 26    self.bucket = [[] for _ in range(n//self.bucket_size+1)]
 27    self.cnt = 0

 29  def add_query(self, l: int, r: int) -> None:
 30    """区間 ``[l, r)`` に対するクエリを追加します。
 31    :math:`O(1)` です。
 32
 33    Args:
 34      l (int):
 35      r (int):
 36
 37    制約:
 38      :math:`0 \\leq l \\leq r \\leq n`
 39    """
 40    assert 0 <= l <= r <= self.n, \
 41        f'IndexError: {self.__class__.__name__}.add_query({l}, {r}), self.n={self.n}'
 42    self.bucket[l//self.bucket_size].append((((r<<self.bit)|l)<<self.bit)|self.cnt)
 43    self.cnt += 1

 45  def run(self,
 46          add: Callable[[int], None],
 47          delete: Callable[[int], None],
 48          out: Callable[[int], None],
 49          ) -> None:
 50    """クエリを実行します。
 51    :math:`O(q\\sqrt{n})` です。
 52
 53    Args:
 54      add (Callable[[int], None]): 引数のインデックスに対応する要素を追加します。
 55      delete (Callable[[int], None]): 引数のインデックスに対応する要素を削除します。
 56      out (Callable[[int], None]): クエリ番号に対する答えを処理します。
 57
 58    制約:
 59      ``q`` 回のクエリを ``add_query`` メソッドで追加する必要があります。
 60    """
 61    assert self.cnt == self.q, \
 62        f'Not Enough Queries, now:{self.cnt}, expected:{self.q}'
 63    bucket, bit, msk = self.bucket, self.bit, self.msk
 64    for i, b in enumerate(bucket):
 65      b.sort(reverse=i & 1)
 66    nl, nr = 0, 0
 67    for rli in chain(*bucket):
 68      r, l = rli >> bit >> bit, rli >> bit & msk
 69      while nl > l:
 70        nl -= 1
 71        add(nl)
 72      while nr < r:
 73        add(nr)
 74        nr += 1
 75      while nl < l:
 76        delete(nl)
 77        nl += 1
 78      while nr > r:
 79        nr -= 1
 80        delete(nr)
 81      out(rli & msk)
