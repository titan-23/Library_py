  6_titan_pylib_HashString_MOD: Final[int] = (1 << 61) - 1
  7_titan_pylib_HashString_DIC: Final[Dict[str, int]] = {c: i for i, c in enumerate(string.ascii_lowercase, 1)}

 12class HashStringBase():
 13  """HashStringのベースクラスです。
 14  """
 15
 16  def __init__(self, n: int, base: int=-1, seed: Optional[int]=None) -> None:
 17    """
 18    :math:`O(n)` です。
 19
 20    Args:
 21        n (int): 文字列の長さの上限です。
 22        base (int, optional): Defaults to -1.
 23        seed (Optional[int], optional): Defaults to None.
 24    """
 25    random.seed(seed)
 26    base = random.randint(37, 10**9) if base < 0 else base
 27    powb = [1] * (n+1)
 28    invb = [1] * (n+1)
 29    invbpow = pow(base, -1, _titan_pylib_HashString_MOD)
 30    for i in range(1, n+1):
 31      powb[i] = HashStringBase.get_mul(powb[i-1], base)
 32      invb[i] = HashStringBase.get_mul(invb[i-1], invbpow)
 33    self.n = n
 34    self.powb = powb
 35    self.invb = invb

 37  @staticmethod
 38  def get_mul(a: int, b: int) -> int:
 39    au = a >> 31
 40    ad = a & _titan_pylib_HashString_MASK31
 41    bu = b >> 31
 42    bd = b & _titan_pylib_HashString_MASK31
 43    mid = ad * bu + au * bd
 44    midu = mid >> 30
 45    midd = mid & _titan_pylib_HashString_MASK30
 46    return HashStringBase.get_mod(au * bu * 2 + midu + (midd << 31) + ad * bd)

 48  @staticmethod
 49  def get_mod(x: int) -> int:
 50    xu = x >> 61
 51    xd = x & _titan_pylib_HashString_MASK61
 52    res = xu + xd
 53    if res >= _titan_pylib_HashString_MOD:
 54      res -= _titan_pylib_HashString_MOD
 55    return res

 62class HashString():
 63
 64  def __init__(self, hsb: HashStringBase, s: str, update: bool=False) -> None:
 65    """ロリハを構築します。
 66
 67    :math:`O(n)` です。
 68
 69    Args:
 70        hsb (HashStringBase): ベースクラスです。
 71        s (str): ロリハを構築する文字列です。
 72        update (bool, optional): ``update=True`` のとき、1点更新が可能になります。
 73    """
 74    n = len(s)
 75    data = [0] * n
 76    acc = [0] * (n+1)
 77    powb = hsb.powb
 78    for i, c in enumerate(s):
 79      data[i] = hsb.get_mul(powb[n-i-1], _titan_pylib_HashString_DIC[c])
 80      acc[i+1] = hsb.get_mod(acc[i] + data[i])
 81    self.hsb = hsb
 82    self.n = n
 83    self.acc = acc
 84    self.used_seg = False
 85    if update:
 86      self.seg = SegmentTree(data, lambda s, t: (s+t)%_titan_pylib_HashString_MOD, 0)

 88  def get(self, l: int, r: int) -> int:
 89    """``s[l, r)`` のハッシュ値を返します。
 90
 91    1点更新処理後は :math:`O(\\log{n})` 、そうでなければ :math:`O(1)` です。
 92
 93    Args:
 94      l (int): インデックスです。
 95      r (int): インデックスです。
 96
 97    Returns:
 98      int: ハッシュ値です。
 99    """
100    if self.used_seg:
101      return self.hsb.get_mul(self.seg.prod(l, r), self.hsb.invb[self.n-r])
102    return self.hsb.get_mul(self.hsb.get_mod(self.acc[r]-self.acc[l]), self.hsb.invb[self.n-r])

135  def get_lcp(self) -> List[int]:
136    """lcp配列を返します。
137
138    :math:`O(n\\log{n})` です。
139    """
140    a = [0] * self.n
141    memo = [-1] * (self.n+1)
142    for i in range(self.n):
143      ok, ng = 0, self.n-i+1
144      while ng - ok > 1:
145        mid = (ok + ng) >> 1
146        if memo[mid] == -1:
147          memo[mid] = self.get(0, mid)
148        if memo[mid] == self.get(i, i+mid):
149          ok = mid
150        else:
151          ng = mid
152      a[i] = ok
153    return a
