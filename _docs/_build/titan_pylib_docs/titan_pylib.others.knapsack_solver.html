13    def naive(vw):
14      ret = []
15      n = len(vw)
16      for i in range(1<<n):
17        value, weight = 0, 0
18        for j in range(n):
19          if i >> j & 1:
20            value += vw[j][0]
21            weight += vw[j][1]
22        ret.append([weight, value])
23      return ret
24    left, right = self.VW[:self.N//2], self.VW[self.N//2:]
25    left_ret = naive(left)
26    left_ret.sort()
27    right_ret = naive(right)
28    right_ret.sort()
29    for i in range(len(right_ret)-1):
30      right_ret[i+1][1] = max(right_ret[i+1][1], right_ret[i][1])
31    ans = -1
32    W = self.W
33    for weight, value in left_ret:
34      if weight > W: break
35      i = bisect_right(right_ret, [W-weight, float('inf')]) - 1
36      ans = max(ans, value + right_ret[i][1])
37    return ans

40    dp = [-1] * (self.W+1)
