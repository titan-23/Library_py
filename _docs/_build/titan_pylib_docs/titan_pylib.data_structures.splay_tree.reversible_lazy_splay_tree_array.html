170      pnode = path.pop()
171      gnode = path.pop()
172      f = d >> 1 & 1
173      node = arr[pnode<<2|g^1]
174      nnode = (pnode if g == f else node) << 2 | f
175      arr[pnode<<2|g^1] = arr[node<<2|g]
176      arr[node<<2|g] = pnode
177      arr[gnode<<2|f^1] = arr[nnode]
178      arr[nnode] = gnode
179      self._update_triple(gnode, pnode, node)
180      if not path:
181        return
182      d >>= 2
183      g = d & 1
184      arr[path[-1]<<2|g^1] = node
185    pnode = path.pop()
186    node = arr[pnode<<2|g^1]
187    arr[pnode<<2|g^1] = arr[node<<2|g]
188    arr[node<<2|g] = pnode
189    self._update_double(pnode, node)
190
191  def _kth_elm_splay(self, node: int, k: int) -> int:
192    arr = self.data.arr
193    if k < 0: k += arr[node<<2|2]
194    d = 0
195    path = []
196    while True:
197      self._propagate(node)
198      t = arr[arr[node<<2]<<2|2]
199      if t == k:
200        if path:
201          self._splay(path, d)
202        return node
203      d = d << 1 | (t > k)
204      path.append(node)
205      node = arr[node<<2|(t<k)]
206      if t < k:
207        k -= t + 1
208
209  def _left_splay(self, node: int) -> int:
210    if not node: return 0
211    self._propagate(node)
212    arr = self.data.arr
213    if not arr[node<<2]: return node
214    path = []
215    while arr[node<<2]:
216      path.append(node)
217      node = arr[node<<2]
218      self._propagate(node)
219    self._splay(path, (1<<len(path))-1)
220    return node
221
222  def _right_splay(self, node: int) -> int:
223    if not node: return 0
224    self._propagate(node)
225    arr = self.data.arr
226    if not arr[node<<2|1]: return node
227    path = []
228    while arr[node<<2|1]:
229      path.append(node)
230      node = arr[node<<2|1]
231      self._propagate(node)
232    self._splay(path, 0)
233    return node
234
235  def reserve(self, n: int) -> None:
236    self.data.reserve(n)

238  def merge(self, other: 'ReversibleLazySplayTreeArray') -> None:
239    assert self.data is other.data
240    if not other.root: return
241    if not self.root:
242      self.root = other.root
243      return
244    self.root = self._right_splay(self.root)
245    self.data.arr[self.root<<2|1] = other.root
246    self._update(self.root)
247
248  def split(self, k: int) -> Tuple['ReversibleLazySplayTreeArray', 'ReversibleLazySplayTreeArray']:
249    assert -len(self) < k <= len(self), \
250        f'IndexError: ReversibleLazySplayTreeArray.split({k}), len={len(self)}'
251    if k < 0: k += len(self)
252    if k >= self.data.arr[self.root<<2|2]:
253      return self, ReversibleLazySplayTreeArray(self.data, _root=0)
254    self.root = self._kth_elm_splay(self.root, k)
255    left = ReversibleLazySplayTreeArray(self.data, _root=self.data.arr[self.root<<2])
256    self.data.arr[self.root<<2] = 0
257    self._update(self.root)
258    return left, self
259
260  def _internal_split(self, k: int) -> Tuple[int, int]:
261    if k >= self.data.arr[self.root<<2|2]:
262      return self.root, 0
263    self.root = self._kth_elm_splay(self.root, k)
264    left = self.data.arr[self.root<<2]
265    self._propagate(left)
266    self.data.arr[self.root<<2] = 0
267    self._update(self.root)
268    return left, self.root
269
270  def reverse(self, l: int, r: int) -> None:
271    assert 0 <= l <= r <= len(self), \
272        f'IndexError: ReversibleLazySplayTreeArray.reverse({l}, {r}), len={len(self)}'
273    if l == r: return
274    data = self.data
275    left, right = self._internal_split(r)
276    if l:
277      left = self._kth_elm_splay(left, l-1)
278    data.arr[(data.arr[left<<2|1] if l else left)<<2|3] ^= 1
279    if right:
280      data.arr[right<<2] = left
281      self._update(right)
282    self.root = right if right else left
283
284  def all_reverse(self) -> None:
285    self.data.arr[self.root<<2|3] ^= 1
286    self._propagate(self.root)
287
288  def apply(self, l: int, r: int, f: F) -> None:
289    assert 0 <= l <= r <= len(self), \
290        f'IndexError: ReversibleLazySplayTreeArray.apply({l}, {r}), len={len(self)}'
291    data = self.data
292    left, right = self._internal_split(r)
293    keydata, lazy = data.keydata, data.lazy
294    if l:
295      left = self._kth_elm_splay(left, l-1)
296    node = data.arr[left<<2|1] if l else left
297    keydata[node*3+0] = data.mapping(f, keydata[node*3+0])
298    keydata[node*3+1] = data.mapping(f, keydata[node*3+1])
299    keydata[node*3+2] = data.mapping(f, keydata[node*3+2])
300    lazy[node] = data.composition(f, lazy[node])
301    if l:
302      self._update(left)
303    if right:
304      data.arr[right<<2] = left
305      self._update(right)
306    self.root = right if right else left
307
308  def all_apply(self, f: F) -> None:
309    if not self.root: return
310    data, node = self.data, self.root
311    data.keydata[node*3+0] = data.mapping(f, data.keydata[node*3+0])
312    data.keydata[node*3+1] = data.mapping(f, data.keydata[node*3+1])
313    data.keydata[node*3+2] = data.mapping(f, data.keydata[node*3+2])
314    data.lazy[node] = data.composition(f, data.lazy[node])
315
316  def prod(self, l: int, r: int) -> T:
317    assert 0 <= l <= r <= len(self), \
318        f'IndexError: LazySplayTree.prod({l}, {r}), len={len(self)}'
319    data = self.data
320    left, right = self._internal_split(r)
321    if l:
322      left = self._kth_elm_splay(left, l-1)
323    node = (data.arr[left<<2|1] if l else left)
324    self._propagate(node)
325    res = data.keydata[node*3+1]
326    if right:
327      data.arr[right<<2] = left
328      self._update(right)
329    self.root = right if right else left
330    return res

332  def all_prod(self) -> T:
333    return self.data.keydata[self.root*3+1]
334
335  def insert(self, k: int, key: T) -> None:
336    assert -len(self) <= k <= len(self), \
337        f'IndexError: ReversibleLazySplayTreeArray.insert({k}, {key}), len={len(self)}'
338    if k < 0: k += len(self)
339    data = self.data
340    node = self._make_node(key)
341    if not self.root:
342      self._update(node)
343      self.root = node
344      return
345    arr = data.arr
346    if k == data.arr[self.root<<2|2]:
347      arr[node<<2] = self._right_splay(self.root)
348    else:
349      node_ = self._kth_elm_splay(self.root, k)
350      if arr[node_<<2]:
351        arr[node<<2] = arr[node_<<2]
352        arr[node_<<2] = 0
353        self._update(node_)
354      arr[node<<2|1] = node_
355    self._update(node)
356    self.root = node

358  def append(self, key: T) -> None:
359    data = self.data
360    node = self._right_splay(self.root)
361    self.root = self._make_node(key)
362    data.arr[self.root<<2] = node
363    self._update(self.root)

365  def appendleft(self, key: T) -> None:
366    node = self._left_splay(self.root)
367    self.root = self._make_node(key)
368    self.data.arr[self.root<<2|1] = node
369    self._update(self.root)

371  def pop(self, k: int=-1) -> T:
372    assert -len(self) <= k < len(self), \
373        f'IndexError: ReversibleLazySplayTreeArray.pop({k})'
374    data = self.data
375    if k == -1:
376      node = self._right_splay(self.root)
377      self._propagate(node)
378      self.root = data.arr[node<<2]
379      return data.keydata[node*3+0]
380    self.root = self._kth_elm_splay(self.root, k)
381    res = data.keydata[self.root*3+0]
382    if not data.arr[self.root<<2]:
383      self.root = data.arr[self.root<<2|1]
384    elif not data.arr[self.root<<2|1]:
385      self.root = data.arr[self.root<<2]
386    else:
387      node = self._right_splay(data.arr[self.root<<2])
388      data.arr[node<<2|1] = data.arr[self.root<<2|1]
389      self.root = node
390      self._update(self.root)
391    return res

393  def popleft(self) -> T:
394    assert self, 'IndexError: ReversibleLazySplayTreeArray.popleft()'
395    node = self._left_splay(self.root)
396    self.root = self.data.arr[node<<2|1]
397    return self.data.keydata[node*3+0]

399  def rotate(self, x: int) -> None:
400    # 「末尾をを削除し先頭に挿入」をx回
401    n = self.data.arr[self.root<<2|2]
402    l, self = self.split(n-(x%n))
403    self.merge(l)

405  def tolist(self) -> List[T]:
406    node = self.root
407    arr, keydata = self.data.arr, self.data.keydata
408    stack = newlist_hint(len(self))
409    res = newlist_hint(len(self))
410    while stack or node:
411      if node:
412        self._propagate(node)
413        stack.append(node)
414        node = arr[node<<2]
415      else:
416        node = stack.pop()
417        res.append(keydata[node*3+0])
418        node = arr[node<<2|1]
419    return res
