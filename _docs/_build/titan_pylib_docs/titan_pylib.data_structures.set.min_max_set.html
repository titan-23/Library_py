 13class DoubleEndedHeap(Generic[T]):
 14  """
 15  - [両端優先度付きキューのInterval-Heap実装](https://natsugiri.hatenablog.com/entry/2016/10/10/035445)
 16  - [Double-ended priority queue(wikipedia)](https://en.wikipedia.org/wiki/Double-ended_priority_queue)
 17  """
 18
 19  def __init__(self, a: Iterable[T]=[]) -> None:
 20    """構築します。
 21    :math:`O(n)` です。
 22
 23    Args:
 24      a (Iterable[T], optional): 構築元の配列です。
 25    """
 26    self._data = list(a)
 27    self._heapify()
 28
 29  def _heapify(self) -> None:
 30    n = len(self._data)
 31    for i in range(n-1, -1, -1):
 32      if i & 1 and self._data[i-1] < self._data[i]:
 33        self._data[i-1], self._data[i] = self._data[i], self._data[i-1]
 34      k = self._down(i)
 35      self._up(k, i)
 36
 37  def add(self, key: T) -> None:
 38    """``key`` を1つ追加します。
 39    :math:`O(\\log{n})` です。
 40    """
 41    self._data.append(key)
 42    self._up(len(self._data)-1)
 43
 44  def pop_min(self) -> T:
 45    """最小の要素を削除して返します。
 46    :math:`O(\\log{n})` です。
 47    """
 48    if len(self._data) < 3:
 49      res = self._data.pop()
 50    else:
 51      self._data[1], self._data[-1] = self._data[-1], self._data[1]
 52      res = self._data.pop()
 53      k = self._down(1)
 54      self._up(k)
 55    return res
 56
 57  def pop_max(self) -> T:
 58    """最大の要素を削除して返します。
 59    :math:`O(\\log{n})` です。
 60    """
 61    if len(self._data) < 2:
 62      res = self._data.pop()
 63    else:
 64      self._data[0], self._data[-1] = self._data[-1], self._data[0]
 65      res = self._data.pop()
 66      self._up(self._down(0))
 67    return res
 68
 69  def get_min(self) -> T:
 70    """最小の要素を返します。
 71    :math:`O(1)` です。
 72    """
 73    return self._data[0] if len(self._data) < 2 else self._data[1]
 74
 75  def get_max(self) -> T:
 76    """最大の要素を返します。
 77    :math:`O(1)` です。
 78    """
 79    return self._data[0]

 81  def __len__(self) -> int:
 82    return len(self._data)

 84  def __bool__(self) -> bool:
 85    return len(self._data) > 0

 87  def _parent(self, k: int) -> int:
 88    return ((k>>1)-1) & ~1

 90  def _down(self, k: int) -> int:
 91    n = len(self._data)
 92    if k & 1:
 93      while k<<1|1 < n:
 94        c = 2*k+3
 95        if n <= c or self._data[c-2] < self._data[c]:
 96          c -= 2
 97        if c < n and self._data[c] < self._data[k]:
 98          self._data[k], self._data[c] = self._data[c], self._data[k]
 99          k = c
100        else:
101          break
102    else:
103      while 2*k+2 < n:
104        c = 2*k+4
105        if n <= c or self._data[c] < self._data[c-2]:
106          c -= 2
107        if c < n and self._data[k] < self._data[c]:
108          self._data[k], self._data[c] = self._data[c], self._data[k]
109          k = c
110        else:
111          break
112    return k
113
114  def _up(self, k: int, root: int=1) -> int:
115    if (k|1) < len(self._data) and self._data[k&~1] < self._data[k|1]:
116      self._data[k&~1], self._data[k|1] = self._data[k|1], self._data[k&~1]
117      k ^= 1
118    while root < k:
119      p = self._parent(k)
120      if not self._data[p] < self._data[k]:
121        break
122      self._data[p], self._data[k] = self._data[k], self._data[p]
123      k = p
124    while root < k:
125      p = self._parent(k) | 1
126      if not self._data[k] < self._data[p]:
127        break
128      self._data[p], self._data[k] = self._data[k], self._data[p]
129      k = p
130    return k

140from typing import Generic, Iterable, TypeVar, List
141T = TypeVar('T', bound=SupportsLessThan)

143class MinMaxSet(Generic[T]):
144
145  def __init__(self, a: Iterable[T]=[]):
146    a = set(a)
147    self.data = a
148    self.heap = DoubleEndedHeap(a)
149
150  def add(self, key: T) -> bool:
151    if key not in self.data:
152      self.heap.add(key)
153      self.data.add(key)
154      return True
155    return False
156
157  def discard(self, key: T) -> bool:
158    if key in self.data:
159      self.data.discard(key)
160      return True
161    return False
162
163  def pop_min(self) -> T:
164    while True:
165      v = self.heap.pop_min()
166      if v in self.data:
167        self.data.discard(v)
168        return v
169
170  def pop_max(self) -> T:
171    while True:
172      v = self.heap.pop_max()
173      if v in self.data:
174        self.data.discard(v)
175        return v
176
177  def get_min(self) -> T:
178    while True:
179      v = self.heap.get_min()
180      if v in self.data:
181        return v
182      else:
183        self.heap.pop_min()
184
185  def get_max(self) -> T:
186    while True:
187      v = self.heap.get_max()
188      if v in self.data:
189        return v
190      else:
191        self.heap.pop_max()
192
193  def tolist(self) -> List[T]:
194    return sorted(self.data)
195
196  def __contains__(self, key: T):
197    return key in self.data
198
199  def __getitem__(self, k: int):  # 末尾と先頭のみ
200    if k == -1 or k == len(self.data)-1:
201      return self.get_max()
202    elif k == 0:
203      return self.get_min()
204    raise IndexError
205
206  def __len__(self):
207    return len(self.data)
208
209  def __str__(self):
