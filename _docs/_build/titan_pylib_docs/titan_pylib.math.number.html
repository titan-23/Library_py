13def modinv(a, mod):
14  b = mod
15  x, y, u, v = 1, 0, 0, 1
16  while b:
17    k = a // b
18    x -= k * u
19    y -= k * v
20    x, u = u, x
21    y, v = v, y
22    a, b = b, a % b
23  x %= mod
24  return x

26def isqrt(n: int) -> int:
27  assert n >= 0
28  if n == 0:
29    return 0
30  x = 1 << (n.bit_length() + 1) >> 1
31  y = (x + n // x) >> 1
32  while y < x:
33    x, y = y, (y + n // y) >> 1
34  return x

41def lcm_mod(o: Divs.Osa_k, A: list, mod: int) -> int:
42  cou = Counter()
43  for a in A:
44    for k, v in Counter(o.p_factorization(a)).items():
45      cou[k] = max(cou[k], v)

71def mat_mul(A: list, B: list, mod: int) -> list:
72  l, m = len(A), len(A[0])
73  n = len(B[0])
74  assert m == len(B)
75  return [[sum([A[i][k]*B[k][j] for k in range(m)]) % mod for j in range(n)] for i in range(l)]

77def mat_powmod(A: list, n: int, mod: int) -> list:
78  res = [[0]*len(A) for _ in range(len(A))]
79  for i in range(len(A)):
80    res[i][i] = 1
81  while n > 0:
82    if n & 1 == 1:
83      res = mat_mul(A, res, mod)
84    A = mat_mul(A, A, mod)
85    n >>= 1
86  return res
