 3T = TypeVar('T')

 5class Doubling(Generic[T]):
 6  """ダブリングテーブルを構築します。
 7  :math:`O(n\\log{lim})` です。
 8
 9  Args:
10    Generic (T): 状態の型です。
11    n (int): テーブルサイズです。
12    lim (int): クエリの最大数です。
13    move_to (Callable[[T], T]): 遷移関数です。 ``u`` から ``v`` へ遷移します。
14  """
15
16  def __init__(self, n: int, lim: int, move_to: Callable[[T], T]) -> None:
17    self.move_to = move_to
18    self.n = n
19    self.lim = lim
20    self.log = lim.bit_length()
21    self._build()
22
23  def _build(self) -> None:
24    db = [[0]*self.n for _ in range(self.log+1)]
25    for i in range(self.n):
26      db[0][i] = self.move_to(i)
27    for k in range(self.log):
28      for i in range(self.n):
29        db[k+1][i] = db[k][db[k][i]]
30    self.db = db
31
32  def kth(self, start: T, k: int) -> T:
33    """``start`` から ``k`` 個進んだ状態を返します。
34    :math:`O(\\log{k})` です。
35
36    Args:
37      start (T): スタートの状態です。
38      k (int): 遷移関数を適用する回数です。
39    """
40    now = start
41    for i in range(self.log):
42      if k & 1:
43        now = self.db[i][now]
44      k >>= 1
45    return now
