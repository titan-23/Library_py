 20
 21    Args:
 22        n_or_a (Union[Iterable[int], int]): `n_or_a` が `int` のとき、初期値 `0` 、長さ `n` で構築します。
 23                                            `n_or_a` が `Iterable` のとき、初期値 `a` で構築します。
 24    """
 25    if isinstance(n_or_a, int):
 26      self._size = n_or_a
 27      self._tree = [0] * (self._size + 1)
 28    else:
 29      a = n_or_a if isinstance(n_or_a, list) else list(n_or_a)

 31      _tree = [0] + a
 32      for i in range(1, _size):
 33        if i + (i & -i) <= _size:
 34          _tree[i + (i & -i)] += _tree[i]

 36      self._tree = _tree

181  def __init__(self, _used: Union[int, Iterable[T]], _a: Iterable[T]=[], compress=True, _multi=False):
182    self._len = 0
183    self._to_origin = list(range(_used)) if isinstance(_used, int) else sorted(set(_used))
184    self._to_zaatsu: Dict[T, int] = {key: i for i, key in enumerate(self._to_origin)} if compress else self._to_origin
185    self._size = len(self._to_origin)
186    self._cnt = [0] * self._size
187    _a = list(_a)
188    if _a:
189      a_ = [0] * self._size
190      if _multi:
191        self._len = len(_a)
192        for v in _a:
193          i = self._to_zaatsu[v]
194          a_[i] += 1
195          self._cnt[i] += 1
196      else:
197        for v in _a:
198          i = self._to_zaatsu[v]
199          if self._cnt[i] == 0:
200            self._len += 1
201            a_[i] = 1
202            self._cnt[i] = 1
203      self._fw = FenwickTree(a_)
204    else:
205      self._fw = FenwickTree(self._size)
206
207  def add(self, key: T) -> bool:
208    i = self._to_zaatsu[key]
209    if self._cnt[i]:
210      return False
211    self._len += 1
212    self._cnt[i] = 1
213    self._fw.add(i, 1)
214    return True
215
216  def remove(self, key: T) -> None:
217    if not self.discard(key):
218      raise KeyError(key)
219
220  def discard(self, key: T) -> bool:
221    i = self._to_zaatsu[key]
222    if self._cnt[i]:
223      self._len -= 1
224      self._cnt[i] = 0
225      self._fw.add(i, -1)
226      return True
227    return False
228
229  def le(self, key: T) -> Optional[T]:
230    i = self._to_zaatsu[key]
231    if self._cnt[i]: return key
232    pref = self._fw.pref(i) - 1
233    return None if pref < 0 else self._to_origin[self._fw.bisect_right(pref)]
234
235  def lt(self, key: T) -> Optional[T]:
236    pref = self._fw.pref(self._to_zaatsu[key]) - 1
237    return None if pref < 0 else self._to_origin[self._fw.bisect_right(pref)]
238
239  def ge(self, key: T) -> Optional[T]:
240    i = self._to_zaatsu[key]
241    if self._cnt[i]: return key
242    pref = self._fw.pref(i + 1)
243    return None if pref >= self._len else self._to_origin[self._fw.bisect_right(pref)]
244
245  def gt(self, key: T) -> Optional[T]:
246    pref = self._fw.pref(self._to_zaatsu[key] + 1)
247    return None if pref >= self._len else self._to_origin[self._fw.bisect_right(pref)]

249  def index(self, key: T) -> int:

252  def index_right(self, key: T) -> int:

255  def pop(self, k: int=-1) -> T:
256    assert -self._len <= k < self._len, \
