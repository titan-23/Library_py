 10from typing import Generic, TypeVar, Iterable
 11T = TypeVar('T', bound=SupportsLessThan)
 12
 13class SparseTableRmQ(Generic[T]):
 14  """
 15  2項演算を :math:`\\min` にしたものです。
 16  """
 17
 18  def __init__(self, a: Iterable[T], e: T):
 19    if not isinstance(a, list):
 20      a = list(a)
 21    self.size = len(a)
 22    log = self.size.bit_length()-1
 23    data = [a] + [[]] * log
 24    for i in range(log):
 25      pre = data[i]
 26      l = 1 << i
 27      data[i+1] = [pre[j] if pre[j] < pre[j+l] else pre[j+l] for j in range(len(pre)-l)]
 28    self.data = data
 29    self.e = e
 30
 31  def prod(self, l: int, r: int) -> T:
 32    assert 0 <= l <= r <= self.size
 33    if l == r: return self.e
 34    u = (r-l).bit_length()-1
 35    return self.data[u][l] if self.data[u][l] < self.data[u][r-(1<<u)] else self.data[u][r-(1<<u)]
 36
 37  def __getitem__(self, k: int) -> T:
 38    assert 0 <= k < self.size
 39    return self.data[0][k]
 40
 41  def __len__(self):
 42    return self.size
 43
 44  def __str__(self):
 45    return str(self.data[0])
 46
 47  def __repr__(self):
 48    return f'{self.__class__.__name__}({self.data[0]}, {self.e})'

 50from typing import List
 51
 52class LCA():
 53  """LCA を定数倍良く求めます。
 54
 55  :math:`O(< O(NlogN), O(1) >)`
 56  https://github.com/cheran-senthil/PyRival/blob/master/pyrival/graphs/lca.py
 57  """
 58
 59  def __init__(self, G: List[List[int]], root: int) -> None:
 60    """根が ``root`` の重み無し隣接リスト ``G`` で表されるグラフに対して LCA を求めます。
 61    時間・空間 :math:`O(n\\log{n})` です。
 62
 63    Args:
 64        G (List[List[int]]): 隣接リストです。
 65        root (int): 根です。
 66    """
 67    _n = len(G)
 68    path = [-1] * _n
 69    nodein = [-1] * _n
 70    par = [-1] * _n
 71    curtime = -1
 72    stack = [root]
 73    while stack:
 74      v = stack.pop()
 75      path[curtime] = par[v]
 76      curtime += 1
 77      nodein[v] = curtime
 78      for x in G[v]:
 79        if nodein[x] != -1:
 80          continue
 81        par[x] = v
 82        stack.append(x)
 83    self._n = _n
 84    self._path = path
 85    self._nodein = nodein
 86    self._st: SparseTableRmQ[int] = SparseTableRmQ((nodein[v] for v in path), e=_n)
 87
 88  def lca(self, u: int, v: int) -> int:
 89    """頂点 ``u`` と頂点 ``v`` の ``lca`` を返します。
 90    :math:`O(1)` です。
 91    """
 92    if u == v:
 93      return u
 94    l, r = self._nodein[u], self._nodein[v]
 95    if l > r:
 96      l, r = r, l
 97    return self._path[self._st.prod(l, r)]
 98
 99  def lca_mul(self, a: List[int]) -> int:
100    if all(a[i] == a[i+1] for i in range(len(a)-1)):
101      return a[0]
102    l = self._n+1
103    r = -l
104    for e in a:
105      e = self._nodein[e]
106      if l > e: l = e
107      if r < e: r = e
108    return self._path[self._st.prod(l, r)]
