 5def cartesian_tree(a: List[int]) -> Tuple[List[int], List[int], List[int]]:
 6  '''Get cartesian_tree. / O(N)'''
 7  n = len(a)
 8  par = [-1] * n
 9  left = [-1] * n
10  right = [-1] * n
11  path = newlist_hint(n)
12  for i, aa in enumerate(a):
13    pre = -1
14    while path and aa < a[path[-1]]:
15      pre = path.pop()
16    if pre != -1:
17      par[pre] = i
18    if path:
19      par[i] = path[-1]
20    path.append(i)
21  for i, p in enumerate(par):
22    if p == -1:
23      continue
24    if i < p:
25      left[p] = i
26    else:
27      right[p] = i
28  return par, left, right
