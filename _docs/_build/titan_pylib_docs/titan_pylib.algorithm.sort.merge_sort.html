 1from supports_less_than import SupportsLessThan
 2from typing import Iterable, TypeVar, Callable, List
 3from __pypy__ import newlist_hint
 4T = TypeVar('T', bound=SupportsLessThan)
 5
 6def merge_sort(a: Iterable[T], key: Callable[[T, T], bool]=lambda s, t: s < t) -> List[T]:
 7  """マージソートです。
 8
 9  非破壊的です。
10  最悪 :math:`O(n\\log{n})` 時間です。
11
12  Args:
13    a (Iterable[T]): ソートする列です。
14    key (Callable[[T, T], bool], optional): 比較関数 `key` にしたがって比較演算をします。
15                                            (第1引数)<(第2引数) のとき、 ``True`` を返すようにしてください。
16  """
17  def _sort(a: List[T]) -> List[T]:
18    n = len(a)
19    if n <= 1:
20      return a
21    if n == 2:
22      if not key(a[0], a[1]):
23        a[0], a[1] = a[1], a[0]
24      return a
25    left = _sort(a[:n//2])
26    right = _sort(a[n//2:])
27    res = newlist_hint(n)
28    i, j, l, r = 0, 0, len(left), len(right)
29    while i < l and j < r:
30      if key(left[i], right[j]):
31        res.append(left[i])
32        i += 1
33      else:
34        res.append(right[j])
35        j += 1
36    for i in range(i, l):
37      res.append(left[i])
38    for j in range(j, r):
39      res.append(right[j])
40    return res
41  return _sort(list(a))
