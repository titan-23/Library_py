250        if node.child[1].len_key() > self._m//2:
251          cnode = node.child[1]
252          node.child[0].append_key(node.key[0])
253          node.key[0] = cnode.pop_key(0)
254          if cnode.child:
255            node.child[0].append_child(cnode.pop_child(0))
256          node = node.child[0]
257          continue
258        cnode = self._merge(node, 0)
259        if node is self._root and not node.key:
260          self._root = cnode
261        node = cnode
262        continue
263      node = node.child[0]
264    v = node.pop_key(0)
265    self._update_stack(stack, v)
266    return v

268  def _merge(self, node: 'BTreeList._Node', i: int) -> 'BTreeList._Node':
269    y = node.child[i]
270    z = node.pop_child(i+1, size=0)
271    v = node.pop_key(i)
272    y.append_key(v)
273    node.sum += v
274    node.size += 1
275    y.extend_key(z.key)
276    y.extend_child(z.child)
277    return y
278
279  def _merge_key(self, node: 'BTreeList._Node', i: int) -> None:
280    if node.child[i].len_key() > self._m//2:
281      node.key[i] = self._discard_right(node.child[i])
282      return
283    if node.child[i+1].len_key() > self._m//2:
284      node.key[i] = self._discard_left(node.child[i+1])
285      return
286    indx = node.child[i if i+1 < len(node.child) else (i-1)].size
287    y = self._merge(node, i)
288    self._pop(indx, y)
289    if node is self._root and not node.key:
290      self._root = y
291
292  def _update_stack(self, stack, key):
293    for s in stack:
294      s.size -= 1
295      s.sum -= key

297  def tolist(self) -> List[T]:
298    a = newlist_hint(len(self))
299    def dfs(node):
300      if not node.child:
301        a.extend(node.key)
302        return
303      dfs(node.child[0])
304      for i in range(node.len_key()):
305        a.append(node.key[i])
306        dfs(node.child[i+1])
307    dfs(self._root)
308    return a

327  def pop(self, k: int) -> T:
328    assert 0 <= k < len(self), f'IndexError'
329    self._len -= 1
330    return self._pop(k, self._root)

332  def _pop(self, k: int, node: 'BTreeList._Node') -> T:
333    stack = []
334    while True:
335      if node.is_leaf():
336        v = node.pop_key(k)
337        self._update_stack(stack, v)
338        return v
339      stack.append(node)
340      for i in range(node.len_key()+1):
341        if k < node.child[i].size:
342          break
343        k -= node.child[i].size
344        if k == 0 and i < node.len_key():
345          v = node.key[i]
346          self._merge_key(node, i)
347          self._update_stack(stack, v)
348          return v
349        k -= 1
350      if node.child[i].len_key() == self._m//2:
351        if i+1 < len(node.child) and node.child[i+1].len_key() > self._m//2:
352          cnode = node.child[i+1]
353          node.child[i].append_key(node.key[i])
354          node.key[i] = cnode.pop_key(0)
355          if cnode.child:
356            node.child[i].append_child(cnode.pop_child(0))
357          node = node.child[i]
358          continue
359        if i-1 >= 0 and node.child[i-1].len_key() > self._m//2:
360          cnode = node.child[i-1]
361          k += 1
362          if cnode.child:
363            k += cnode.child[-1].size
364          node.child[i].insert_key(0, node.key[i-1])
365          node.key[i-1] = cnode.pop_key()
366          if cnode.child:
367            node.child[i].insert_child(0, cnode.pop_child())
368          node = node.child[i]
369          continue
370        if i+1 >= len(node.child):
371          i -= 1
372          k += node.child[i].size + 1
373        cnode = self._merge(node, i)
374        if node is self._root and not node.key:
375          self._root = cnode
376        node = cnode
377        continue
378      node = node.child[i]
