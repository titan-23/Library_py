  6class DynamicFenwickTree():
  7  """必要なところだけノードを作ります。
  8  """
  9
 10  def __init__(self, u: int):
 11    '''Build DynamicFenwickTree [0, u).'''
 12    assert isinstance(u, int), \
 13        f'TypeError: DynamicFenwickTree({u}), {u} must be int'
 14    self._u: Final[int] = u
 15    self._tree: Dict[int, int] = {}
 16    self._s: Final[int] = 1 << (u-1).bit_length()
 17
 18  def add(self, k: int, x: int) -> None:
 19    assert 0 <= k < self._u, \
 20        f'IndexError: DynamicFenwickTree.add({k}, {x}), u={self._u}'
 21    k += 1
 22    while k <= self._u:
 23      if k in self._tree:
 24        self._tree[k] += x
 25      else:
 26        self._tree[k] = x
 27      k += k & -k
 28
 29  def pref(self, r: int) -> int:
 30    assert 0 <= r <= self._u, \
 31        f'IndexError: DynamicFenwickTree.pref({r}), u={self._u}'
 32    ret = 0
 33    while r > 0:
 34      ret += self._tree.get(r, 0)
 35      r -= r & -r
 36    return ret
 37
 38  def sum(self, l: int, r: int) -> int:
 39    assert 0 <= l <= r <= self._u, \
 40        f'IndexError: DynamicFenwickTree.sum({l}, {r}), u={self._u}'
 41    # return self.pref(r) - self.pref(l)
 42    _tree = self._tree
 43    res = 0
 44    while r > l:
 45      res += _tree.get(r, 0)
 46      r -= r & -r
 47    while l > r:
 48      res -= _tree.get(l, 0)
 49      l -= l & -l
 50    return res
 51
 52  def bisect_left(self, w: int) -> Optional[int]:
 53    i, s = 0, self._s
 54    while s:
 55      if i+s <= self._u:
 56        if i+s in self._tree and self._tree[i+s] < w:
 57          w -= self._tree[i+s]
 58          i += s
 59        elif i+s not in self._tree and 0 < w:
 60          i += s
 61      s >>= 1
 62    return i if w else None
 63
 64  def bisect_right(self, w: int) -> int:
 65    i, s = 0, self._s
 66    while s:
 67      if i+s <= self._u:
 68        if i+s in self._tree and self._tree[i+s] <= w:
 69          w -= self._tree[i+s]
 70          i += s
 71        elif i+s not in self._tree and 0 <= w:
 72          i += s
 73      s >>= 1
 74    return i
 75
 76  def __str__(self):
 77    return str(self._tree)

 85  def __init__(self, u: int, a: Iterable[int]=[]):
 86    self._size: int = u
 87    self._len: int = 0
 88    self._cnt: Dict[int, int] = {}
 89    self._fw = DynamicFenwickTree(self._size)
 90    for _a in a:
 91      self.add(_a)
 92
 93  def add(self, key: int) -> bool:
 94    if key in self._cnt:
 95      return False
 96    self._len += 1
 97    self._cnt[key] = 1
 98    self._fw.add(key, 1)
 99    return True
100
101  def remove(self, key: int) -> None:
102    if self.discard(key): return
103    raise KeyError(key)
104
105  def discard(self, key: int) -> bool:
106    if key in self._cnt:
107      self._len -= 1
108      del self._cnt[key]
109      self._fw.add(key, -1)
110      return True
111    return False
112
113  def le(self, key: int) -> Optional[int]:
114    if key in self._cnt: return key
115    pref = self._fw.pref(key) - 1
116    return None if pref < 0 else self._fw.bisect_right(pref)
117
118  def lt(self, key: int) -> Optional[int]:
119    pref = self._fw.pref(key) - 1
120    return None if pref < 0 else self._fw.bisect_right(pref)
121
122  def ge(self, key: int) -> Optional[int]:
123    if key in self._cnt: return key
124    pref = self._fw.pref(key + 1)
125    return None if pref >= self._len else self._fw.bisect_right(pref)

127  def gt(self, key: int) -> Optional[int]:
128    pref = self._fw.pref(key + 1)

187class DynamicFenwickTreeMultiset(DynamicFenwickTreeSet):
188
189  def __init__(self, n: int, a: Iterable[int]=[]) -> None:
190    super().__init__(n, a)
191
192  def add(self, key: int, val: int=1) -> None:
193    self._len += val
194    if key in self._cnt:
195      self._cnt[key] += val
196    else:
197      self._cnt[key] = val
198    self._fw.add(key, val)
199
200  def discard(self, key: int, val: int=1) -> bool:
201    if key not in self._cnt:
202      return False
203    cnt = self._cnt[key]
204    if val >= cnt:
205      self._len -= cnt
206      del self._cnt[key]
207      self._fw.add(key, -cnt)
208    else:
209      self._len -= val
210      self._cnt[key] -= val
211      self._fw.add(key, -val)
212    return True
213
214  def discard_all(self, key: int) -> bool:
215    return self.discard(key, val=self.count(key))
216
217  def count(self, key: int) -> int:
218    return self._cnt[key]
219
220  def items(self) -> Iterable[Tuple[int, int]]:
221    _iter = 0
222    while _iter < self._len:
223      res = self.__getitem__(_iter)
224      cnt = self.count(res)
225      _iter += cnt
226      yield res, cnt
227
228  def show(self) -> None:
229    print('{' + ', '.join(f'{i[0]}: {i[1]}' for i in self.items()) + '}')
