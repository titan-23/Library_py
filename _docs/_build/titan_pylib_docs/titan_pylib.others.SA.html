 21  def _xor(self) -> int:
 22    t = (self._x ^ ((self._x << 11) & 0xFFFFFFFF)) & 0xFFFFFFFF
 23    self._x, self._y, self._z = self._y, self._z, self._w
 24    self._w = (self._w ^ (self._w >> 19)) ^ (t ^ ((t >> 8))&0xFFFFFFFF) & 0xFFFFFFFF
 25    return self._w

 27  def random(self) -> float:
 28    """random
 29    0以上1以下の一様ランダムな値を1つ生成して返すはずです。
 30    """
 31    return self._xor() / 0xFFFFFFFF

 33  def randint(self, begin: int, end: int) -> int:
 34    """``begin`` 以上 ``end`` **以下** のランダムな整数を返します。
 35    """
 36    assert begin <= end
 37    return begin + self._xor() % (end - begin + 1)
 38
 39  def randrange(self, begin: int, end: int) -> int:
 40    """``begin`` 以上 ``end`` **未満** のランダムな整数を返します。
 41    """
 42    assert begin < end
 43    return begin + self._xor() % (end - begin)

 45  def shuffle(self, a: List[Any]) -> None:
 46    """インプレースにシャッフルします。
 47
 48    :math:`O(n)` です。
 49
 50    Args:
 51      a (List[Any]): ``a`` をシャッフルします。
 52    """
 53    n = len(a)
 54    for i in range(n-1):
 55      j = self.randrange(i, n)
 56      a[i], a[j] = a[j], a[i]

102      arg = new_score - score
