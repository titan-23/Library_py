 65class SegmentTree(SegmentTreeInterface, Generic[T]):
 66  """セグ木です。非再帰です。
 67  """
 68
 69  def __init__(self,
 70               n_or_a: Union[int, Iterable[T]],
 71               op: Callable[[T, T], T],
 72               e: T) -> None:
 73    """``SegmentTree`` を構築します。
 74    :math:`O(n)` です。
 75
 76    Args:
 77      n_or_a (Union[int, Iterable[T]]): ``n: int`` のとき、 ``e`` を初期値として長さ ``n`` の ``SegmentTree`` を構築します。
 78                                        ``a: Iterable[T]`` のとき、 ``a`` から ``SegmentTree`` を構築します。
 79      op (Callable[[T, T], T]): 2項演算の関数です。
 80      e (T): 単位元です。
 81    """
 82    self._op = op
 83    self._e = e
 84    if isinstance(n_or_a, int):
 85      self._n = n_or_a
 86      self._log = (self._n - 1).bit_length()
 87      self._size = 1 << self._log
 88      self._data = [e] * (self._size << 1)
 89    else:
 90      n_or_a = list(n_or_a)
 91      self._n = len(n_or_a)
 92      self._log = (self._n - 1).bit_length()
 93      self._size = 1 << self._log
 94      _data = [e] * (self._size << 1)
 95      _data[self._size:self._size+self._n] = n_or_a
 96      for i in range(self._size-1, 0, -1):
 97        _data[i] = op(_data[i<<1], _data[i<<1|1])
 98      self._data = _data
 99
100  def set(self, k: int, v: T) -> None:
101    """一点更新です。
102    :math:`O(\\log{n})` です。
103
104    Args:
105      k (int): 更新するインデックスです。
106      v (T): 更新する値です。
107    """
108    assert -self._n <= k < self._n, \
109        f'IndexError: {self.__class__.__name__}.set({k}, {v}), n={self._n}'
110    if k < 0:
111      k += self._n
112    k += self._size
113    self._data[k] = v
114    for _ in range(self._log):
115      k >>= 1
116      self._data[k] = self._op(self._data[k<<1], self._data[k<<1|1])
117
118  def get(self, k: int) -> T:
119    """一点取得です。
120    :math:`O(1)` です。
121
122    Args:
123      k (int): インデックスです。
124    """
125    assert -self._n <= k < self._n, \
126        f'IndexError: {self.__class__.__name__}.get({k}), n={self._n}'
127    if k < 0:
128      k += self._n
129    return self._data[k+self._size]
130
131  def prod(self, l: int, r: int) -> T:
132    """区間 ``[l, r)`` の総積を返します。
133    :math:`O(\\log{n})` です。
134
135    Args:
136      l (int): インデックスです。
137      r (int): インデックスです。
138    """
139    assert 0 <= l <= r <= self._n, \
140        f'IndexError: {self.__class__.__name__}.prod({l}, {r})'
141    l += self._size
142    r += self._size
143    lres = self._e
144    rres = self._e
145    while l < r:
146      if l & 1:
147        lres = self._op(lres, self._data[l])
148        l += 1
149      if r & 1:
150        rres = self._op(self._data[r^1], rres)
151      l >>= 1
152      r >>= 1
153    return self._op(lres, rres)
154
155  def all_prod(self) -> T:
156    """区間 ``[0, n)`` の総積を返します。
157    :math:`O(1)` です。
158    """
159    return self._data[1]

161  def max_right(self, l: int, f: Callable[[T], bool]) -> int:
162    '''Find the largest index R s.t. f([l, R)) == True. / O(logN)'''
163    assert 0 <= l <= self._n, \
164        f'IndexError: {self.__class__.__name__}.max_right({l}, f) index out of range'
165    assert f(self._e), \
166        f'{self.__class__.__name__}.max_right({l}, f), f({self._e}) must be true.'
167    if l == self._n:
168      return self._n
169    l += self._size
170    s = self._e
171    while True:
172      while l & 1 == 0:
173        l >>= 1
174      if not f(self._op(s, self._data[l])):
175        while l < self._size:
176          l <<= 1
177          if f(self._op(s, self._data[l])):
178            s = self._op(s, self._data[l])
179            l |= 1
180        return l - self._size
181      s = self._op(s, self._data[l])
182      l += 1
183      if l & -l == l:
184        break
185    return self._n
186
187  def min_left(self, r: int, f: Callable[[T], bool]) -> int:
188    '''Find the smallest index L s.t. f([L, r)) == True. / O(logN)'''
189    assert 0 <= r <= self._n, \
190        f'IndexError: {self.__class__.__name__}.min_left({r}, f) index out of range'
191    assert f(self._e), \
192        f'{self.__class__.__name__}.min_left({r}, f), f({self._e}) must be true.'
193    if r == 0:
194      return 0
195    r += self._size
196    s = self._e
197    while True:
198      r -= 1
199      while r > 1 and r & 1:
200        r >>= 1
201      if not f(self._op(self._data[r], s)):
202        while r < self._size:
203          r = r << 1 | 1
204          if f(self._op(self._data[r], s)):
205            s = self._op(self._data[r], s)
206            r ^= 1
207        return r + 1 - self._size
208      s = self._op(self._data[r], s)
209      if r & -r == r:
210        break
211    return 0

246_titan_pylib_HashString_MOD: Final[int] = (1 << 61) - 1
247_titan_pylib_HashString_DIC: Final[Dict[str, int]] = {c: i for i, c in enumerate(string.ascii_lowercase, 1)}
248_titan_pylib_HashString_MASK30: Final[int] = (1 << 30) - 1
249_titan_pylib_HashString_MASK31: Final[int] = (1 << 31) - 1
250_titan_pylib_HashString_MASK61: Final[int] = _titan_pylib_HashString_MOD

252class HashStringBase():
253  """HashStringのベースクラスです。
254  """
255
256  def __init__(self, n: int, base: int=-1, seed: Optional[int]=None) -> None:
257    """
258    :math:`O(n)` です。
259
260    Args:
261        n (int): 文字列の長さの上限です。
262        base (int, optional): Defaults to -1.
263        seed (Optional[int], optional): Defaults to None.
264    """
265    random.seed(seed)
266    base = random.randint(37, 10**9) if base < 0 else base
267    powb = [1] * (n+1)
268    invb = [1] * (n+1)
269    invbpow = pow(base, -1, _titan_pylib_HashString_MOD)
270    for i in range(1, n+1):
271      powb[i] = HashStringBase.get_mul(powb[i-1], base)
272      invb[i] = HashStringBase.get_mul(invb[i-1], invbpow)
273    self.n = n
274    self.powb = powb
275    self.invb = invb
276
277  @staticmethod
278  def get_mul(a: int, b: int) -> int:
279    au = a >> 31
280    ad = a & _titan_pylib_HashString_MASK31
281    bu = b >> 31
282    bd = b & _titan_pylib_HashString_MASK31
283    mid = ad * bu + au * bd
284    midu = mid >> 30
285    midd = mid & _titan_pylib_HashString_MASK30
286    return HashStringBase.get_mod(au * bu * 2 + midu + (midd << 31) + ad * bd)
287
288  @staticmethod
289  def get_mod(x: int) -> int:
290    xu = x >> 61
291    xd = x & _titan_pylib_HashString_MASK61
292    res = xu + xd
293    if res >= _titan_pylib_HashString_MOD:
294      res -= _titan_pylib_HashString_MOD
295    return res
296
297  def unite(self, h1: int, h2: int, k: int) -> int:
298    # len(h2) == k
299    # h1 <- h2
300    return self.get_mod(self.get_mul(h1, self.powb[k]) + h2)

302class HashString():
303
304  def __init__(self, hsb: HashStringBase, s: str, update: bool=False) -> None:
305    """ロリハを構築します。
306
307    :math:`O(n)` です。
308
309    Args:
310        hsb (HashStringBase): ベースクラスです。
311        s (str): ロリハを構築する文字列です。
312        update (bool, optional): ``update=True`` のとき、1点更新が可能になります。
313    """
314    n = len(s)
315    data = [0] * n
316    acc = [0] * (n+1)
317    powb = hsb.powb
318    for i, c in enumerate(s):
319      data[i] = hsb.get_mul(powb[n-i-1], _titan_pylib_HashString_DIC[c])
320      acc[i+1] = hsb.get_mod(acc[i] + data[i])
321    self.hsb = hsb
322    self.n = n
323    self.acc = acc
324    self.used_seg = False
325    if update:
326      self.seg = SegmentTree(data, lambda s, t: (s+t)%_titan_pylib_HashString_MOD, 0)
327
328  def get(self, l: int, r: int) -> int:
329    """``s[l, r)`` のハッシュ値を返します。
330
331    1点更新処理後は :math:`O(\\log{n})` 、そうでなければ :math:`O(1)` です。
332
333    Args:
334      l (int): インデックスです。
335      r (int): インデックスです。
336
337    Returns:
338      int: ハッシュ値です。
339    """
340    if self.used_seg:
341      return self.hsb.get_mul(self.seg.prod(l, r), self.hsb.invb[self.n-r])
342    return self.hsb.get_mul(self.hsb.get_mod(self.acc[r]-self.acc[l]), self.hsb.invb[self.n-r])
343
344  def __getitem__(self, k: int) -> int:
345    """``s[k]`` のハッシュ値を返します。
346
347    1点更新処理後は :math:`O(\\log{n})` 、そうでなければ :math:`O(1)` です。
348
349    Args:
350      k (int): インデックスです。
351
352    Returns:
353      int: ハッシュ値です。
354    """
355    return self.get(k, k+1)

357  def set(self, k: int, c: str) -> None:
358    """`k` 番目の文字を `c` に更新します。
359
360    :math:`O(\\log{n})` です。また、今後の ``get()`` が :math:`O(\\log{n})` になります。
361
362    Args:
363      k (int): インデックスです。
364      c (str): 更新する文字です。
365    """
366    self.used_seg = True
367    self.seg[k] = self.hsb.get_mul(self.hsb.powb[self.n-k-1], _titan_pylib_HashString_DIC[c])
368
369  def __setitem__(self, k: int, c: str) -> None:
370    return self.set(k, c)
371
372  def __len__(self):
373    return self.n
374
375  def get_lcp(self) -> List[int]:
376    """lcp配列を返します。
377
378    :math:`O(n\\log{n})` です。
379    """
380    a = [0] * self.n
381    memo = [-1] * (self.n+1)
382    for i in range(self.n):
383      ok, ng = 0, self.n-i+1
384      while ng - ok > 1:
385        mid = (ok + ng) >> 1
386        if memo[mid] == -1:
387          memo[mid] = self.get(0, mid)
388        if memo[mid] == self.get(i, i+mid):
389          ok = mid
390        else:
391          ng = mid
392      a[i] = ok
393    return a

398class MultiHashStringBase():

400  def __init__(self, n: int, base_cnt: int=1, base_list: List[int]=[], seed: Optional[int]=None) -> None:
401    if seed is None:
402      seed = random.randint(0, 10**9)
403    assert base_cnt > 0, f'ValueError: {self.__class__.__name__} base_cnt must be > 0'
404    base_list = base_list if len(base_list) == base_cnt else [random.randint(37, 10**9) for _ in range(base_cnt)]
405    hsb = tuple(HashStringBase(n, base_list[i]) for i in range(base_cnt))
406    self.hsb = hsb
