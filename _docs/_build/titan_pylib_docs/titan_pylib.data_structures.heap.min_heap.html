23    a = self.a
24    n = len(a)
25    while i*2+1 < n:
26      u, v = i*2+1, i*2+2
27      if v < n and a[u] > a[v]:
28        u = v
29      if a[i] > a[u]:
30        a[i], a[u] = a[u], a[i]
31        i = u
32      else:
33        break
