 53class BitVector(BitVectorInterface):
 54  """コンパクトな bit vector です。
 55  """
 56
 57  def __init__(self, n: int):
 58    """長さ ``n`` の ``BitVector`` です。
 59
 60    bit を保持するのに ``array[I]`` を使用します。
 61    ``block_size= n / 32`` として、使用bitは ``32*block_size=2n bit`` です。
 62
 63    累積和を保持するのに同様の ``array[I]`` を使用します。
 64    32bitごとの和を保存しています。同様に使用bitは ``2n bit`` です。
 65    """
 66    assert 0 <= n < 4294967295
 67    self.N = n
 68    self.block_size = (n + 31) >> 5
 69    b = bytes(4*(self.block_size+1))
 70    self.bit = array('I', b)
 71    self.acc = array('I', b)
 72
 73  @staticmethod
 74  def _popcount(x: int) -> int:
 75    x = x - ((x >> 1) & 0x55555555)
 76    x = (x & 0x33333333) + ((x >> 2) & 0x33333333)
 77    x = x + (x >> 4) & 0x0f0f0f0f
 78    x += x >> 8
 79    x += x >> 16
 80    return x & 0x0000007f
 81
 82  def set(self, k: int) -> None:
 83    """``k`` 番目の bit を ``1`` にします。
 84    :math:`O(1)` です。
 85
 86    Args:
 87      k (int): インデックスです。
 88    """
 89    self.bit[k>>5] |= 1 << (k & 31)
 90
 91  def build(self) -> None:
 92    """構築します。
 93    **これ以降 ``set`` メソッドを使用してはいけません。**
 94    :math:`O(n)` です。
 95    """
 96    acc, bit = self.acc, self.bit
 97    for i in range(self.block_size):
 98      acc[i+1] = acc[i] + BitVector._popcount(bit[i])
 99
100  def access(self, k: int) -> int:
101    """``k`` 番目の bit を返します。
102    :math:`O(1)` です。
103    """
104    return (self.bit[k >> 5] >> (k & 31)) & 1
105
106  def __getitem__(self, k: int) -> int:
107    return (self.bit[k >> 5] >> (k & 31)) & 1
108
109  def rank0(self, r: int) -> int:
110    """``a[0, r)`` に含まれる ``0`` の個数を返します。
111    :math:`O(1)` です。
112    """
113    return r - (self.acc[r>>5] + BitVector._popcount(self.bit[r>>5] & ((1 << (r & 31)) - 1)))
114
115  def rank1(self, r: int) -> int:
116    """``a[0, r)`` に含まれる ``1`` の個数を返します。
117    :math:`O(1)` です。
118    """
119    return self.acc[r>>5] + BitVector._popcount(self.bit[r>>5] & ((1 << (r & 31)) - 1))
120
121  def rank(self, r: int, v: int) -> int:
122    """``a[0, r)`` に含まれる ``v`` の個数を返します。
123    :math:`O(1)` です。
124    """
125    return self.rank1(r) if v else self.rank0(r)
126
127  def select0(self, k: int) -> int:
128    """``k`` 番目の ``0`` のインデックスを返します。
129    :math:`O(\\log{n})` です。
130    """
131    if k < 0 or self.rank0(self.N) <= k:
132      return -1
133    l, r = 0, self.block_size+1
134    while r - l > 1:
135      m = (l + r) >> 1
136      if m*32 - self.acc[m] > k:
137        r = m
138      else:
139        l = m
140    indx = 32 * l
141    k = k - (l*32 - self.acc[l]) + self.rank0(indx)
142    l, r = indx, indx+32
143    while r - l > 1:
144      m = (l + r) >> 1
145      if self.rank0(m) > k:
146        r = m
147      else:
148        l = m
149    return l
150
151  def select1(self, k: int) -> int:
152    """``k`` 番目の ``1`` のインデックスを返します。
153    :math:`O(\\log{n})` です。
154    """
155    if k < 0 or self.rank1(self.N) <= k:
156      return -1
157    l, r = 0, self.block_size+1
158    while r - l > 1:
159      m = (l + r) >> 1
160      if self.acc[m] > k:
161        r = m
162      else:
163        l = m
164    indx = 32 * l
165    k = k - self.acc[l] + self.rank1(indx)
166    l, r = indx, indx+32
167    while r - l > 1:
168      m = (l + r) >> 1
169      if self.rank1(m) > k:
170        r = m
171      else:
172        l = m
173    return l
174
175  def select(self, k: int, v: int) -> int:
176    """``k`` 番目の ``v`` のインデックスを返します。
177    :math:`O(\\log{n})` です。
178    """
179    return self.select1(k) if v else self.select0(k)
180
181  def __len__(self):
182    return self.N
183
184  def __str__(self):
185    return str([self.access(i) for i in range(self.N)])
186
187  def __repr__(self):
188    return f'{self.__class__.__name__}({self})'
