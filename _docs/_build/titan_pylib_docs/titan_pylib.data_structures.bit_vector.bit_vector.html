 52class BitVector(BitVectorInterface):
 53  """コンパクトな bit vector です。
 54  """
 55
 56  def __init__(self, n: int):
 57    """長さ ``n`` の ``BitVector`` です。
 58
 59    bit を保持するのに ``array[I]`` を使用します。
 60    ``block_size= n / 32`` として、使用bitは ``32*block_size=2n bit`` です。
 61
 62    累積和を保持するのに同様の ``array[I]`` を使用します。
 63    32bitごとの和を保存しています。同様に使用bitは ``2n bit`` です。
 64    """
 65    assert 0 <= n < 4294967295
 66    self.N = n
 67    self.block_size = (n + 31) >> 5
 68    b = bytes(4*(self.block_size+1))
 69    self.bit = array('I', b)
 70    self.acc = array('I', b)
 71
 72  @staticmethod
 73  def _popcount(x: int) -> int:
 74    x = x - ((x >> 1) & 0x55555555)
 75    x = (x & 0x33333333) + ((x >> 2) & 0x33333333)
 76    x = x + (x >> 4) & 0x0f0f0f0f
 77    x += x >> 8
 78    x += x >> 16
 79    return x & 0x0000007f
 80
 81  def set(self, k: int) -> None:
 82    """``k`` 番目の bit を ``1`` にします。
 83    :math:`O(1)` です。
 84
 85    Args:
 86      k (int): インデックスです。
 87    """
 88    self.bit[k>>5] |= 1 << (k & 31)
 89
 90  def build(self) -> None:
 91    """構築します。
 92    **これ以降 ``set`` メソッドを使用してはいけません。**
 93    :math:`O(n)` です。
 94    """
 95    acc, bit = self.acc, self.bit
 96    for i in range(self.block_size):
 97      acc[i+1] = acc[i] + BitVector._popcount(bit[i])
 98
 99  def access(self, k: int) -> int:
100    """``k`` 番目の bit を返します。
101    :math:`O(1)` です。
102    """
103    return (self.bit[k >> 5] >> (k & 31)) & 1
104
105  def __getitem__(self, k: int) -> int:
106    return (self.bit[k >> 5] >> (k & 31)) & 1
107
108  def rank0(self, r: int) -> int:
109    """``a[0, r)`` に含まれる ``0`` の個数を返します。
110    :math:`O(1)` です。
111    """
112    return r - (self.acc[r>>5] + BitVector._popcount(self.bit[r>>5] & ((1 << (r & 31)) - 1)))
113
114  def rank1(self, r: int) -> int:
115    """``a[0, r)`` に含まれる ``1`` の個数を返します。
116    :math:`O(1)` です。
117    """
118    return self.acc[r>>5] + BitVector._popcount(self.bit[r>>5] & ((1 << (r & 31)) - 1))
119
120  def rank(self, r: int, v: int) -> int:
121    """``a[0, r)`` に含まれる ``v`` の個数を返します。
122    :math:`O(1)` です。
123    """
124    return self.rank1(r) if v else self.rank0(r)
125
126  def select0(self, k: int) -> int:
127    """``k`` 番目の ``0`` のインデックスを返します。
128    :math:`O(\\log{n})` です。
129    """
130    if k < 0 or self.rank0(self.N) <= k:
131      return -1
132    l, r = 0, self.block_size+1
133    while r - l > 1:
134      m = (l + r) >> 1
135      if m*32 - self.acc[m] > k:
136        r = m
137      else:
138        l = m
139    indx = 32 * l
140    k = k - (l*32 - self.acc[l]) + self.rank0(indx)
141    l, r = indx, indx+32
142    while r - l > 1:
143      m = (l + r) >> 1
144      if self.rank0(m) > k:
145        r = m
146      else:
147        l = m
148    return l
149
150  def select1(self, k: int) -> int:
151    """``k`` 番目の ``1`` のインデックスを返します。
152    :math:`O(\\log{n})` です。
153    """
154    if k < 0 or self.rank1(self.N) <= k:
155      return -1
156    l, r = 0, self.block_size+1
157    while r - l > 1:
158      m = (l + r) >> 1
159      if self.acc[m] > k:
160        r = m
161      else:
162        l = m
163    indx = 32 * l
164    k = k - self.acc[l] + self.rank1(indx)
165    l, r = indx, indx+32
166    while r - l > 1:
167      m = (l + r) >> 1
168      if self.rank1(m) > k:
169        r = m
170      else:
171        l = m
172    return l
173
174  def select(self, k: int, v: int) -> int:
175    """``k`` 番目の ``v`` のインデックスを返します。
176    :math:`O(\\log{n})` です。
177    """
178    return self.select1(k) if v else self.select0(k)
179
180  def __len__(self):
181    return self.N
182
183  def __str__(self):
184    return str([self.access(i) for i in range(self.N)])
