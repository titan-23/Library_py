 4def bellman_ford(G: List[List[Tuple[int, int]]], s: int, inf: Union[int, float]=float('inf')) -> Optional[List[Union[int, float]]]:
 5  '''Return dist from s. / O(nm)'''
 6  n = len(G)
 7  dist = [inf] * n
 8  dist[s] = 0
 9  for _ in range(n):
10    update = 0
11    for v, e in enumerate(G):
12      for x, c in e:
13        if dist[v] != inf and dist[v] + c < dist[x]:
14          dist[x] = dist[v] + c
15          update = 1
16    if not update:
17      break
18  else:
19    return None  # NEGATIVE CYCLE
20  return dist
