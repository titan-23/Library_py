 63class SegmentTree(SegmentTreeInterface, Generic[T]):
 64  """セグ木です。非再帰です。
 65  """
 66
 67  def __init__(self,
 68               n_or_a: Union[int, Iterable[T]],
 69               op: Callable[[T, T], T],
 70               e: T) -> None:
 71    """``SegmentTree`` を構築します。
 72    :math:`O(n)` です。
 73
 74    Args:
 75      n_or_a (Union[int, Iterable[T]]): ``n: int`` のとき、 ``e`` を初期値として長さ ``n`` の ``SegmentTree`` を構築します。
 76                                        ``a: Iterable[T]`` のとき、 ``a`` から ``SegmentTree`` を構築します。
 77      op (Callable[[T, T], T]): 2項演算の関数です。
 78      e (T): 単位元です。
 79    """
 80    self._op = op
 81    self._e = e
 82    if isinstance(n_or_a, int):
 83      self._n = n_or_a
 84      self._log = (self._n - 1).bit_length()
 85      self._size = 1 << self._log
 86      self._data = [e] * (self._size << 1)
 87    else:
 88      n_or_a = list(n_or_a)
 89      self._n = len(n_or_a)
 90      self._log = (self._n - 1).bit_length()
 91      self._size = 1 << self._log
 92      _data = [e] * (self._size << 1)
 93      _data[self._size:self._size+self._n] = n_or_a
 94      for i in range(self._size-1, 0, -1):
 95        _data[i] = op(_data[i<<1], _data[i<<1|1])
 96      self._data = _data

 98  def set(self, k: int, v: T) -> None:
 99    """一点更新です。
100    :math:`O(\\log{n})` です。
101
102    Args:
103      k (int): 更新するインデックスです。
104      v (T): 更新する値です。
105
106    制約:
107      :math:`-n \\leq n \\leq k < n`
108    """
109    assert -self._n <= k < self._n, \
110        f'IndexError: {self.__class__.__name__}.set({k}, {v}), n={self._n}'
111    if k < 0:
112      k += self._n
113    k += self._size
114    self._data[k] = v
115    for _ in range(self._log):
116      k >>= 1
117      self._data[k] = self._op(self._data[k<<1], self._data[k<<1|1])

119  def get(self, k: int) -> T:
120    """一点取得です。
121    :math:`O(1)` です。
122
123    Args:
124      k (int): インデックスです。
125
126    制約:
127      :math:`-n \\leq n \\leq k < n`
128    """
129    assert -self._n <= k < self._n, \
130        f'IndexError: {self.__class__.__name__}.get({k}), n={self._n}'
131    if k < 0:
132      k += self._n
133    return self._data[k+self._size]

135  def prod(self, l: int, r: int) -> T:
136    """区間 ``[l, r)`` の総積を返します。
137    :math:`O(\\log{n})` です。
138
139    Args:
140      l (int): インデックスです。
141      r (int): インデックスです。
142
143    制約:
144      :math:`0 \\leq l \\leq r \\leq n`
145    """
146    assert 0 <= l <= r <= self._n, \
147        f'IndexError: {self.__class__.__name__}.prod({l}, {r})'
148    l += self._size
149    r += self._size
150    lres = self._e
151    rres = self._e
152    while l < r:
153      if l & 1:
154        lres = self._op(lres, self._data[l])
155        l += 1
156      if r & 1:
157        rres = self._op(self._data[r^1], rres)
158      l >>= 1
159      r >>= 1
160    return self._op(lres, rres)

162  def all_prod(self) -> T:
163    """区間 ``[0, n)`` の総積を返します。
164    :math:`O(1)` です。
165    """
166    return self._data[1]

168  def max_right(self, l: int, f: Callable[[T], bool]) -> int:
169    '''Find the largest index R s.t. f([l, R)) == True. / O(\\log{n})'''
170    assert 0 <= l <= self._n, \
171        f'IndexError: {self.__class__.__name__}.max_right({l}, f) index out of range'
172    # assert f(self._e), \
173    #     f'{self.__class__.__name__}.max_right({l}, f), f({self._e}) must be true.'
174    if l == self._n:
175      return self._n
176    l += self._size
177    s = self._e
178    while True:
179      while l & 1 == 0:
180        l >>= 1
181      if not f(self._op(s, self._data[l])):
182        while l < self._size:
183          l <<= 1
184          if f(self._op(s, self._data[l])):
185            s = self._op(s, self._data[l])
186            l |= 1
187        return l - self._size
188      s = self._op(s, self._data[l])
189      l += 1
190      if l & -l == l:
191        break
192    return self._n
193
194  def min_left(self, r: int, f: Callable[[T], bool]) -> int:
195    '''Find the smallest index L s.t. f([L, r)) == True. / O(\\log{n})'''
196    assert 0 <= r <= self._n, \
197        f'IndexError: {self.__class__.__name__}.min_left({r}, f) index out of range'
198    # assert f(self._e), \
199    #     f'{self.__class__.__name__}.min_left({r}, f), f({self._e}) must be true.'
200    if r == 0:
201      return 0
202    r += self._size
203    s = self._e
204    while True:
205      r -= 1
206      while r > 1 and r & 1:
207        r >>= 1
208      if not f(self._op(self._data[r], s)):
209        while r < self._size:
210          r = r << 1 | 1
211          if f(self._op(self._data[r], s)):
212            s = self._op(self._data[r], s)
213            r ^= 1
214        return r + 1 - self._size
215      s = self._op(self._data[r], s)
216      if r & -r == r:
217        break
218    return 0

254  def __init__(self, G: List[List[int]], root: int):
255    """``root`` を根とする木 ``G`` を HLD します。
256    :math:`O(n)` です。
257
258    Args:
259      G (List[List[int]]): 木を表す隣接リストです。
260      root (int): 根です。
261    """
262    n = len(G)
263    self.n: int = n
264    self.G: List[List[int]] = G
265    self.size: List[int] = [1] * n
266    self.par: List[int] = [-1] * n
267    self.dep: List[int] = [-1] * n
268    self.nodein: List[int] = [0] * n
269    self.nodeout: List[int] = [0] * n
270    self.head: List[int] = [0] * n
271    self.hld: List[int] = [0] * n
272    self._dfs(root)
273
274  def _dfs(self, root: int) -> None:
275    dep, par, size, G = self.dep, self.par, self.size, self.G
276    dep[root] = 0
277    stack = [root]
278    while stack:
279      v = stack.pop()
280      if v >= 0:
281        dep_nxt = dep[v] + 1
282        for x in G[v]:
283          if dep[x] != -1:
284            continue
285          dep[x] = dep_nxt
286          stack.append(~x)
287          stack.append(x)
288      else:
289        v = ~v
290        G_v, dep_v = G[v], dep[v]
291        for i, x in enumerate(G_v):
292          if dep[x] < dep_v:
293            par[v] = x
294            continue
295          size[v] += size[x]
296          if size[x] > size[G_v[0]]:
297            G_v[0], G_v[i] = G_v[i], G_v[0]
298
299    head, nodein, nodeout, hld = self.head, self.nodein, self.nodeout, self.hld
300    curtime = 0
301    stack = [~root, root]
302    while stack:
303      v = stack.pop()
304      if v >= 0:
305        if par[v] == -1:
306          head[v] = v
307        nodein[v] = curtime
308        hld[curtime] = v
309        curtime += 1
310        if not G[v]:
311          continue
312        G_v0 = G[v][0]
313        for x in reversed(G[v]):
314          if x == par[v]:
315            continue
316          head[x] = head[v] if x == G_v0 else x
317          stack.append(~x)
318          stack.append(x)
319      else:
320        nodeout[~v] = curtime

334  def for_each_vertex(self, u: int, v: int) -> Iterator[Tuple[int, int]]:
335    """``u-v`` パスに対応する区間のインデックスを返します。
336    :math:`O(\\log{n})` です。
337    """
338    head, nodein, dep, par = self.head, self.nodein, self.dep, self.par
339    while head[u] != head[v]:
340      if dep[head[u]] < dep[head[v]]:
341        u, v = v, u
342      yield nodein[head[u]], nodein[u]+1
343      u = par[head[u]]
344    if dep[u] < dep[v]:
345      u, v = v, u
346    yield nodein[v], nodein[u]+1

348  def for_each_vertex_subtree(self, v: int) -> Iterator[Tuple[int, int]]:
349    """頂点 ``v`` の部分木に対応する区間のインデックスを返します。
350    :math:`O(1)` です。

354  def path_kth_elm(self, s: int, t: int, k: int) -> int:
355    """``s`` から ``t`` に向かって ``k`` 個進んだ頂点のインデックスを返します。
356    存在しないときは ``-1`` を返します。
357    :math:`O(\\log{n})` です。
358    """
359    head, dep, par = self.head, self.dep, self.par
360    lca = self.lca(s, t)
361    d = dep[s] + dep[t] - 2*dep[lca]
362    if d < k:
363      return -1
364    if dep[s] - dep[lca] < k:
365      s = t
366      k = d - k
367    hs = head[s]
368    while dep[s] - dep[hs] < k:
369      k -= dep[s] - dep[hs] + 1
370      s = par[hs]
371      hs = head[s]
372    return self.hld[self.nodein[s] - k]

374  def lca(self, u: int, v: int) -> int:
375    """``u``, ``v`` の LCA を返します。
376    :math:`O(\\log{n})` です。
377    """
378    nodein, head, par = self.nodein, self.head, self.par
379    while True:
380      if nodein[u] > nodein[v]:
381        u, v = v, u
382      if head[u] == head[v]:
383        return u
384      v = par[head[v]]

389class HLDNoncommutativeSegmentTree(Generic[T]):
390  """セグ木搭載HLDです。
391

393    **非可換に対応してます。**
394  """
395
396  def __init__(self, hld: HLD, n_or_a: Union[int, Iterable[T]], op: Callable[[T, T], T], e: T):
397    self.hld: HLD = hld
398    a = [e] * n_or_a if isinstance(n_or_a, int) else self.hld.build_list(list(n_or_a))
399    self.seg: SegmentTree[T] = SegmentTree(a, op, e)
400    self.rseg: SegmentTree[T] = SegmentTree(a[::-1], op, e)
401    self.op: Callable[[T, T], T] = op
402    self.e: T = e
403
404  def path_prod(self, u: int, v: int) -> T:
405    """頂点 ``u`` から頂点 ``v`` へのパスの集約値を返します。
406    :math:`O(\\log^2{n})` です。
407
408    Args:
409      u (int): パスの **始点** です。
410      v (int): パスの **終点** です。
411
412    Returns:
413      T: 求める集約値です。
414    """
415    head, nodein, dep, par, n = self.hld.head, self.hld.nodein, self.hld.dep, self.hld.par, self.hld.n
416    lres, rres = self.e, self.e
417    seg, rseg = self.seg, self.rseg
418    while head[u] != head[v]:
419      if dep[head[u]] > dep[head[v]]:
420        lres = self.op(lres, rseg.prod(n-nodein[u]-1, n-nodein[head[u]]))
421        u = par[head[u]]
422      else:
423        rres = self.op(seg.prod(nodein[head[v]], nodein[v]+1), rres)
424        v = par[head[v]]
425    if dep[u] > dep[v]:
426      lres = self.op(lres, rseg.prod(n-nodein[u]-1, n-nodein[v]))
427    else:
428      lres = self.op(lres, seg.prod(nodein[u], nodein[v]+1))
429    return self.op(lres, rres)
430
431  def get(self, k: int) -> T:
432    """頂点の値を返します。
433    :math:`O(\\log{n})` です。
434
435    Args:
436      k (int): 頂点のインデックスです。
437
438    Returns:
439      T: 頂点の値です。
440    """
441    return self.seg[self.hld.nodein[k]]

443  def set(self, k: int, v: T) -> None:
444    """頂点の値を更新します。
445    :math:`O(\\log{n})` です。

448      k (int): 頂点のインデックスです。
449      v (T): 更新する値です。
