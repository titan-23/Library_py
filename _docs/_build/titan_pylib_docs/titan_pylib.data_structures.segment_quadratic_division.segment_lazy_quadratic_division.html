  8class SegmentLazyQuadraticDivision(Generic[T, F]):
  9  """
 10  区間の総積取得・区間への作用適用クエリをそれぞれ時間計算量 :math:`O(\\sqrt{n})` で処理できるデータ構造です。
 11  定数倍が軽いのでそこまで遅くはないはずです。
 12  """
 13
 14  def __init__(self,
 15               n_or_a: Union[int, Iterable[T]],
 16               op: Callable[[T, T], T],
 17               mapping: Callable[[F, T], T],
 18               composition: Callable[[F, F], F],
 19               e: T,
 20               id: F) -> None:
 21    """
 22    引数は遅延セグ木のアレです。
 23
 24    :math:`O(n)` です。
 25    """
 26    if isinstance(n_or_a, int):
 27      self.n = n_or_a
 28      a = [e] * self.n
 29    else:
 30      if not isinstance(n_or_a, list):
 31        a = list(n_or_a)
 32      else:
 33        a = n_or_a
 34    self.n = len(a)
 35    self.op = op
 36    self.mapping = mapping
 37    self.composition = composition
 38    self.e = e
 39    self.id = id
 40    self.bucket_size = int(self.n**.5) + 1
 41    self.bucket_cnt = (self.n+self.bucket_size-1) // self.bucket_size
 42    self.data = [a[k*self.bucket_size:(k+1)*self.bucket_size] for k in range(self.bucket_cnt)]
 43    self.bucket_data = [reduce(self.op, v) for v in self.data]
 44    self.bucket_lazy = [id] * self.bucket_cnt
 45
 46  def apply(self, l: int, r: int, f: F) -> None:
 47    """区間 ``[l, r)`` に ``f`` を作用します。
 48
 49    :math:`O(\\sqrt{n})` です。
 50    """
 51    assert 0 <= l <= r <= self.n
 52    def _change_data(k: int, l: int, r: int) -> None:
 53      self._propagate(k)
 54      d = self.data[k]
 55      for i in range(l, r):
 56        d[i] = self.mapping(f, d[i])
 57      self.bucket_data[k] = reduce(self.op, self.data[k])
 58    k1 = l // self.bucket_size
 59    k2 = r // self.bucket_size
 60    l -= k1 * self.bucket_size
 61    r -= k2 * self.bucket_size
 62    if k1 == k2:
 63      if k1 < self.bucket_cnt:
 64        _change_data(k1, l, r)
 65    else:
 66      if k1 < self.bucket_cnt:
 67        if l == 0:
 68          self.bucket_lazy[k1] = f if self.bucket_lazy[k1] == self.id else self.composition(f, self.bucket_lazy[k1])
 69          self.bucket_data[k1] = self.mapping(f, self.bucket_data[k1])
 70        else:
 71          _change_data(k1, l, len(self.data[k1]))
 72      for i in range(k1+1, k2):
 73        self.bucket_lazy[i] = f if self.bucket_lazy[i] == self.id else self.composition(f, self.bucket_lazy[i])
 74        self.bucket_data[i] = self.mapping(f, self.bucket_data[i])
 75      if k2 < self.bucket_cnt:
 76        if r == len(self.data[k2]):
 77          self.bucket_lazy[k2] = f if self.bucket_lazy[k2] == self.id else self.composition(f, self.bucket_lazy[k2])
 78          self.bucket_data[k2] = self.mapping(f, self.bucket_data[k2])
 79        else:
 80          _change_data(k2, 0, r)
 81
 82  def all_apply(self, f: F) -> None:
 83    """区間 ``[0, n)`` に ``f`` を作用します。
 84
 85    :math:`O(\\sqrt{n})` です。
 86    """
 87    self.bucket_lazy = [f if bl == self.id else self.composition(f, bl) for bl in self.bucket_lazy]
 88
 89  def _propagate(self, k: int) -> None:
 90    if self.bucket_lazy[k] == self.id: return
 91    f = self.bucket_lazy[k]
 92    dk = self.data[k]
 93    for i, d in enumerate(dk):
 94      dk[i] = self.mapping(f, d)
 95    self.bucket_lazy[k] = self.id
 96
 97  def _all_propagatae(self) -> None:
 98    for k in range(self.bucket_cnt):
 99      self._propagate(k)
100    for i in range(self.bucket_cnt):
101      self.bucket_lazy[i] = self.id
102
103  def prod(self, l: int, r: int) -> T:
104    """区間 ``[l, r)`` の総積を返します。
105
106    :math:`O(\\sqrt{n})` です。
107
108    Args:
109      l (int): インデックスです。
110      r (int): インデックスです。
111    """
112    assert 0 <= l <= r <= self.n
113    if l == r: return self.e
114    k1 = l // self.bucket_size
115    k2 = r // self.bucket_size
116    l -= k1 * self.bucket_size
117    r -= k2 * self.bucket_size
118    s = self.e
119    if k1 == k2:
120      s = reduce(self.op, self.data[k1][l:r])
121      if self.bucket_lazy[k1] != self.id:
122        s = self.mapping(self.bucket_lazy[k1], s)
123    else:
124      if l < len(self.data[k1]):
125        s = reduce(self.op, self.data[k1][l:])
126        if self.bucket_lazy[k1] != self.id:
127          s = self.mapping(self.bucket_lazy[k1], s)
128      if k1+1 < k2:
129        s = reduce(self.op, self.bucket_data[k1+1:k2], s)
130      if k2 < self.bucket_cnt and r > 0:
131        s_ = reduce(self.op, self.data[k2][:r])
132        if self.bucket_lazy[k2] != self.id:
133          s_ = self.mapping(self.bucket_lazy[k2], s_)
134        s = self.op(s, s_)
135    return s
136
137  def all_prod(self) -> T:
138    """区間 ``[0, n)`` の総積を返します。
139
140    :math:`O(\\sqrt{n})` です。
141    """
142    return reduce(self.op, self.bucket_data)
143
144  def tolist(self) -> List[T]:
145    self._all_propagatae()
146    return list(chain(*self.data))
147
148  def __getitem__(self, k: int) -> T:
149    p = k // self.bucket_size
150    return self.data[p][k-p*self.bucket_size] if self.bucket_lazy[p] == self.id else self.mapping(self.bucket_lazy[p], self.data[p][k-p*self.bucket_size])
151
152  def __setitem__(self, k, v):
153    p = k // self.bucket_size
154    self._propagate(p)
155    self.data[p][k-p*self.bucket_size] = v
156    self.bucket_data[p] = reduce(self.op, self.data[p])
157
158  def __str__(self):
159    return str(self.tolist())
