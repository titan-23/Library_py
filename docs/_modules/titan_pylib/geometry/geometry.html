186    def area(self) -> float:
187        """面積を求める / `O(n)`
188
189        Returns:
190            float: 面積
191        """
192        res = 0
193        p = self.ps
194        for i in range(self.n - 1):
195            res += Geometry.cross(p[i], p[i + 1])
196        res += Geometry.cross(p[-1], p[0])
197        return res / 2

199    def is_convex(self) -> bool:
200        """凸多角形かどうか / `O(n)`
201
202        Returns:
203            bool:
204        """
205        ps = self.ps
206        for i in range(self.n):
207            pre = (i - 1 + self.n) % self.n
208            now = i
209            nxt = (i + 1) % self.n
210            if Geometry.ccw(ps[pre], ps[now], ps[nxt]) == -1:
211                return False
212        return True

217    def contains(self, p):
218        """点の包含関係を返す / O(n)
219
220        Args:
221            p (Point): 点
222
223        Returns:
224            int: `2`: `p` を含む
225                 `1`: `p` が辺上にある
226                 `0`: それ以外
227        """
228        ps = self.ps
229        deg = 0
230        for i in range(self.n):
231            if ps[i] == ps[(i + 1) % self.n]:
232                continue
233            if GeometryUtil.eq(
234                Geometry.dist_p_to_segmemt(p, Segment(ps[i], ps[(i + 1) % self.n])), 0
235            ):
236                return 1
237            ax = ps[i].x - p.x
238            ay = ps[i].y - p.y
239            bx = ps[(i + 1) % self.n].x - p.x
240            by = ps[(i + 1) % self.n].y - p.y
241            cos = (ax * bx + ay * by) / (
242                GeometryUtil.sqrt((ax * ax + ay * ay) * (bx * bx + by * by))
243            )
244            cos = min(1, max(-1, cos))
245            deg += self.get_degree(GeometryUtil.acos(cos))
246        return 2 if GeometryUtil.eq(deg, 360) else 0

255    def diameter(self) -> tuple[float, int, int]:
256        ps = self.ps
257        n = len(ps)
258        if n == 0:
259            return -1, -1, -1
260        if n == 1:
261            return 0, ps[0], ps[0]
262        if n == 2:
263            return abs(ps[0] - ps[1]), ps[0], ps[1]
264        u, v = 0, 0
265        up, vp = None, None
266        for i in range(n):
267            if ps[u] > ps[i]:
268                u = i
269                up = ps[i]
270            if ps[v] < ps[i]:
271                v = i
272                vp = ps[i]
273        d = 0
274        su, sv = u, v
275        loop = False
276        while (u != su or v != sv) or (not loop):
277            loop = True
278            if GeometryUtil.lt(d, abs(ps[u] - ps[v])):
279                d = abs(ps[u] - ps[v])
280                up = ps[u]
281                vp = ps[v]
282            if GeometryUtil.lt(
283                Geometry.cross(ps[(u + 1) % n] - ps[u], ps[(v + 1) % n] - ps[v]), 0
284            ):
285                u = (u + 1) % n
286            else:
287                v = (v + 1) % n
288        return d, up, vp

290    def contains(self, p: Point) -> int:
291        """点の包含関係を返す / O(n)
292
293        Args:
294            p (Point): 点
295
296        Returns:
297            int: `2`: `p` を含む
298                 `1`: `p` が辺上にある
299                 `0`: それ以外
300        """
301        ps = self.ps
302        n = len(ps)
303        b1 = Geometry.cross(ps[1] - ps[0], p - ps[0])
304        b2 = Geometry.cross(ps[-1] - ps[0], p - ps[0])
305        if GeometryUtil.lt(b1, 0) or GeometryUtil.gt(b2, 0):
306            return 0
307        l, r = 1, n - 1
308        while r - l > 1:
309            mid = (l + r) >> 1
310            c = Geometry.cross(p - ps[0], ps[mid] - ps[0])
311            if GeometryUtil.eq(c, 0) or GeometryUtil.gt(c, 0):
312                r = mid
313            else:
314                l = mid
315        b3 = Geometry.cross(ps[l] - p, ps[r] - p)
316        if GeometryUtil.eq(b3, 0):
317            return 1
318        if GeometryUtil.gt(b3, 0):
319            if GeometryUtil.eq(b1, 0) or GeometryUtil.eq(b2, 0):
320                return 1
321            else:
322                return 2
323        return 0

373        """u->vに対し、v->pの位置関係を求める
374
375        Args:
376            u (Point):
377            v (Point):
378            p (Point):
379
380        Returns:
381            int: `+1`: a->bに対し、b->cが半時計回りに進む
382                 `-1`: a->bに対し、b->cが時計回りに進む
383                 `+2`: a->bに対し、b->cが逆方向に進む
384                 `-2`: a->bに対し、b->cが同一方向に進む
385                 ` 0`: cが線分a->b上にある
386        """
387        if cls.cross(v - u, p - u) > GeometryUtil.GEO_EPS:
388            return 1
389        if cls.cross(v - u, p - u) < -GeometryUtil.GEO_EPS:
390            return -1
391        if cls.dot(v - u, p - u) < -GeometryUtil.GEO_EPS:
392            return 2
393        if GeometryUtil.lt(abs(v - u), abs(p - u)):
394            # if abs(v - u) + GeometryUtil.GEO_EPS < abs(p - u):
395            return -2
396        return 0

477                 `0`: 内包している
478                 `1`: 内接している
479                 `2`: 2点で交わっている
480                 `3`: 外接している
481                 `4`: 離れている
482        """
483        d = abs(c1.p - c2.p)
484        if GeometryUtil.gt(d, c1.r + c2.r):
485            # if d > c1.r + c2.r + GeometryUtil.GEO_EPS:
486            return 4
487        if GeometryUtil.eq(d, c1.r + c2.r):
488            return 3
489        if GeometryUtil.eq(d, abs(c1.r - c2.r)):
490            return 1
491        if GeometryUtil.lt(d, abs(c1.r - c2.r)):
492            # if d < abs(c1.r - c2.r) - GeometryUtil.GEO_EPS:
493            return 0
494        return 2

506    @classmethod
507    def cross_point_circle(cls, c1: Circle, c2: Circle) -> tuple[Point, ...]:
508        r = cls.is_intersect_circle(c1, c2)
509        if r == 4 or r == 0:
510            return ()
511        if r == 3:  # 外接
512            t = c1.r / (c1.r + c2.r)
513            return (c1.p + (c2.p - c1.p) * t,)
514        d = abs(c1.p - c2.p)
515        if r == 1:  # 内接
516            if GeometryUtil.lt(c2.r, c1.r):
517                # if c2.r < c1.r - GeometryUtil.GEO_EPS:
518                return (c1.p + (c2.p - c1.p) * (c1.r / d),)
519            else:
520                return (c2.p + (c1.p - c2.p) * (c2.r / d),)
521        # 交点が2つ
522        rcos = (c1.r * c1.r + d * d - c2.r * c2.r) / (2 * d)
523        rsin = GeometryUtil.sqrt(c1.r * c1.r - rcos * rcos)
524        if GeometryUtil.lt(c1.r - abs(rcos), 0):
525            # if c1.r - abs(rcos) < GeometryUtil.GEO_EPS:
526            rsin = 0
527        e12 = (c2.p - c1.p) / abs(c2.p - c1.p)
528        return tuple(
529            sorted(
530                [
531                    c1.p + rcos * e12 + rsin * Point(-e12.y, e12.x),
532                    c1.p + rcos * e12 + rsin * Point(e12.y, -e12.x),
533                ]
534            )
535        )

538    def cross_point_circle_line(cls, c: Circle, l: Line) -> list[Point]:
539        res = []
540        d = cls.dist_p_to_line(c.p, l)
541        if d > c.r + GeometryUtil.GEO_EPS:
542            return res
543        h = cls.projection_point(c.p, l)
544        if GeometryUtil.eq(d, c.r):
545            res.append(h)
546            return res
547        e = (l.p2 - l.p1).unit()
548        ph = GeometryUtil.sqrt(c.r * c.r - d * d)
549        res.append(h - e * ph)
550        res.append(h + e * ph)
551        res.sort()
552        return res

569            return 0.0

585    @classmethod
586    def triangle_circumcircle(cls, p1: Point, p2: Point, p3: Point) -> Circle:
587        a = 2 * (p1.x - p2.x)
588        b = 2 * (p1.y - p2.y)
589        p = p1.x**2 - p2.x**2 + p1.y**2 - p2.y**2
590        c = 2 * (p1.x - p3.x)
591        d = 2 * (p1.y - p3.y)
592        q = p1.x**2 - p3.x**2 + p1.y**2 - p3.y**2
593        u = a * d - b * c
594        x = d * p - b * q
595        y = a * q - c * p
596        if GeometryUtil.lt(u, 0):
597            x = -x
598            y = -y
599            u = -u
600        x /= u
601        y /= u
602        r = GeometryUtil.sqrt((x - p1.x) ** 2 + (y - p1.y) ** 2)
603        return Circle(Point(x, y), r)

605    @classmethod
606    def triangle_excircle(
607        cls, p1: Point, p2: Point, p3: Point
608    ) -> tuple[Circle, Circle, Circle]:
609        dx1 = p2.x - p1.x
610        dy1 = p2.y - p1.y
611        dx2 = p3.x - p1.x
612        dy2 = p3.y - p1.y
613        d1 = GeometryUtil.sqrt((p3.x - p2.x) ** 2 + (p3.y - p2.y) ** 2)
614        d3 = GeometryUtil.sqrt(dx1**2 + dy1**2)
615        d2 = GeometryUtil.sqrt(dx2**2 + dy2**2)
616        S2 = abs(dx1 * dy2 - dx2 * dy1)
617        dsum1 = -d1 + d2 + d3
618        r1 = S2 / dsum1
619        ex1 = (-p1.x * d1 + p2.x * d2 + p3.x * d3) / dsum1
620        ey1 = (-p1.y * d1 + p2.y * d2 + p3.y * d3) / dsum1
621
622        dsum2 = d1 - d2 + d3
623        r2 = S2 / dsum2
624        ex2 = (p1.x * d1 - p2.x * d2 + p3.x * d3) / dsum2
625        ey2 = (p1.y * d1 - p2.y * d2 + p3.y * d3) / dsum2
626
627        dsum3 = d1 + d2 - d3
628        r3 = S2 / dsum3
629        ex3 = (p1.x * d1 + p2.x * d2 - p3.x * d3) / dsum3
630        ey3 = (p1.y * d1 + p2.y * d2 - p3.y * d3) / dsum3
631
632        return (
633            Circle(Point(ex1, ey1), r1),
634            Circle(Point(ex2, ey2), r2),
635            Circle(Point(ex3, ey3), r3),
636        )

644    @classmethod
645    def circle_common_tangent(cls, c1: Circle, c2: Circle) -> list[Line]:
646        ret = []
647        d = abs(c1.p - c2.p)
648        if GeometryUtil.eq(d, 0):
649            return ret
650        u = (c2.p - c1.p).unit()
651        v = u.rotate(GeometryUtil.pi / 2)
652        for s in [-1, 1]:
653            h = (c1.r + c2.r * s) / d
654            if GeometryUtil.eq(h * h, 1):
655                ret.append(
656                    Line.from_points(
657                        c1.p + (u if h > 0 else -u) * c1.r,
658                        c1.p + (u if h > 0 else -u) * c1.r + v,
659                    )
660                )
661            elif 1 - h * h > 0:
662                U = u * h
663                V = v * GeometryUtil.sqrt(1 - h * h)
664                ret.append(
665                    Line.from_points(c1.p + (U + V) * c1.r, c2.p - (U + V) * (c2.r * s))
666                )
667                ret.append(
668                    Line.from_points(c1.p + (U - V) * c1.r, c2.p - (U - V) * (c2.r * s))
669                )
670        return ret

672    @classmethod
673    def convex_hull(cls, ps: list[Point], line_contains: bool = False) -> list[Point]:
674        """凸包を求める / `O(n)`
675
676        Args:
677            line_contains (bool, optional): 一直線上の点を含める場合、 `True` .
678
679        Returns:
680            list[Point]: 凸包
681        """
682        ps = cls.sort_by_y(ps)
683        if not line_contains:
684            ps = cls.unique(ps)
685        if len(ps) <= 1:
686            return ps
687
688        def cross(a, p):
689            if line_contains:
690                return cls.cross(a[-1] - a[-2], p - a[-1]) < -GeometryUtil.GEO_EPS
691            else:
692                return cls.cross(a[-1] - a[-2], p - a[-1]) < GeometryUtil.GEO_EPS
693
694        lower = []
695        for p in ps:
696            while len(lower) > 1 and cross(lower, p):
697                lower.pop()
698            lower.append(p)
699
700        upper = []
701        for p in reversed(ps):
702            while len(upper) > 1 and cross(upper, p):
703                upper.pop()
704            upper.append(p)
705
706        return lower[:-1] + upper[:-1]

731    def argumant_sort(cls, a: list[Point]) -> list[Point]:
732        upper = []
733        lower = []
734        zeros = []
735        for p in a:
736            if GeometryUtil.lt(p.y, 0) or (
737                GeometryUtil.eq(p.y, 0) and GeometryUtil.lt(p.x, 0)
738            ):
739                lower.append(p)
740            elif GeometryUtil.eq(p.x, 0) and GeometryUtil.eq(p.y, 0):
741                zeros.append(p)
742            else:
743                upper.append(p)
744
745        def cmp(s: Point, t: Point) -> bool:
746            return GeometryUtil.lt(s.y * t.x, s.x * t.y) or GeometryUtil.eq(
747                s.y * t.x, s.x * t.y
748            )
749
750        upper = merge_sort(upper, key=cmp)
751        lower = merge_sort(lower, key=cmp)
752        return lower + zeros + upper

774        p = ConvexPolygon(a)

786    @classmethod
787    def closest_pair(cls, a: list[Point]) -> tuple[float, int, int]:
788        """最近点対を求める / O(nlogn)
789
790        Args:
791            a (list[Point]): 距離配列
792
793        Returns:
794            tuple[float, int, int]: 距離、最近点対
795        """
796
797        buff = [None] * len(a)
798        d, p, q = GeometryUtil.GEO_INF, -1, -1
799
800        def f(l: int, r: int) -> None:
801            nonlocal d, p, q
802            mid = (l + r) // 2
803            x = b[mid][0].x
804            if mid - l > 1:
805                f(l, mid)
806            if r - mid > 1:
807                f(mid, r)
808            i, j = l, mid
809            indx = 0
810            while i < mid and j < r:
811                if cls._cmp_y(b[i][0], b[j][0]):
812                    buff[indx] = b[i]
813                    i += 1
814                else:
815                    buff[indx] = b[j]
816                    j += 1
817                indx += 1
818            for k in range(i, mid):
819                b[l + indx + k - i] = b[k]
820            for k in range(l, l + indx):
821                b[k] = buff[k - l]
822
823            near_line = []
824            for i in range(l, r):
825                e_p, e_i = b[i]
826                v = abs(e_p.x - x)
827                if cls._gt(v, d) or GeometryUtil.eq(v, d):
828                    continue
829                for j in range(len(near_line) - 1, -1, -1):
830                    ne_p, ne_i = near_line[j]
831                    dx = e_p.x - ne_p.x
832                    dy = e_p.y - ne_p.y
833                    if cls._gt(dy, d) or GeometryUtil.eq(dy, d):
834                        break
835                    cand = GeometryUtil.sqrt(dx * dx + dy * dy)
836                    if GeometryUtil.lt(cand, d):
837                        d, p, q = cand, ne_i, e_i
838                near_line.append(b[i])
839
840        b = merge_sort((e, i) for i, e in enumerate(a))
841        f(0, len(b))
842        return d, p, q
