 6def get_articulation_points(G: List[List[int]]) -> List[int]:
 7  n = len(G)
 8  order = [-1] * n
 9  res: List[int] = []
10  cnt = 0
11
12  def antirec(func):
13    # 再帰用デコレータ
14    # yieldにするのを忘れないこと
15    # 参考: https://github.com/cheran-senthil/PyRival/blob/master/pyrival/misc/bootstrap.py
16    stack = []
17    def wrappedfunc(*args, **kwargs):
18      if stack:
19        return func(*args, **kwargs)
20      to = func(*args, **kwargs)
21      while True:
22        if isinstance(to, GeneratorType):
23          stack.append(to)
24          to = next(to)
25        else:
26          stack.pop()
27          if not stack:
28            break
29          to = stack[-1].send(to)
30      return to
31    return wrappedfunc
32
33  @antirec
34  def dfs(v: int, p: int) -> int:
35    nonlocal cnt
36    r_min = order[v] = cnt
37    fcnt = 0
38    p_art = 0
39    cnt += 1
40    for w in G[v]:
41      if w == p:
42        continue
43      if order[w] == -1:
44        ret = yield dfs(w, v)
45        p_art |= (order[v] <= ret)
46        r_min = min(r_min, ret)
47        fcnt += 1
48      else:
49        r_min = min(r_min, order[w])
50    p_art |= (r_min == order[v] and len(G[v]) > 1)
51    if (p == -1 and fcnt > 1) or (p != -1 and p_art):
52      res.append(v)
53    yield r_min
54  if G:
55    dfs(0, -1)
56  return res
