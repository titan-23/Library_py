 10  def _inv(cls, a: int) -> int:
 11    res = 1
 12    b = cls.self.mod - 2
 13    while b:
 14      if b & 1:
 15        res = res * a % cls.mod
 16      a = a * a % cls.mod
 17      b >>= 1
 18    return res
