 77    def _process_file(self, input_file: str) -> float:
 78        with open(input_file, "r", encoding="utf-8") as f:
 79            input_text = "".join(f.read())
 80        try:
 81            result = subprocess.run(
 82                self.execute_command,
 83                input=input_text,
 84                stderr=subprocess.PIPE,
 85                stdout=subprocess.PIPE,
 86                text=True,
 87                check=True,
 88            )
 89            score_line = result.stderr.rstrip().split("\n")[-1]
 90            _, score = score_line.split(" = ")
 91            score = float(score)
 92            if self.verbose:
 93                logger.info(f"{input_file}: {score=}")
 94            return score
 95        except Exception as e:
 96            logger.exception(e)
 97            logger.error(f"Error occured in {input_file}")
 98            raise ValueError(input_file)

100    def run(self) -> list[float]:
101        """"""
102        pool = multiprocessing.Pool(processes=self.cpu_count)
103        scores = pool.map(
104            partial(self._process_file), self.input_file_names, chunksize=1
105        )
106        pool.close()
107        pool.join()
108        return scores

165def main():
166    """ ``tester`` """
167    args = ParallelTester.get_args()
168    njobs = min(args.number_of_jobs, multiprocessing.cpu_count() - 1)
169    logger.info(f"{njobs=}")
170
171    tester = build_tester(AHCSettings, njobs, args.verbose)
172
173    if args.compile:
174        tester.compile()
175
176    logger.info("start.")
177
178    start = time.time()
179    scores = tester.run()
180    score = tester.show_score(scores)
181    logger.info(f"{time.time() - start:.4f}sec")
182    return score
