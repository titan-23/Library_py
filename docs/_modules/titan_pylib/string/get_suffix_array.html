 5def get_suffix_array(s: str, hs: HashString) -> List[int]:
 6  """suffix_arrayを求めます。
 7
 8  ロリハで大小比較をするため、比較関数に :math:`O(logn)` 、
 9  ソートが :math:`O(nlogn*(比較関数の計算慮))` で、
10  全体 :math:`O(nlog^2n)` です。
11
12  Args:
13      s (str): 文字列です。
14      hs (HashString): ``HashString`` です。
15
16  Returns:
17      List[int]: suffix_array です。
18  """
19  def cmp(u: int, v: int) -> bool:
20    ok, ng = 0, min(n-u, n-v)
21    while ng - ok > 1:
22      mid = (ok + ng) >> 1
23      if hs.get(u, u+mid) == hs.get(v, v+mid):
24        ok = mid
25      else:
26        ng = mid
27    return s[u+ok] < s[v+ok]
28  n = len(s)
29  return merge_sort(range(n), key=cmp)
