24  def unite(self, x: int, y: int, w: int) -> Optional[int]:
25    '''Untie x and y, weight[y] = weight[x] + w. / O(α(N))'''
26    rx = self.root(x)
27    ry = self.root(y)
28    if rx == ry:
29      return rx if self.diff(x, y) == w else None
30    w += self._weight[x] - self._weight[y]
31    self._group_numbers -= 1
32    if self._parents[rx] > self._parents[ry]:
33      rx, ry = ry, rx

35    self._parents[rx] += self._parents[ry]
36    self._parents[ry] = rx
37    self._weight[ry] = w
