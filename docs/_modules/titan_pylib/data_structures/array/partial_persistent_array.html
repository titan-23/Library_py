33  def get(self, k: int, t: int=-1) -> T:
34    """位置 ``k`` 、バージョン ``t`` の要素を返します。
35    :math:`O(\\log{n})` です。
36
37    Args:
38      k (int): インデックスです。
39      t (int, optional): バージョンです。デフォルトは最新バージョンです。
40    """
41    if t == -1 or t >= self.t[k][-1]:
42      return self.a[k][-1]
43    tk = self.t[k]
44    ok, ng = 0, len(tk)
45    while ng - ok > 1:
46      mid = (ok + ng) // 2
47      if tk[mid] <= t:
48        ok = mid
49      else:
50        ng = mid
51    return self.a[k][ok]
