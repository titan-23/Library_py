 14  class UndoableUnionFind():
 15    """内部で管理される `UndoableUnionFind` です。
 16    """
 17
 18    def __init__(self, n: int):
 19      self._n: int = n
 20      self._parents: List[int] = [-1] * n
 21      self._all_sum: List[int] = [0] * n
 22      self._one_sum: List[int] = [0] * n
 23      self._history: List[Tuple[int, int, int]] = []
 24      self._group_count: int = n
 25
 26    def _undo(self) -> None:
 27      assert self._history, 'UndoableUnionFind._undo() with non history'
 28      y, py, all_sum_y = self._history.pop()
 29      if y == -1:
 30        return
 31      x, px, all_sum_x = self._history.pop()
 32      self._group_count += 1
 33      self._parents[y] = py
 34      self._parents[x] = px
 35      s = (self._all_sum[x] - all_sum_y - all_sum_x) // (-py-px) * (-py)
 36      self._all_sum[y] += s
 37      self._all_sum[x] -= all_sum_y + s
 38      self._one_sum[x] -= self._one_sum[y]

 40    def root(self, x: int) -> int:
 41      """要素 ``x`` を含む集合の代表元を返します。
 42
 43      :math:`O(\\log{n})` です。
 44      """
 45      while self._parents[x] >= 0:
 46        x = self._parents[x]
 47      return x

 49    def unite(self, x: int, y: int) -> bool:
 50      """要素 ``x`` を含む集合と要素 ``y`` を含む集合を併合します。
 51
 52      :math:`O(\\log{n})` です。
 53
 54      Returns:
 55        bool: もともと同じ集合であれば ``False``、そうでなければ ``True`` を返します。
 56      """
 57      x = self.root(x)
 58      y = self.root(y)
 59      if x == y:
 60        self._history.append((-1, -1, -1))
 61        return False
 62      if self._parents[x] > self._parents[y]:
 63        x, y = y, x
 64      self._group_count -= 1
 65      self._history.append((x, self._parents[x], self._all_sum[x]))
 66      self._history.append((y, self._parents[y], self._all_sum[y]))
 67      self._all_sum[x] += self._all_sum[y]
 68      self._one_sum[x] += self._one_sum[y]
 69      self._parents[x] += self._parents[y]
 70      self._parents[y] = x
 71      return True

135  def __init__(self, n: int) -> None:
136    """初期状態を頂点数 ``n`` の無向グラフとします。
137

140    Args:
141      n (int): 頂点数です。
142    """
143    self._n = n
144    self._query_count = 0
145    self._bit = n.bit_length() + 1
146    self._msk = (1 << self._bit) - 1
147    self._start = defaultdict(lambda: [0, 0])
148    self._edge_data = []
149    self.uf = OfflineDynamicConnectivity.UndoableUnionFind(n)

151  def add_edge(self, u: int, v: int) -> None:
152    """辺 ``{u, v}`` を追加します。
153
154    :math:`O(1)` です。
155    """
156    assert 0 <= u < self._n and 0 <= v < self._n
157    if u > v:
158      u, v = v, u
159    edge = u<<self._bit|v
160    if self._start[edge][0] == 0:
161      self._start[edge][1] = self._query_count
162    self._start[edge][0] += 1

184  def run(self, out: Callable[[int], None]) -> None:
185    """実行します。
186
187    :math:`O(q \\log{q} \\log{n})` です。
188
189    Args:
190        out (Callable[[int], None]): クエリ番号 ``k`` を引数にとります。
191    """
192    # O(qlogqlogn)
193    uf, bit, msk, q = self.uf, self._bit, self._msk, self._query_count
194    log = (q - 1).bit_length()
195    size = 1 << log
196    size2 = size * 2
197    data = [[] for _ in range(size<<1)]
198
199    def add(l, r, edge):
200      l += size
201      r += size
202      while l < r:
203        if l & 1:
204          data[l].append(edge)
205          l += 1
206        if r & 1:
207          data[r^1].append(edge)
208        l >>= 1
209        r >>= 1
210
211    for edge, p in self._start.items():
212      if p[0] != 0:
213        add(p[1], self._query_count, edge)
214    for l, r, edge in self._edge_data:
215      add(l, r, edge)
216
217    todo = [1]
218    while todo:
219      v = todo.pop()
220      if v >= 0:
221        for uv in data[v]:
222          uf.unite(uv>>bit, uv&msk)
223        todo.append(~v)
224        if v<<1|1 < size2:
225          todo.append(v<<1|1)
226          todo.append(v<<1)
227        elif v - size < q:
228          out(v-size)
229      else:
230        for _ in data[~v]:
231          uf._undo()
