  6  """セグ木です。非再帰です。

  9  def __init__(self,
 10               n_or_a: Union[int, Iterable[T]],
 11               op: Callable[[T, T], T],
 12               e: T) -> None:
 13    """``SegmentTree`` を構築します。
 14    :math:`O(n)` です。
 15
 16    Args:
 17      n_or_a (Union[int, Iterable[T]]): ``n: int`` のとき、 ``e`` を初期値として長さ ``n`` の ``SegmentTree`` を構築します。
 18                                        ``a: Iterable[T]`` のとき、 ``a`` から ``SegmentTree`` を構築します。
 19      op (Callable[[T, T], T]): 2項演算の関数です。
 20      e (T): 単位元です。
 21    """
 22    self._op = op
 23    self._e = e
 24    if isinstance(n_or_a, int):
 25      self._n = n_or_a
 26      self._log = (self._n - 1).bit_length()
 27      self._size = 1 << self._log
 28      self._data = [e] * (self._size << 1)
 29    else:
 30      n_or_a = list(n_or_a)
 31      self._n = len(n_or_a)
 32      self._log = (self._n - 1).bit_length()
 33      self._size = 1 << self._log
 34      _data = [e] * (self._size << 1)
 35      _data[self._size:self._size+self._n] = n_or_a
 36      for i in range(self._size-1, 0, -1):
 37        _data[i] = op(_data[i<<1], _data[i<<1|1])
 38      self._data = _data

 40  def set(self, k: int, v: T) -> None:
 41    """一点更新です。
 42    :math:`O(\\log{n})` です。
 43
 44    Args:
 45      k (int): 更新するインデックスです。
 46      v (T): 更新する値です。
 47    """
 48    assert -self._n <= k < self._n, \
 49        f'IndexError: {self.__class__.__name__}.set({k}, {v}), n={self._n}'
 50    if k < 0:
 51      k += self._n
 52    k += self._size
 53    self._data[k] = v
 54    for _ in range(self._log):
 55      k >>= 1
 56      self._data[k] = self._op(self._data[k<<1], self._data[k<<1|1])

 71  def prod(self, l: int, r: int) -> T:
 72    """区間 ``[l, r)`` の総積を返します。
 73    :math:`O(\\log{n})` です。
 74
 75    Args:
 76      l (int): インデックスです。
 77      r (int): インデックスです。
 78    """
 79    assert 0 <= l <= r <= self._n, \
 80        f'IndexError: {self.__class__.__name__}.prod({l}, {r})'
 81    l += self._size
 82    r += self._size
 83    lres = self._e
 84    rres = self._e
 85    while l < r:
 86      if l & 1:
 87        lres = self._op(lres, self._data[l])
 88        l += 1
 89      if r & 1:
 90        rres = self._op(self._data[r^1], rres)
 91      l >>= 1
 92      r >>= 1
 93    return self._op(lres, rres)

101  def max_right(self, l: int, f: Callable[[T], bool]) -> int:
102    '''Find the largest index R s.t. f([l, R)) == True. / O(logN)'''
103    assert 0 <= l <= self._n, \
104        f'IndexError: {self.__class__.__name__}.max_right({l}, f) index out of range'
105    assert f(self._e), \
106        f'{self.__class__.__name__}.max_right({l}, f), f({self._e}) must be true.'
107    if l == self._n:
108      return self._n
109    l += self._size
110    s = self._e
111    while True:
112      while l & 1 == 0:
113        l >>= 1
114      if not f(self._op(s, self._data[l])):
115        while l < self._size:
116          l <<= 1
117          if f(self._op(s, self._data[l])):
118            s = self._op(s, self._data[l])
119            l |= 1
120        return l - self._size
121      s = self._op(s, self._data[l])
122      l += 1
123      if l & -l == l:
124        break
125    return self._n

127  def min_left(self, r: int, f: Callable[[T], bool]) -> int:
128    '''Find the smallest index L s.t. f([L, r)) == True. / O(logN)'''
129    assert 0 <= r <= self._n, \
130        f'IndexError: {self.__class__.__name__}.min_left({r}, f) index out of range'
131    assert f(self._e), \
132        f'{self.__class__.__name__}.min_left({r}, f), f({self._e}) must be true.'
133    if r == 0:
134      return 0
135    r += self._size
136    s = self._e
137    while True:
138      r -= 1
139      while r > 1 and r & 1:
140        r >>= 1
141      if not f(self._op(self._data[r], s)):
142        while r < self._size:
143          r = r << 1 | 1
144          if f(self._op(self._data[r], s)):
145            s = self._op(self._data[r], s)
146            r ^= 1
147        return r + 1 - self._size
148      s = self._op(self._data[r], s)
149      if r & -r == r:
150        break
151    return 0
