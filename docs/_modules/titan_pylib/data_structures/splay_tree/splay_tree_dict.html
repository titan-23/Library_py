 40  def _set_search_splay(self, key: T) -> None:
 41    node = self._root
 42    keys, child = self._keys, self._child
 43    if (not node) or keys[node] == key: return
 44    left, right = 0, 0
 45    while keys[node] != key:
 46      d = key > keys[node]
 47      if not child[node<<1|d]: break
 48      if (d and key > keys[child[node<<1|1]]) or (d^1 and key < keys[child[node<<1]]):
 49        new = child[node<<1|d]
 50        child[node<<1|d] = child[new<<1|(d^1)]
 51        child[new<<1|(d^1)] = node
 52        node = new
 53        if not child[node<<1|d]: break
 54      if d:
 55        child[left<<1|1] = node
 56        left = node
 57      else:
 58        child[right<<1] = node
 59        right = node
 60      node = child[node<<1|d]
 61    child[right<<1] = child[node<<1|1]
 62    child[left<<1|1] = child[node<<1]
 63    child[node<<1] = child[1]
 64    child[node<<1|1] = child[0]
 65    self._root = node

 67  def _get_min_splay(self, node: int) -> int:
 68    child = self._child
 69    if (not node) or (not child[node<<1]): return node
 70    right = 0
 71    while child[node<<1]:
 72      new = child[node<<1]
 73      child[node<<1] = child[new<<1|1]
 74      child[new<<1|1] = node
 75      if not child[new<<1]: break
 76      child[right<<1] = new
 77      right = new
 78      node = child[new<<1]
 79    child[right<<1] = child[node<<1|1]

 81    child[node<<1] = child[1]
