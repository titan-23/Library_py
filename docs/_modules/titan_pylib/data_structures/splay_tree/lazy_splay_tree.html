 12    def __init__(self, key: T, lazy: F) -> None:
 13      self.key: T = key
 14      self.data: T = key
 15      self.rdata: T = key
 16      self.lazy: F = lazy
 17      self.left: Optional['LazySplayTree._Node'] = None

 23  def __init__(self,
 24               n_or_a: Union[int, Iterable[T]],
 25               op: Callable[[T, T], T],
 26               mapping: Callable[[F, T], T],
 27               composition: Callable[[F, F], F],
 28               e: T,
 29               id: F,
 30               _root: Optional[_Node]=None,
 31               ) -> None:
 32    """構築します。
 33    :math:`O(n)` です。
 34
 35    Args:
 36      n_or_a (Union[int, Iterable[T]]): ``n`` のとき、 ``e`` から長さ ``n`` で構築します。
 37                                        ``a`` のとき、 ``a`` から構築します。
 38      op (Callable[[T, T], T]): 遅延セグ木のあれです。
 39      mapping (Callable[[F, T], T]): 遅延セグ木のあれです。
 40      composition (Callable[[F, F], F]): 遅延セグ木のあれです。
 41      e (T): 遅延セグ木のあれです。
 42      id (F): 遅延セグ木のあれです。
 43    """
 44    self.op = op
 45    self.mapping = mapping
 46    self.composition = composition
 47    self.e = e
 48    self.id = id
 49    self.root = _root
 50    if _root:
 51      return
 52    a = n_or_a
 53    if isinstance(a, int):
 54      if a > 0:
 55        a = [e for _ in range(a)]
 56    elif not isinstance(a, list):
 57      a = list(a)
 58    if a:
 59      self._build(a)

 61  def _build(self, a: List[T]) -> None:
 62    _Node = self._Node
 63    id = self.id
 64    def build(l: int, r: int) -> _Node:
 65      mid = (l + r) >> 1
 66      node = _Node(a[mid], id)
 67      if l != mid:
 68        node.left = build(l, mid)
 69        node.left.par = node
 70      if mid+1 != r:
 71        node.right = build(mid+1, r)
 72        node.right.par = node
 73      self._update(node)
 74      return node
 75    self.root = build(0, len(a))
 76
 77  def _rotate(self, node: _Node) -> None:
 78    pnode = node.par
 79    gnode = pnode.par
 80    if gnode:
 81      if gnode.left is pnode:
 82        gnode.left = node
 83      else:
 84        gnode.right = node
 85    node.par = gnode
 86    if pnode.left is node:
 87      pnode.left = node.right
 88      if node.right:
 89        node.right.par = pnode
 90      node.right = pnode
 91    else:
 92      pnode.right = node.left
 93      if node.left:
 94        node.left.par = pnode
 95      node.left = pnode
 96    pnode.par = node
 97    self._update_double(pnode, node)

 99  def _propagate_rev(self, node: Optional[_Node]) -> None:
100    if not node: return
101    node.rev ^= 1
102
103  def _propagate_lazy(self, node: Optional[_Node], f: F) -> None:
104    if not node: return
105    node.key = self.mapping(f, node.key)
106    node.data = self.mapping(f, node.data)
107    node.rdata = self.mapping(f, node.rdata)
108    node.lazy = f if node.lazy == self.id else self.composition(f, node.lazy)
109
110  def _propagate(self, node: Optional[_Node]) -> None:
111    if not node: return
112    if node.rev:
113      node.data, node.rdata = node.rdata, node.data
114      node.left, node.right = node.right, node.left
115      self._propagate_rev(node.left)
116      self._propagate_rev(node.right)
117      node.rev = 0
118    if node.lazy != self.id:
119      self._propagate_lazy(node.left, node.lazy)
120      self._propagate_lazy(node.right, node.lazy)
121      node.lazy = self.id
122
123  def _update_double(self, pnode: _Node, node: _Node) -> None:
124    node.data = pnode.data
125    node.rdata = pnode.rdata
126    node.size = pnode.size
127    self._update(pnode)
128
129  def _update(self, node: _Node) -> None:
130    node.data = node.key
131    node.rdata = node.key
132    node.size = 1
133    if node.left:
134      node.data = self.op(node.left.data, node.data)
135      node.rdata = self.op(node.rdata, node.left.rdata)
136      node.size += node.left.size
137    if node.right:
138      node.data = self.op(node.data, node.right.data)
139      node.rdata = self.op(node.right.rdata, node.rdata)
140      node.size += node.right.size
141
142  def _splay(self, node: _Node) -> None:
143    while node.par and node.par.par:
144      pnode = node.par
145      self._rotate(pnode if (pnode.par.left is pnode) == (pnode.left is node) else node)
146      self._rotate(node)
147    if node.par:
148      self._rotate(node)
149
150  def _get_kth_elm_splay(self, node: Optional[_Node], k: int) -> None:
151    if k < 0:
152      k += len(self)
153    while True:
154      self._propagate(node)
155      t = node.left.size if node.left else 0
156      if t == k:
157        break
158      if t > k:
159        node = node.left
160      else:
161        node = node.right
162        k -= t + 1
163    self._splay(node)
164    return node
165
166  def _get_left_splay(self, node: Optional[_Node]) -> Optional[_Node]:
167    self._propagate(node)
168    if not node or not node.left:
169      return node
170    while node.left:
171      node = node.left
172      self._propagate(node)
173    self._splay(node)
174    return node
175
176  def _get_right_splay(self, node: Optional[_Node]) -> Optional[_Node]:
177    self._propagate(node)
178    if not node or not node.right:
179      return node
180    while node.right:
181      node = node.right
182      self._propagate(node)
183    self._splay(node)
184    return node

203  def split(self, k: int) -> Tuple['LazySplayTree', 'LazySplayTree']:
204    """位置 ``k`` で split します。
205    償却 :math:`O(\\log{n})` です。
206
207    Returns:
208      Tuple['LazySplayTree', 'LazySplayTree']:
209    """
210    left, right = self._internal_split(self.root, k)
211    left_splay = LazySplayTree(0, self.op, self.mapping, self.composition, self.e, self.id, left)
212    right_splay = LazySplayTree(0, self.op, self.mapping, self.composition, self.e, self.id, right)
213    return left_splay, right_splay

215  def _internal_split(self, k: int) -> Tuple[_Node, _Node]:
216    # self.root will be broken
217    if k >= len(self):
218      return self.root, None
219    right = self._get_kth_elm_splay(self.root, k)
220    left = right.left
221    if left:
222      left.par = None
223    right.left = None
224    self._update(right)
225    return left, right
226
227  def _internal_merge(self, left: Optional[_Node], right: Optional[_Node]) -> Optional[_Node]:
228    # need (not right) or (not right.left)
229    if not right:
230      return left
231    assert right.left is None
232    right.left = left
233    if left:
234      left.par = right
235    self._update(right)
236    return right

238  def reverse(self, l: int, r: int) -> None:
239    """区間 ``[l, r)`` を反転します。
240    償却 :math:`O(\\log{n})` です。
241
242    Args:
243      l (int):
244      r (int):
245    """
246    assert 0 <= l <= r <= len(self), \
247        f'IndexError: {self.__class__.__name__}.reverse({l}, {r}), len={len(self)}'
248    left, right = self._internal_split(r)
249    if l == 0:
250      self._propagate_rev(left)
251    else:
252      left = self._get_kth_elm_splay(left, l-1)
253      self._propagate_rev(left.right)
254    self.root = self._internal_merge(left, right)

288  def prod(self, l: int, r: int) -> T:
289    """区間 ``[l, r)`` の総積を求めます。
290    償却 :math:`O(\\log{n})` です。
291
292    Args:
293      l (int):
294      r (int):
295
296    Returns:
297      T:
298    """
299    assert 0 <= l <= r <= len(self), \
300        f'IndexError: {self.__class__.__name__}.prod({l}, {r}), len={len(self)}'
301    if l == r:
302      return self.e
303    left, right = self._internal_split(r)
304    if l == 0:
305      res = left.data
306    else:
307      left = self._get_kth_elm_splay(left, l-1)
308      res = left.right.data
309    self.root = self._internal_merge(left, right)
310    return res

319  def insert(self, k: int, key: T) -> None:
320    """位置 ``k`` に ``key`` を挿入します。
321    償却 :math:`O(\\log{n})` です。
322
323    Args:
324      k (int):
325      key (T):
326    """
327    node = self._Node(key, self.id)
328    if not self.root:
329      self.root = node
330      return
331    if k >= len(self):
332      root = self._get_kth_elm_splay(self.root, len(self)-1)
333      node.left = root
334    else:
335      root = self._get_kth_elm_splay(self.root, k)
336      if root.left:
337        node.left = root.left
338        root.left.par = node
339        root.left = None
340        self._update(root)
341      node.right = root
342    root.par = node
343    self.root = node
344    self._update(self.root)

374  def pop(self, k: int=-1) -> T:
375    """位置 ``k`` の要素を削除し、その値を返します。
376    償却 :math:`O(\\log{n})` です。
377
378    Args:
379      k (int, optional): 指定するインデックスです。 Defaults to -1.
380
381    Returns:
382      T:
383    """
384    if k == -1:
385      node = self._get_right_splay(self.root)
386      if node.left:
387        node.left.par = None
388      self.root = node.left
389      return node.key
390    root = self._get_kth_elm_splay(self.root, k)
391    res = root.key
392    if root.left and root.right:
393      node = self._get_right_splay(root.left)
394      node.par = None
395      node.right = root.right
396      if node.right:
397        node.right.par = node
398      self._update(node)
399      self.root = node
400    else:
401      self.root = root.right if root.right else root.left
402      if self.root:
403        self.root.par = None
404    return res

436  def tolist(self) -> List[T]:
437    """``list`` にして返します。
438    :math:`O(n)` です。非再帰です。
439
440    Returns:
441      List[T]:
442    """
443    node = self.root
444    stack = []
445    a = newlist_hint(len(self))
446    while stack or node:
447      if node:
448        self._propagate(node)
449        stack.append(node)
450        node = node.left
451      else:
452        node = stack.pop()
453        a.append(node.key)
454        node = node.right
455    return a
