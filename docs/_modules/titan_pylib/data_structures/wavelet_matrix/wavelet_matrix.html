  6class WaveletMatrix():
  7  """``WaveletMatrix`` です。
  8  静的であることに注意してください。
  9
 10  以下の仕様の計算量には嘘があるかもしれません。import 元の ``BitVector`` の計算量も参考にしてください。
 11
 12  - 参考:
 13    - `https://miti-7.hatenablog.com/entry/2018/04/28/152259 <https://miti-7.hatenablog.com/entry/2018/04/28/152259>`
 14    - `https://www.slideshare.net/pfi/ss-15916040 <https://www.slideshare.net/pfi/ss-15916040>`
 15    - `デwiki <https://scrapbox.io/data-structures/Wavelet_Matrix>`
 16  """
 17
 18  def __init__(self, sigma: int, a: Sequence[int]=[]):
 19    """``[0, sigma)`` の整数列を管理する ``WaveletMatrix`` を構築します。
 20    :math:`O(n\\log{\\sigma})` です。
 21
 22    Args:
 23      sigma (int): 扱う整数の上限です。
 24      a (Sequence[int], optional): 構築する配列です。
 25    """
 26    self.sigma: int = sigma
 27    self.log: int = (sigma-1).bit_length()
 28    self.mid: array[int] = array('I', bytes(4*self.log))
 29    self.size: int = len(a)
 30    self.v: List[BitVector] = [BitVector(self.size) for _ in range(self.log)]
 31    self._build(a)
 32
 33  def _build(self, a: Sequence[int]) -> None:
 34    # 列 a から wm を構築する
 35    for bit in range(self.log-1, -1, -1):
 36      # bit目の0/1に応じてvを構築 + aを安定ソート
 37      v = self.v[bit]
 38      zero, one = [], []
 39      for i, e in enumerate(a):
 40        if e >> bit & 1:
 41          v.set(i)
 42          one.append(e)
 43        else:
 44          zero.append(e)
 45      v.build()
 46      self.mid[bit] = len(zero)  # 境界をmid[bit]に保持
 47      a = zero + one

 49  def access(self, k: int) -> int:
 50    """k番目の値を返します。
 51    :math:`O(\\log{\\sigma})` です。
 52
 53    Args:
 54      k (int): インデックスです。
 55    """
 56    assert -self.size <= k < self.size, \
 57        f'IndexError: {self.__class__.__name__}.access({k}), size={self.size}'
 58    if k < 0:
 59      k += self.size
 60    s = 0  # 答え
 61    for bit in range(self.log-1, -1, -1):
 62      if self.v[bit].access(k):
 63        # k番目が立ってたら、
 64        # kまでの1とすべての0が次のk
 65        s |= 1 << bit
 66        k = self.v[bit].rank1(k) + self.mid[bit]
 67      else:
 68        # kまでの0が次のk
 69        k = self.v[bit].rank0(k)
 70    return s

 76  def rank(self, r: int, x: int) -> int:
 77    """``a[0, r)`` に含まれる ``x`` の個数を返します。
 78    :math:`O(\\log{\\sigma})` です。
 79    """
 80    assert 0 <= r <= self.size, f'IndexError: {self.__class__.__name__}.rank(), r={r}, size={self.size}'
 81    assert 0 <= x < 1<<self.log, f'ValueError: {self.__class__.__name__}.rank(), x={x}, LIM={1<<self.log}'
 82    l = 0
 83    mid = self.mid
 84    for bit in range(self.log-1, -1, -1):
 85      # 位置 r より左に x が何個あるか
 86      # x の bit 目で場合分け
 87      if x >> bit & 1:
 88        # 立ってたら、次のl, rは以下
 89        l = self.v[bit].rank1(l) + mid[bit]
 90        r = self.v[bit].rank1(r) + mid[bit]
 91      else:
 92        # そうでなければ次のl, rは以下
 93        l = self.v[bit].rank0(l)
 94        r = self.v[bit].rank0(r)
 95    return r - l

 97  def select(self, k: int, x: int) -> int:
 98    """``k`` 番目の ``v`` のインデックスを返します。
 99    :math:`O(\\log{\\sigma})` です。
100    """
101    assert 0 <= k < self.size, f'IndexError: {self.__class__.__name__}.select({k}, {x}), k={k}, size={self.size}'
102    assert 0 <= x < 1<<self.log, f'ValueError: {self.__class__.__name__}.select({k}, {x}), x={x}, LIM={1<<self.log}'
103    # x の開始位置 s を探す
104    s = 0
105    for bit in range(self.log-1, -1, -1):
106      if x >> bit & 1:
107        s = self.v[bit].rank0(self.size) + self.v[bit].rank1(s)
108      else:
109        s = self.v[bit].rank0(s)
110    s += k  # s から k 進んだ位置が、元の列で何番目か調べる
111    for bit in range(self.log):
112      if x >> bit & 1:
113        s = self.v[bit].select1(s - self.v[bit].rank0(self.size))
114      else:
115        s = self.v[bit].select0(s)
116    return s

118  def kth_smallest(self, l: int, r: int, k: int) -> int:
119    """``a[l, r)`` の中で k 番目に **小さい** 値を返します。
120    :math:`O(\\log{\\sigma})` です。
121    """
122    assert 0 <= l <= r <= self.size, f'IndexError: {self.__class__.__name__}.kth_smallest({l}, {r}, {k}), size={self.size}'
123    assert 0 <= k < r-l, f'IndexError: {self.__class__.__name__}.kth_smallest({l}, {r}, {k}), wrong k'
124    s = 0
125    mid = self.mid
126    for bit in range(self.log-1, -1, -1):
127      r0, l0 = self.v[bit].rank0(r), self.v[bit].rank0(l)
128      cnt = r0 - l0  # 区間内の 0 の個数
129      if cnt <= k:  # 0 が k 以下のとき、 k 番目は 1
130        s |= 1 << bit
131        k -= cnt
132        # この 1 が次の bit 列でどこに行くか
133        l = l - l0 + mid[bit]
134        r = r - r0 + mid[bit]
135      else:
136        # この 0 が次の bit 列でどこに行くか
137        l = l0
138        r = r0
139    return s

151  def topk(self, l: int, r: int, k: int) -> List[Tuple[int, int]]:
152    """``a[l, r)`` の中で、要素を出現回数が多い順にその頻度とともに ``k`` 個返します。
153    :math:`O(\\min(r-l, \\sigam) \\log(\\sigam))` です。
154
155    Note:
156      :math:`\\sigma` が大きい場合、計算量に注意です。
157
158    Returns:
159      List[Tuple[int, int]]: ``(要素, 頻度)`` を要素とする配列です。
160    """
161    assert 0 <= l <= r <= self.size, f'IndexError: {self.__class__.__name__}.topk({l}, {r}, {k}), size={self.size}'
162    assert 0 <= k < r-l, f'IndexError: {self.__class__.__name__}.topk({l}, {r}, {k}), wrong k'
163    # heap[-length, x, l, bit]
164    hq: List[Tuple[int, int, int, int]] = [(-(r-l), 0, l, self.log-1)]
165    ans = []
166    while hq:
167      length, x, l, bit = heappop(hq)
168      length = -length
169      if bit == -1:
170        ans.append((x, length))
171        k -= 1
172        if k == 0:
173          break
174      else:
175        r = l + length
176        l0 = self.v[bit].rank0(l)
177        r0 = self.v[bit].rank0(r)
178        if l0 < r0:
179          heappush(hq, (-(r0-l0), x, l0, bit-1))
180        l1 = self.v[bit].rank1(l) + self.mid[bit]
181        r1 = self.v[bit].rank1(r) + self.mid[bit]
182        if l1 < r1:
183          heappush(hq, (-(r1-l1), x|(1<<bit), l1, bit-1))
184    return ans

194  def _range_freq(self, l: int, r: int, x: int) -> int:
195    '''a[l, r) で x 未満の要素の数を返す'''
196    ans = 0
197    for bit in range(self.log-1, -1, -1):
198      l0, r0 = self.v[bit].rank0(l), self.v[bit].rank0(r)
199      if x >> bit & 1:
200        # bit が立ってたら、区間の 0 の個数を答えに加算し、新たな区間は 1 のみ
201        ans += r0 - l0
202        # 1 が次の bit 列でどこに行くか
203        l += self.mid[bit] - l0
204        r += self.mid[bit] - r0
205      else:
206        # 0 が次の bit 列でどこに行くか
207        l, r = l0, r0
208    return ans
