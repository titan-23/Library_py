  6class DynamicWaveletMatrix(WaveletMatrix):
  7  """動的ウェーブレット行列です。
  8
  9  (静的)ウェーブレット行列でできる操作に加えて ``insert / pop / update`` 等ができます。
 10    - ``BitVector`` を平衡二分木にしています(``AVLTreeBitVector``)。あらゆる操作に平衡二分木の log がつきます。これヤバくね
 11
 12  :math:`O(n\\log{(\\sigma)})` です。
 13  """
 14
 15  def __init__(self, sigma: int, a: Sequence[int]=[]):
 16    self.sigma: int = sigma
 17    self.log: int = (sigma-1).bit_length()
 18    self.v: List[AVLTreeBitVector] = [AVLTreeBitVector()] * self.log
 19    self.mid: array[int] = array('I', bytes(4*self.log))
 20    self.size: int = len(a)
 21    self._build(a)
 22
 23  def _build(self, a: Sequence[int]) -> None:
 24    v = array('B', bytes(self.size))
 25    for bit in range(self.log-1, -1, -1):
 26      # bit目の0/1に応じてvを構築 + aを安定ソート
 27      zero, one = [], []
 28      for i, e in enumerate(a):
 29        if e >> bit & 1:
 30          v[i] = 1
 31          one.append(e)
 32        else:
 33          v[i] = 0
 34          zero.append(e)
 35      self.mid[bit] = len(zero)  # 境界をmid[bit]に保持
 36      self.v[bit] = AVLTreeBitVector(v)
 37      a = zero + one

 46  def insert(self, k: int, x: int) -> None:
 47    """位置 ``k`` に ``x`` を挿入します。 :math:`O(\\log{(n)}\\log{(\\sigma)})` です。
 48    """
 49    assert 0 <= k <= self.size, f'IndexError: {self.__class__.__name__}.insert({k}, {x}), n={self.size}'
 50    assert 0 <= x < 1<<self.log, f'ValueError: {self.__class__.__name__}.insert({k}, {x}), LIM={1<<self.log}'
 51    mid = self.mid
 52    for bit in range(self.log-1, -1, -1):
 53      v = self.v[bit]
 54      # if x >> bit & 1:
 55      #   v.insert(k, 1)
 56      #   k = v.rank1(k) + mid[bit]
 57      # else:
 58      #   v.insert(k, 0)
 59      #   mid[bit] += 1
 60      #   k = v.rank0(k)
 61      if x >> bit & 1:
 62        s = v._insert_and_rank1(k, 1)
 63        k = s + mid[bit]
 64      else:
 65        s = v._insert_and_rank1(k, 0)
 66        k -= s
 67        mid[bit] += 1
 68    self.size += 1

 70  def pop(self, k: int) -> int:
 71    """位置 ``k`` の要素を削除し、その値を返します。 :math:`O(\\log{(n)}\\log{(\\sigma)})` です。
 72    """
 73    assert 0 <= k < self.size, f'IndexError: {self.__class__.__name__}.pop({k}), n={self.size}'
 74    mid = self.mid
 75    ans = 0
 76    for bit in range(self.log-1, -1, -1):
 77      v = self.v[bit]
 78      # K = k
 79      # if v.access(k):
 80      #   ans |= 1 << bit
 81      #   k = v.rank1(k) + mid[bit]
 82      # else:
 83      #   mid[bit] -= 1
 84      #   k = v.rank0(k)
 85      # v.pop(K)
 86      sb = v._access_pop_and_rank1(k)
 87      s = sb >> 1
 88      if sb & 1:
 89        ans |= 1 << bit
 90        k = s + mid[bit]
 91      else:
 92        mid[bit] -= 1
 93        k -= s
 94    self.size -= 1
 95    return ans
