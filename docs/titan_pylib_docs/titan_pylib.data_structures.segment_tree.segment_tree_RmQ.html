 73  def __init__(self, _n_or_a: Union[int, Iterable[T]], e: T) -> None:
 74    self._e = e
 75    if isinstance(_n_or_a, int):
 76      self._n = _n_or_a
 77      self._log = (self._n - 1).bit_length()
 78      self._size = 1 << self._log
 79      self._data = [self._e] * (self._size << 1)
 80    else:
 81      _n_or_a = list(_n_or_a)
 82      self._n = len(_n_or_a)
 83      self._log = (self._n - 1).bit_length()
 84      self._size = 1 << self._log
 85      _data = [self._e] * (self._size << 1)
 86      _data[self._size:self._size+self._n] = _n_or_a
 87      for i in range(self._size-1, 0, -1):
 88        _data[i] = _data[i<<1] if _data[i<<1] < _data[i<<1|1] else _data[i<<1|1]
 89      self._data = _data

 91  def set(self, k: int, v: T) -> None:
 92    if k < 0: k += self._n
 93    assert 0 <= k < self._n, \

107  def prod(self, l: int, r: int) -> T:
108    assert 0 <= l <= r <= self._n, \
109        f'IndexError: {self.__class__.__name__}.prod({l}: int, {r}: int)'
110    l += self._size
111    r += self._size
112    res = self._e
113    while l < r:
114      if l & 1:
115        if res > self._data[l]:
116          res = self._data[l]
117        l += 1
118      if r & 1:
119        r ^= 1
120        if res > self._data[r]:
121          res = self._data[r]
122      l >>= 1
123      r >>= 1
124    return res

129  def max_right(self, l: int, f=lambda lr: lr):
130    assert 0 <= l <= self._n, \
131        f'IndexError: {self.__class__.__name__}.max_right({l}, f) index out of range'
132    assert f(self._e), \
133        f'{self.__class__.__name__}.max_right({l}, f), f({self._e}) must be true.'
134    if l == self._n: return self._n
135    l += self._size
136    s = self._e
137    while True:
138      while l & 1 == 0:
139        l >>= 1
140      if not f(min(s, self._data[l])):
141        while l < self._size:
142          l <<= 1
143          if f(min(s, self._data[l])):
144            if s > self._data[l]:
145              s = self._data[l]
146            l += 1
147        return l - self._size
148      s = min(s, self._data[l])
149      l += 1
150      if l & -l == l:
151        break
152    return self._n

154  def min_left(self, r: int, f=lambda lr: lr):
155    assert 0 <= r <= self._n, \
156        f'IndexError: {self.__class__.__name__}.min_left({r}, f) index out of range'
157    assert f(self._e), \
158        f'{self.__class__.__name__}.min_left({r}, f), f({self._e}) must be true.'
159    if r == 0: return 0
160    r += self._size
161    s = self._e
162    while True:
163      r -= 1
164      while r > 1 and r & 1:
165        r >>= 1
166      if not f(min(self._data[r], s)):
