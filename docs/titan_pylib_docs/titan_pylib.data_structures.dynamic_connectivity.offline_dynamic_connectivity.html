 16    """内部で管理される `UndoableUnionFind` です。

 19    def __init__(self, n: int):
 20      self._n: int = n
 21      self._parents: List[int] = [-1] * n
 22      self._all_sum: List[int] = [0] * n
 23      self._one_sum: List[int] = [0] * n
 24      self._history: List[Tuple[int, int, int]] = []
 25      self._group_count: int = n

 27    def _undo(self) -> None:
 28      assert self._history, 'UndoableUnionFind._undo() with non history'
 29      y, py, all_sum_y = self._history.pop()
 30      if y == -1:
 31        return
 32      x, px, all_sum_x = self._history.pop()
 33      self._group_count += 1
 34      self._parents[y] = py
 35      self._parents[x] = px
 36      s = (self._all_sum[x] - all_sum_y - all_sum_x) // (-py-px) * (-py)
 37      self._all_sum[y] += s
 38      self._all_sum[x] -= all_sum_y + s
 39      self._one_sum[x] -= self._one_sum[y]
 40
 41    def root(self, x: int) -> int:
 42      """要素 ``x`` を含む集合の代表元を返します。
 43
 44      :math:`O(\\log{n})` です。
 45      """
 46      while self._parents[x] >= 0:
 47        x = self._parents[x]
 48      return x
 49
 50    def unite(self, x: int, y: int) -> bool:
 51      """要素 ``x`` を含む集合と要素 ``y`` を含む集合を併合します。
 52
 53      :math:`O(\\log{n})` です。
 54
 55      Returns:
 56        bool: もともと同じ集合であれば ``False``、そうでなければ ``True`` を返します。
 57      """
 58      x = self.root(x)
 59      y = self.root(y)
 60      if x == y:
 61        self._history.append((-1, -1, -1))
 62        return False
 63      if self._parents[x] > self._parents[y]:
 64        x, y = y, x
 65      self._group_count -= 1
 66      self._history.append((x, self._parents[x], self._all_sum[x]))
 67      self._history.append((y, self._parents[y], self._all_sum[y]))
 68      self._all_sum[x] += self._all_sum[y]
 69      self._one_sum[x] += self._one_sum[y]
 70      self._parents[x] += self._parents[y]
 71      self._parents[y] = x
 72      return True
 73
 74    def size(self, x: int) -> int:
 75      """要素 ``x`` を含む集合の要素数を返します。
 76
 77      :math:`O(\\log{n})` です。
 78      """
 79      return -self._parents[self.root(x)]
 80
 81    def same(self, x: int, y: int) -> bool:
 82      """
 83      要素 ``x`` と ``y`` が同じ集合に属するなら ``True`` を、
 84      そうでないなら ``False`` を返します。
 85
 86      :math:`O(\\log{n})` です。

 88      return self.root(x) == self.root(y)

114    def group_sum(self, x: int) -> int:
115      """
116      `key` に代表元、 `value` に `key` を代表元とする集合のリストをもつ `defaultdict` を返します。

118      :math:`O(n\\log{n})` です。

123    def all_group_members(self) -> defaultdict:
124      """`key` に代表元、 `value` に `key` を代表元とする集合のリストをもつ `defaultdict` を返します。

126      :math:`O(n\\log{n})` です。

136  def __init__(self, n: int) -> None:
137    """初期状態を頂点数 ``n`` の無向グラフとします。
138
139    :math:`O(n)` です。
140
141    Args:

144    self._n = n
145    self._query_count = 0
146    self._bit = n.bit_length() + 1
147    self._msk = (1 << self._bit) - 1
148    self._start = defaultdict(lambda: [0, 0])
149    self._edge_data = []

152  def add_edge(self, u: int, v: int) -> None:
153    """辺 ``{u, v}`` を追加します。
154
155    :math:`O(1)` です。
156    """
157    assert 0 <= u < self._n and 0 <= v < self._n
158    if u > v:
159      u, v = v, u
160    edge = u<<self._bit|v
161    if self._start[edge][0] == 0:
162      self._start[edge][1] = self._query_count
163    self._start[edge][0] += 1
164
165  def delete_edge(self, u: int, v: int) -> None:
166    """辺 ``{u, v}`` を削除します。
167
168    :math:`O(1)` です。
169    """
170    assert 0 <= u < self._n and 0 <= v < self._n
171    if u > v:
172      u, v = v, u
173    edge = u<<self._bit|v
174    if self._start[edge][0] == 1:
175      self._edge_data.append((self._start[edge][1], self._query_count, edge))
176    self._start[edge][0] -= 1
177
178  def next_query(self) -> None:
179    """クエリカウントを 1 進めます。
180
181    :math:`O(1)` です。
182    """
183    self._query_count += 1

185  def run(self, out: Callable[[int], None]) -> None:
186    """実行します。
187
188    :math:`O(q \\log{q} \\log{n})` です。
189
190    Args:
191        out (Callable[[int], None]): クエリ番号 ``k`` を引数にとります。
192    """
193    # O(qlogqlogn)
194    uf, bit, msk, q = self.uf, self._bit, self._msk, self._query_count
195    log = (q - 1).bit_length()
196    size = 1 << log
197    size2 = size * 2
198    data = [[] for _ in range(size<<1)]
199
200    def add(l, r, edge):
201      l += size
202      r += size
203      while l < r:
204        if l & 1:
205          data[l].append(edge)
206          l += 1
207        if r & 1:
208          data[r^1].append(edge)
209        l >>= 1
210        r >>= 1
211
212    for edge, p in self._start.items():
213      if p[0] != 0:
214        add(p[1], self._query_count, edge)
215    for l, r, edge in self._edge_data:
216      add(l, r, edge)
217
218    todo = [1]
219    while todo:
220      v = todo.pop()
221      if v >= 0:
222        for uv in data[v]:
223          uf.unite(uv>>bit, uv&msk)
224        todo.append(~v)
225        if v<<1|1 < size2:
226          todo.append(v<<1|1)
227          todo.append(v<<1)
228        elif v - size < q:
229          out(v-size)
230      else:
231        for _ in data[~v]:
232          uf._undo()
