 10  '''Random
 11  乱数系のライブラリです。
 12  標準ライブラリよりも高速なつもりでいます。
 13  '''
 14
 15  def __init__(self) -> None:
 16    self._x = 123456789
 17    self._y = 362436069
 18    self._z = 521288629
 19    self._w = 88675123

 21  def _xor(self) -> int:
 22    t = (self._x ^ ((self._x << 11) & 0xFFFFFFFF)) & 0xFFFFFFFF
 23    self._x, self._y, self._z = self._y, self._z, self._w
 24    self._w = (self._w ^ (self._w >> 19)) ^ (t ^ ((t >> 8))&0xFFFFFFFF) & 0xFFFFFFFF
 25    return self._w
 26
 27  def random(self) -> float:
 28    """0以上1以下の一様ランダムな値を1つ生成して返すはずです。
 29    :math:`O(1)` です。
 30    """
 31    return self._xor() / 0xFFFFFFFF
 32
 33  def randint(self, begin: int, end: int) -> int:
 34    """``begin`` 以上 ``end`` **以下** のランダムな整数を返します。
 35    :math:`O(1)` です。
 36
 37    制約:
 38      :math:`begin \\leq end`
 39    """
 40    assert begin <= end
 41    return begin + self._xor() % (end - begin + 1)
 42
 43  def randrange(self, begin: int, end: int) -> int:
 44    """``begin`` 以上 ``end`` **未満** のランダムな整数を返します。
 45    :math:`O(1)` です。
 46
 47    制約:
 48      :math:`begin < end`
 49    """
 50    assert begin < end
 51    return begin + self._xor() % (end - begin)
 52
 53  def shuffle(self, a: List[Any]) -> None:
 54    """``a`` をインプレースにシャッフルします。
 55    :math:`O(n)` です。
 56
 57    Args:
 58      a (List[Any]): ``a`` をシャッフルします。
 59    """
 60    n = len(a)
 61    for i in range(n-1):
 62      j = self.randrange(i, n)
 63      a[i], a[j] = a[j], a[i]

 93            TIME_LIMIT: float=1.8) -> Tuple[State, int]:
 94    START_TIME = process_time()

 96    ans, score = self.make_ans_init()
 97    vest_ans = ans.copy()
 98    vest_score = score
 99    cnt = 0
100    while True:

102      if now_time > TIME_LIMIT:

106      new_score = score + diff_score

110        score = new_score
