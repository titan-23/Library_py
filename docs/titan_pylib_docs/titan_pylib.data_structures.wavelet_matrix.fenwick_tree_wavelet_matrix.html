 54class BitVector(BitVectorInterface):
 55  """コンパクトな bit vector です。
 56  """
 57
 58  def __init__(self, n: int):
 59    """長さ ``n`` の ``BitVector`` です。
 60
 61    bit を保持するのに ``array[I]`` を使用します。
 62    ``block_size= n / 32`` として、使用bitは ``32*block_size=2n bit`` です。
 63
 64    累積和を保持するのに同様の ``array[I]`` を使用します。
 65    32bitごとの和を保存しています。同様に使用bitは ``2n bit`` です。
 66    """
 67    assert 0 <= n < 4294967295
 68    self.N = n
 69    self.block_size = (n + 31) >> 5
 70    b = bytes(4*(self.block_size+1))
 71    self.bit = array('I', b)
 72    self.acc = array('I', b)
 73
 74  @staticmethod
 75  def _popcount(x: int) -> int:
 76    x = x - ((x >> 1) & 0x55555555)
 77    x = (x & 0x33333333) + ((x >> 2) & 0x33333333)
 78    x = x + (x >> 4) & 0x0f0f0f0f
 79    x += x >> 8
 80    x += x >> 16
 81    return x & 0x0000007f
 82
 83  def set(self, k: int) -> None:
 84    """``k`` 番目の bit を ``1`` にします。
 85    :math:`O(1)` です。
 86
 87    Args:
 88      k (int): インデックスです。
 89    """
 90    self.bit[k>>5] |= 1 << (k & 31)
 91
 92  def build(self) -> None:
 93    """構築します。
 94    **これ以降 ``set`` メソッドを使用してはいけません。**
 95    :math:`O(n)` です。
 96    """
 97    acc, bit = self.acc, self.bit
 98    for i in range(self.block_size):
 99      acc[i+1] = acc[i] + BitVector._popcount(bit[i])
100
101  def access(self, k: int) -> int:
102    """``k`` 番目の bit を返します。
103    :math:`O(1)` です。
104    """
105    return (self.bit[k >> 5] >> (k & 31)) & 1

107  def __getitem__(self, k: int) -> int:

110  def rank0(self, r: int) -> int:
111    """``a[0, r)`` に含まれる ``0`` の個数を返します。
112    :math:`O(1)` です。
113    """
114    return r - (self.acc[r>>5] + BitVector._popcount(self.bit[r>>5] & ((1 << (r & 31)) - 1)))

118    :math:`O(1)` です。
119    """
120    return self.acc[r>>5] + BitVector._popcount(self.bit[r>>5] & ((1 << (r & 31)) - 1))

122  def rank(self, r: int, v: int) -> int:
123    """``a[0, r)`` に含まれる ``v`` の個数を返します。
124    :math:`O(1)` です。
125    """
126    return self.rank1(r) if v else self.rank0(r)

128  def select0(self, k: int) -> int:
129    """``k`` 番目の ``0`` のインデックスを返します。
130    :math:`O(\\log{n})` です。
131    """
132    if k < 0 or self.rank0(self.N) <= k:
133      return -1
134    l, r = 0, self.block_size+1
135    while r - l > 1:
136      m = (l + r) >> 1
137      if m*32 - self.acc[m] > k:
138        r = m
139      else:
140        l = m
141    indx = 32 * l
142    k = k - (l*32 - self.acc[l]) + self.rank0(indx)
143    l, r = indx, indx+32
144    while r - l > 1:
145      m = (l + r) >> 1
146      if self.rank0(m) > k:
147        r = m
148      else:
149        l = m
150    return l

152  def select1(self, k: int) -> int:
153    """``k`` 番目の ``1`` のインデックスを返します。
154    :math:`O(\\log{n})` です。
155    """
156    if k < 0 or self.rank1(self.N) <= k:
157      return -1
158    l, r = 0, self.block_size+1
159    while r - l > 1:
160      m = (l + r) >> 1
161      if self.acc[m] > k:
162        r = m
163      else:
164        l = m
165    indx = 32 * l
166    k = k - self.acc[l] + self.rank1(indx)
167    l, r = indx, indx+32
168    while r - l > 1:
169      m = (l + r) >> 1
170      if self.rank1(m) > k:
171        r = m
172      else:
173        l = m
174    return l

196class WaveletMatrix():
197  """``WaveletMatrix`` です。
198  静的であることに注意してください。
199
200  以下の仕様の計算量には嘘があるかもしれません。import 元の ``BitVector`` の計算量も参考にしてください。
201
202  - 参考:
203    - `https://miti-7.hatenablog.com/entry/2018/04/28/152259 <https://miti-7.hatenablog.com/entry/2018/04/28/152259>`
204    - `https://www.slideshare.net/pfi/ss-15916040 <https://www.slideshare.net/pfi/ss-15916040>`
205    - `デwiki <https://scrapbox.io/data-structures/Wavelet_Matrix>`
206  """
207
208  def __init__(self, sigma: int, a: Sequence[int]=[]):
209    """``[0, sigma)`` の整数列を管理する ``WaveletMatrix`` を構築します。
210    :math:`O(n\\log{\\sigma})` です。
211
212    Args:
213      sigma (int): 扱う整数の上限です。
214      a (Sequence[int], optional): 構築する配列です。
215    """
216    self.sigma: int = sigma
217    self.log: int = (sigma-1).bit_length()
218    self.mid: array[int] = array('I', bytes(4*self.log))
219    self.size: int = len(a)
220    self.v: List[BitVector] = [BitVector(self.size) for _ in range(self.log)]
221    self._build(a)
222
223  def _build(self, a: Sequence[int]) -> None:
224    # 列 a から wm を構築する
225    for bit in range(self.log-1, -1, -1):
226      # bit目の0/1に応じてvを構築 + aを安定ソート
227      v = self.v[bit]
228      zero, one = [], []
229      for i, e in enumerate(a):
230        if e >> bit & 1:
231          v.set(i)
232          one.append(e)
233        else:
234          zero.append(e)
235      v.build()
236      self.mid[bit] = len(zero)  # 境界をmid[bit]に保持
237      a = zero + one
238
239  def access(self, k: int) -> int:
240    """k番目の値を返します。
241    :math:`O(\\log{\\sigma})` です。
242
243    Args:
244      k (int): インデックスです。
245    """
246    assert -self.size <= k < self.size, \
247        f'IndexError: {self.__class__.__name__}.access({k}), size={self.size}'
248    if k < 0:
249      k += self.size
250    s = 0  # 答え
251    for bit in range(self.log-1, -1, -1):
252      if self.v[bit].access(k):
253        # k番目が立ってたら、
254        # kまでの1とすべての0が次のk
255        s |= 1 << bit
256        k = self.v[bit].rank1(k) + self.mid[bit]
257      else:
258        # kまでの0が次のk
259        k = self.v[bit].rank0(k)
260    return s

262  def __getitem__(self, k: int) -> int:

266  def rank(self, r: int, x: int) -> int:
267    """``a[0, r)`` に含まれる ``x`` の個数を返します。
268    :math:`O(\\log{\\sigma})` です。
269    """
270    assert 0 <= r <= self.size, f'IndexError: {self.__class__.__name__}.rank(), r={r}, size={self.size}'
271    assert 0 <= x < 1<<self.log, f'ValueError: {self.__class__.__name__}.rank(), x={x}, LIM={1<<self.log}'
272    l = 0
273    mid = self.mid
274    for bit in range(self.log-1, -1, -1):
275      # 位置 r より左に x が何個あるか
276      # x の bit 目で場合分け
277      if x >> bit & 1:
278        # 立ってたら、次のl, rは以下
279        l = self.v[bit].rank1(l) + mid[bit]
280        r = self.v[bit].rank1(r) + mid[bit]
281      else:
282        # そうでなければ次のl, rは以下
283        l = self.v[bit].rank0(l)
284        r = self.v[bit].rank0(r)
285    return r - l

287  def select(self, k: int, x: int) -> int:
288    """``k`` 番目の ``v`` のインデックスを返します。
289    :math:`O(\\log{\\sigma})` です。
290    """
291    assert 0 <= k < self.size, f'IndexError: {self.__class__.__name__}.select({k}, {x}), k={k}, size={self.size}'
292    assert 0 <= x < 1<<self.log, f'ValueError: {self.__class__.__name__}.select({k}, {x}), x={x}, LIM={1<<self.log}'
293    # x の開始位置 s を探す
294    s = 0
295    for bit in range(self.log-1, -1, -1):
296      if x >> bit & 1:
297        s = self.v[bit].rank0(self.size) + self.v[bit].rank1(s)
298      else:
299        s = self.v[bit].rank0(s)
300    s += k  # s から k 進んだ位置が、元の列で何番目か調べる
301    for bit in range(self.log):
302      if x >> bit & 1:
303        s = self.v[bit].select1(s - self.v[bit].rank0(self.size))
304      else:
305        s = self.v[bit].select0(s)
306    return s

308  def kth_smallest(self, l: int, r: int, k: int) -> int:
309    """``a[l, r)`` の中で k 番目に **小さい** 値を返します。
310    :math:`O(\\log{\\sigma})` です。
311    """
312    assert 0 <= l <= r <= self.size, f'IndexError: {self.__class__.__name__}.kth_smallest({l}, {r}, {k}), size={self.size}'
313    assert 0 <= k < r-l, f'IndexError: {self.__class__.__name__}.kth_smallest({l}, {r}, {k}), wrong k'
314    s = 0
315    mid = self.mid
316    for bit in range(self.log-1, -1, -1):
317      r0, l0 = self.v[bit].rank0(r), self.v[bit].rank0(l)
318      cnt = r0 - l0  # 区間内の 0 の個数
319      if cnt <= k:  # 0 が k 以下のとき、 k 番目は 1
320        s |= 1 << bit
321        k -= cnt
322        # この 1 が次の bit 列でどこに行くか
323        l = l - l0 + mid[bit]
324        r = r - r0 + mid[bit]
325      else:
326        # この 0 が次の bit 列でどこに行くか
327        l = l0
328        r = r0
329    return s

341  def topk(self, l: int, r: int, k: int) -> List[Tuple[int, int]]:
342    """``a[l, r)`` の中で、要素を出現回数が多い順にその頻度とともに ``k`` 個返します。
343    :math:`O(\\min(r-l, \\sigam) \\log(\\sigam))` です。
344
345    Note:
346      :math:`\\sigma` が大きい場合、計算量に注意です。
347
348    Returns:
349      List[Tuple[int, int]]: ``(要素, 頻度)`` を要素とする配列です。
350    """
351    assert 0 <= l <= r <= self.size, f'IndexError: {self.__class__.__name__}.topk({l}, {r}, {k}), size={self.size}'
352    assert 0 <= k < r-l, f'IndexError: {self.__class__.__name__}.topk({l}, {r}, {k}), wrong k'
353    # heap[-length, x, l, bit]
354    hq: List[Tuple[int, int, int, int]] = [(-(r-l), 0, l, self.log-1)]
355    ans = []
356    while hq:
357      length, x, l, bit = heappop(hq)
358      length = -length
359      if bit == -1:
360        ans.append((x, length))
361        k -= 1
362        if k == 0:
363          break
364      else:
365        r = l + length
366        l0 = self.v[bit].rank0(l)
367        r0 = self.v[bit].rank0(r)
368        if l0 < r0:
369          heappush(hq, (-(r0-l0), x, l0, bit-1))
370        l1 = self.v[bit].rank1(l) + self.mid[bit]
371        r1 = self.v[bit].rank1(r) + self.mid[bit]
372        if l1 < r1:
373          heappush(hq, (-(r1-l1), x|(1<<bit), l1, bit-1))
374    return ans

376  def sum(self, l: int, r: int) -> int:
377    """``topk`` メソッドを用いて ``a[l, r)`` の総和を返します。
378
379    計算量に注意です。
380    """
381    assert False, 'Yabai Keisanryo Error'
382    return sum(k*v for k, v in self.topk(l, r, r-l))

384  def _range_freq(self, l: int, r: int, x: int) -> int:
385    '''a[l, r) で x 未満の要素の数を返す'''
386    ans = 0
387    for bit in range(self.log-1, -1, -1):
388      l0, r0 = self.v[bit].rank0(l), self.v[bit].rank0(r)
389      if x >> bit & 1:
390        # bit が立ってたら、区間の 0 の個数を答えに加算し、新たな区間は 1 のみ
391        ans += r0 - l0
392        # 1 が次の bit 列でどこに行くか
393        l += self.mid[bit] - l0
394        r += self.mid[bit] - r0
395      else:
396        # 0 が次の bit 列でどこに行くか
397        l, r = l0, r0
398    return ans

444class FenwickTree():
445  """FenwickTreeです。
446  """
447
448  def __init__(self, n_or_a: Union[Iterable[int], int]):
449    """構築します。
450
451    :math:`O(n)` です。
452
453    Args:
454        n_or_a (Union[Iterable[int], int]): `n_or_a` が `int` のとき、初期値 `0` 、長さ `n` で構築します。
455                                            `n_or_a` が `Iterable` のとき、初期値 `a` で構築します。
456    """
457    if isinstance(n_or_a, int):
458      self._size = n_or_a
459      self._tree = [0] * (self._size + 1)
460    else:
461      a = n_or_a if isinstance(n_or_a, list) else list(n_or_a)
462      _size = len(a)
463      _tree = [0] + a
464      for i in range(1, _size):
465        if i + (i & -i) <= _size:
466          _tree[i + (i & -i)] += _tree[i]
467      self._size = _size
468      self._tree = _tree
469    self._s = 1 << (self._size - 1).bit_length()
470
471  def pref(self, r: int) -> int:
472    """区間 ``[0, r)`` の総和を返します。
473
474    :math:`O(\\log{n})` です。
475    """
476    assert 0 <= r <= self._size, \
477        f'IndexError: {self.__class__.__name__}.pref({r}), n={self._size}'
478    ret, _tree = 0, self._tree
479    while r > 0:
480      ret += _tree[r]
481      r &= r - 1
482    return ret
483
484  def suff(self, l: int) -> int:
485    """区間 ``[l, n)`` の総和を返します。
486
487    :math:`O(\\log{n})` です。
488    """
489    assert 0 <= l < self._size, \
490        f'IndexError: {self.__class__.__name__}.suff({l}), n={self._size}'
491    return self.pref(self._size) - self.pref(l)
492
493  def sum(self, l: int, r: int) -> int:
494    """区間 ``[l, r)`` の総和を返します。
495
496    :math:`O(\\log{n})` です。
497    """
498    assert 0 <= l <= r <= self._size, \
499        f'IndexError: {self.__class__.__name__}.sum({l}, {r}), n={self._size}'
500    _tree = self._tree
501    res = 0
502    while r > l:
503      res += _tree[r]
504      r &= r - 1
505    while l > r:
506      res -= _tree[l]
507      l &= l - 1
508    return res

510  prod = sum

512  def __getitem__(self, k: int) -> int:
513    """位置 ``k`` の要素を返します。
514
515    :math:`O(\\log{n})` です。
516    """
517    assert -self._size <= k < self._size, \
518        f'IndexError: {self.__class__.__name__}[{k}], n={self._size}'
519    if k < 0:
520      k += self._size
521    return self.sum(k, k+1)
522
523  def add(self, k: int, x: int) -> None:
524    """``k`` 番目の値に ``x`` を加えます。
525
526    :math:`O(\\log{n})` です。
527    """
528    assert 0 <= k < self._size, \
529        f'IndexError: {self.__class__.__name__}.add({k}, {x}), n={self._size}'
530    k += 1
531    _tree = self._tree
532    while k <= self._size:
533      _tree[k] += x
534      k += k & -k
535
536  def __setitem__(self, k: int, x: int):
537    """``k`` 番目の値を ``x`` に更新します。
538
539    :math:`O(\\log{n})` です。
540    """
541    assert -self._size <= k < self._size, \
542        f'IndexError: {self.__class__.__name__}[{k}] = {x}, n={self._size}'
543    if k < 0: k += self._size
544    pre = self[k]
545    self.add(k, x - pre)
546
547  def bisect_left(self, w: int) -> Optional[int]:
548    i, s, _size, _tree = 0, self._s, self._size, self._tree
549    while s:
550      if i + s <= _size and _tree[i + s] < w:
551        w -= _tree[i + s]
552        i += s
553      s >>= 1
554    return i if w else None
555
556  def bisect_right(self, w: int) -> int:
557    i, s, _size, _tree = 0, self._s, self._size, self._tree
558    while s:
559      if i + s <= _size and _tree[i + s] <= w:
560        w -= _tree[i + s]
561        i += s
562      s >>= 1
563    return i
564
565  def _pop(self, k: int) -> int:
566    assert k >= 0
567    i, acc, s, _size, _tree = 0, 0, self._s, self._size, self._tree
568    while s:
569      if i+s <= _size:
570        if acc + _tree[i+s] <= k:
571          acc += _tree[i+s]
572          i += s
573        else:
574          _tree[i+s] -= 1
575      s >>= 1
576    return i

578  def show(self) -> None:
579    print('[' + ', '.join(map(str, (self.pref(i) for i in range(self._size+1)))) + ']')
580
581  def tolist(self) -> List[int]:
582    sub = [self.pref(i) for i in range(self._size+1)]
583    return [sub[i+1]-sub[i] for i in range(self._size)]

585  @staticmethod
586  def get_inversion_num(a: List[int], compress: bool=False) -> int:
587    inv = 0
588    if compress:
589      a_ = sorted(set(a))
590      z = {e: i for i, e in enumerate(a_)}
591      fw = FenwickTree(len(a_))
592      for i, e in enumerate(a):
593        inv += i - fw.pref(z[e])
594        fw.add(z[e], 1)
595    else:
596      fw = FenwickTree(len(a))
597      for i, e in enumerate(a):
598        inv += i - fw.pref(e)
599        fw.add(e, 1)
600    return inv

612class FenwickTreeWaveletMatrix(WaveletMatrix):
613
614  def __init__(self, sigma: int, pos: List[Tuple[int, int, int]] = []):
615    self.sigma: int = sigma
616    self.log: int = (sigma-1).bit_length()
617    self.mid: array[int] = array('I', bytes(4*self.log))
618    self.xy: List[Tuple[int, int]] = self._zaatsu([(x, y) for x, y, _ in pos])
619    self.y: List[int] = self._zaatsu([y for _, y, _ in pos])
620    self.size: int = len(self.xy)
621    self.v: List[BitVector] = [BitVector(self.size) for _ in range(self.log)]
622    self._build([bisect_left(self.y, y) for _, y in self.xy])
623    ws = [[0]*self.size for _ in range(self.log)]
624    for x, y, w in pos:
625      k = bisect_left(self.xy, (x, y))
626      i_y = bisect_left(self.y, y)
627      for bit in range(self.log-1, -1, -1):
628        if i_y >> bit & 1:
629          k = self.v[bit].rank1(k) + self.mid[bit]
630        else:
631          k = self.v[bit].rank0(k)
632        ws[bit][k] += w
633    self.bit: List[FenwickTree] = [FenwickTree(a) for a in ws]
634
635  def _zaatsu(self, a: List) -> List:
636    if not a:
637      return a
638    a.sort()
639    b = [a[0]]
640    for e in a:
641      if b[-1] == e:
642        continue
643      b.append(e)
644    return b
645
646  def add_point(self, x: int, y: int, w: int) -> None:
647    k = bisect_left(self.xy, (x, y))
648    i_y = bisect_left(self.y, y)
649    for bit in range(self.log-1, -1, -1):
650      if i_y >> bit & 1:
651        k = self.v[bit].rank1(k) + self.mid[bit]
652      else:
653        k = self.v[bit].rank0(k)
654      self.bit[bit].add(k, w)
655
656  def _sum(self, l: int, r: int, x: int) -> int:
657    ans = 0
658    for bit in range(self.log-1, -1, -1):
659      l0, r0 = self.v[bit].rank0(l), self.v[bit].rank0(r)
660      if x >> bit & 1:
661        l += self.mid[bit] - l0
662        r += self.mid[bit] - r0
663        ans += self.bit[bit].sum(l0, r0)
664      else:
665        l, r = l0, r0
666    return ans
667
668  def sum(self, w1: int, w2: int, h1: int, h2: int) -> int:
669    # sum([h1, h2) x [w1, w2))
670    l = bisect_left(self.xy, (w1, 0))
671    r = bisect_left(self.xy, (w2, 0))
672    return self._sum(l, r, bisect_left(self.y, h2)) - self._sum(l, r, bisect_left(self.y, h1))
