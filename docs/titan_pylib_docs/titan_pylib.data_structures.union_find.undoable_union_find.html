  7  def __init__(self, n: int) -> None:
  8    """``n`` 個の要素からなる ``UndoableUnionFind`` を構築します。
  9    :math:`O(n)` です。
 10    """
 11    self._n: int = n
 12    self._parents: List[int] = [-1] * n
 13    self._group_count: int = n
 14    self._history: List[Tuple[int, int]] = []
 15
 16  def root(self, x: int) -> int:
 17    """要素 ``x`` を含む集合の代表元を返します。
 18    :math:`O(\\log{n})` です。
 19    """
 20    while self._parents[x] >= 0:
 21      x = self._parents[x]
 22    return x
 23
 24  def unite(self, x: int, y: int) -> bool:
 25    """要素 ``x`` を含む集合と要素 ``y`` を含む集合を併合します。
 26    :math:`O(\\log{n})` です。
 27
 28    Returns:
 29      bool: もともと同じ集合であれば ``False``、そうでなければ ``True`` を返します。
 30    """
 31    x = self.root(x)
 32    y = self.root(y)
 33    if x == y:
 34      self._history.append((-1, -1))
 35      return False
 36    if self._parents[x] > self._parents[y]:
 37      x, y = y, x
 38    self._group_count -= 1
 39    self._history.append((x, self._parents[x]))
 40    self._history.append((y, self._parents[y]))
 41    self._parents[x] += self._parents[y]
 42    self._parents[y] = x
 43    return True
 44
 45  def undo(self) -> None:
 46    """直前の ``unite`` クエリを戻します。
 47    :math:`O(\\log{n})` です。
 48    """
 49    assert self._history, f'Error: {self.__class__.__name__}.undo() with non history.'
 50    y, py = self._history.pop()
 51    if y == -1:
 52      return
 53    self._group_count += 1
 54    x, px = self._history.pop()
 55    self._parents[y] = py
 56    self._parents[x] = px
 57
 58  def size(self, x: int) -> int:
 59    """要素 ``x`` を含む集合の要素数を返します。
 60    :math:`O(\\log{n})` です。
 61    """
 62    return -self._parents[self.root(x)]
 63
 64  def same(self, x: int, y: int) -> bool:
 65    """
 66    要素 ``x`` と ``y`` が同じ集合に属するなら ``True`` を、
 67    そうでないなら ``False`` を返します。
 68    :math:`O(\\log{n})` です。
 69    """
 70    return self.root(x) == self.root(y)
 71
 72  def all_roots(self) -> List[int]:
 73    """全ての集合の代表元からなるリストを返します。
 74    :math:`O(n)` です。
 75
 76    Returns:
 77        List[int]: 昇順であることが保証されます。
 78    """
 79    return [i for i, x in enumerate(self._parents) if x < 0]
 80
 81  def all_group_members(self) -> defaultdict:
 82    """
 83    `key` に代表元、 `value` に `key` を代表元とする集合のリストをもつ `defaultdict` を返します。
 84    :math:`O(n\\log{n})` です。
 85    """
 86    group_members = defaultdict(list)
 87    for member in range(self._n):
 88      group_members[self.root(member)].append(member)
 89    return group_members
 90
 91  def group_count(self) -> int:
 92    """集合の総数を返します。
 93    :math:`O(1)` です。
 94    """
 95    return self._group_count
 96
 97  def clear(self) -> None:
 98    """集合の連結状態をなくします(初期状態に戻します)。
 99    :math:`O(n)` です。
100    """
101    for i in range(self._n):
102      self._parents[i] = -1
103
104  def __str__(self) -> str:
105    """よしなにします。
106
107    :math:`O(n\\log{n})` です。
108    """
109    return f'<{self.__class__.__name__}> [\n' + '\n'.join(f'  {k}: {v}' for k, v in self.all_group_members().items()) + '\n]'
