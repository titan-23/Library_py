 46  def _build(self, a: List[T]) -> None:
 47    Node = LazyAVLTree.Node
 48    id = self.id
 49    def sort(l: int, r: int) -> Node:
 50      mid = (l + r) >> 1

 60  def _propagate(self, node: Node) -> None:
 61    l, r = node.left, node.right
 62    if node.rev:
 63      node.left, node.right = r, l
 64      if l:
 65        l.rev ^= 1
 66      if r:
 67        r.rev ^= 1
 68      node.rev = 0

 72        l.data = self.mapping(lazy, l.data)

178  def _pop_max(self, node: Node) -> Tuple[Node, Node]:
179    self._propagate(node)

219    return LazyAVLTree([], self.op, self.mapping, self.composition, self.e, self.id, l), LazyAVLTree([], self.op, self.mapping, self.composition, self.e, self.id, r)

221  def insert(self, k: int, key: T) -> None:
222    s, t = self._split_node(self.root, k)
223    self.root = self._merge_with_root(s, LazyAVLTree.Node(key, self.id), t)

225  def pop(self, k: int) -> T:
226    s, t = self._split_node(self.root, k+1)
227    s, tmp = self._pop_max(s)
228    self.root = self._merge_node(s, t)
229    return tmp.key

231  def apply(self, l: int, r: int, f: F) -> None:
232    if l >= r or (not self.root): return
233    stack = [(self.root), (self.root, 0, len(self))]
234    while stack:
235      if isinstance(stack[-1], tuple):
236        node, left, right = stack.pop()
237        if right <= l or r <= left: continue
238        self._propagate(node)
239        if l <= left and right < r:
240          node.key = self.mapping(f, node.key)
241          node.data = self.mapping(f, node.data)
242          node.lazy = f if node.lazy == self.id else self.composition(f, node.lazy)
243        else:
244          lsize = node.left.size if node.left else 0
245          stack.append(node)
246          if node.left:
247            stack.append((node.left, left, left+lsize))
248          if l <= left+lsize < r:
249            node.key = self.mapping(f, node.key)
250          if node.right:
251            stack.append((node.right, left+lsize+1, right))
252      else:
253        self._update(stack.pop())
254
255  def all_apply(self, f: F) -> None:
256    if not self.root: return
257    self.root.key = self.mapping(f, self.root.key)
258    self.root.data = self.mapping(f, self.root.data)
259    self.root.lazy = f if self.root.lazy == self.id else self.composition(f, self.root.lazy)
260
261  def reverse(self, l: int, r: int) -> None:
262    if l >= r: return
263    s, t = self._split_node(self.root, r)
264    r, s = self._split_node(s, l)
265    s.rev ^= 1
266    self.root = self._merge_node(self._merge_node(r, s), t)
267
268  def all_reverse(self) -> None:
269    if self.root is None: return
270    self.root.rev ^= 1

272  def prod(self, l: int, r: int) -> T:
273    if l >= r or (not self.root): return self.e
274    def dfs(node: LazyAVLTree.Node, left: int, right: int) -> T:
275      if right <= l or r <= left:
276        return self.e
277      self._propagate(node)
278      if l <= left and right < r:
279        return node.data
280      lsize = node.left.size if node.left else 0
281      res = self.e
282      if node.left:
283        res = dfs(node.left, left, left+lsize)
284      if l <= left+lsize < r:
285        res = self.op(res, node.key)
286      if node.right:
287        res = self.op(res, dfs(node.right, left+lsize+1, right))
288      return res
289    return dfs(self.root, 0, len(self))
290
291  def all_prod(self) -> T:
292    return self.root.data if self.root else self.e

294  def clear(self) -> None:
295    self.root = None

297  def tolist(self) -> List[T]:
298    node = self.root
299    stack = []
300    a = []
301    while stack or node:
302      if node:
303        self._propagate(node)
304        stack.append(node)
305        node = node.left
306      else:
307        node = stack.pop()
308        a.append(node.key)
309        node = node.right
310    return a

336  def __setitem__(self, k, key: T):
337    if k < 0: k += len(self)
338    node = self.root
339    path = []
340    while True:
341      self._propagate(node)
342      path.append(node)
343      t = 0 if node.left is None else node.left.size
344      if t == k:
345        node.key = key
346        break

348        k -= t + 1

350      else:
351        node = node.left
