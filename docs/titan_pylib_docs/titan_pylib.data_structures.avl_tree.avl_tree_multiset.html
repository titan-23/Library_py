281  size を持ちます。

284  def __init__(self, a: Iterable[T]=[]):
285    self.root = 0
286    self.key = [0]
287    self.val = [0]
288    self.valsize = [0]
289    self.size = array('I', bytes(4))
290    self.left = array('I', bytes(4))
291    self.right = array('I', bytes(4))
292    self.balance = array('b', bytes(1))
293    self.end = 1
294    if not isinstance(a, list):
295      a = list(a)
296    if a:

299  def _make_node(self, key: T, val: int) -> int:
300    end = self.end
301    if end >= len(self.key):
302      self.key.append(key)
303      self.val.append(val)
304      self.valsize.append(val)
305      self.size.append(1)
306      self.left.append(0)
307      self.right.append(0)
308      self.balance.append(0)
309    else:
310      self.key[end] = key
311      self.val[end] = val
312      self.valsize[end] = val
313    self.end += 1
314    return end

316  def reserve(self, n: int) -> None:
317    a = [0] * n
318    self.key += a
319    self.val += a
320    self.valsize += a
321    a = array('I', bytes(4*n))
322    self.left += a
323    self.right += a
324    self.size += array('I', [1] * n)
325    self.balance += array('b', bytes(n))

327  def _build(self, a: List[T]) -> None:
328    left, right, size, valsize, balance = self.left, self.right, self.size, self.valsize, self.balance
329    def sort(l: int, r: int) -> Tuple[int, int]:
330      mid = (l + r) >> 1
331      node = mid
332      hl, hr = 0, 0
333      if l != mid:
334        left[node], hl = sort(l, mid)
335        size[node] += size[left[node]]
336        valsize[node] += valsize[left[node]]
337      if mid+1 != r:
338        right[node], hr = sort(mid+1, r)
339        size[node] += size[right[node]]
340        valsize[node] += valsize[right[node]]
341      balance[node] = hl - hr
342      return node, max(hl, hr)+1
343    if not all(a[i] <= a[i + 1] for i in range(len(a) - 1)):
344      a = sorted(a)
345    if not a:
346      return
347    x, y = BSTMultisetArrayBase[AVLTreeMultiset, T]._rle(a)
348    n = len(x)
349    end = self.end
350    self.end += n
351    self.reserve(n)
352    self.key[end:end+n] = x
353    self.val[end:end+n] = y
354    self.valsize[end:end+n] = y
355    self.root = sort(end, n+end)[0]

357  def _rotate_L(self, node: int) -> int:
358    left, right, size, valsize, balance = self.left, self.right, self.size, self.valsize, self.balance
359    u = left[node]
360    size[u] = size[node]
361    valsize[u] = valsize[node]
362    if left[u] == 0:
363      size[node] -= 1
364      valsize[node] -= self.val[u]
365    else:
366      size[node] -= size[left[u]] + 1
367      valsize[node] -= valsize[left[u]] + self.val[u]
368    left[node] = right[u]
369    right[u] = node
370    if balance[u] == 1:
371      balance[u] = 0
372      balance[node] = 0
373    else:
374      balance[u] = -1
375      balance[node] = 1
376    return u
377
378  def _rotate_R(self, node: int) -> int:
379    left, right, size, valsize, balance = self.left, self.right, self.size, self.valsize, self.balance
380    u = right[node]
381    size[u] = size[node]
382    valsize[u] = valsize[node]
383    if right[u] == 0:
384      size[node] -= 1
385      valsize[node] -= self.val[u]
386    else:
387      size[node] -= size[right[u]] + 1
388      valsize[node] -= valsize[right[u]] + self.val[u]
389    right[node] = left[u]
390    left[u] = node
391    if balance[u] == -1:
392      balance[u] = 0
393      balance[node] = 0
394    else:
395      balance[u] = 1
396      balance[node] = -1
397    return u

399  def _update_balance(self, node: int) -> None:
400    left, right, balance = self.left, self.right, self.balance
401    if balance[node] == 1:
402      balance[right[node]] = -1
403      balance[left[node]] = 0
404    elif balance[node] == -1:
405      balance[right[node]] = 0
406      balance[left[node]] = 1
407    else:
408      balance[right[node]] = 0
409      balance[left[node]] = 0
410    balance[node] = 0
411
412  def _rotate_LR(self, node: int) -> int:
413    left, right, size, valsize = self.left, self.right, self.size, self.valsize
414    B = left[node]
415    E = right[B]
416    size[E] = size[node]
417    valsize[E] = valsize[node]
418    if right[E] == 0:
419      size[node] -= size[B]
420      valsize[node] -= valsize[B]
421      size[B] -= 1
422      valsize[B] -= self.val[E]
423    else:
424      size[node] -= size[B] - size[right[E]]
425      valsize[node] -= valsize[B] - valsize[right[E]]
426      size[B] -= size[right[E]] + 1
427      valsize[B] -= valsize[right[E]] + self.val[E]
428    right[B] = left[E]
429    left[E] = B
430    left[node] = right[E]
431    right[E] = node
432    self._update_balance(E)
433    return E
434
435  def _rotate_RL(self, node: int) -> int:
436    left, right, size, valsize = self.left, self.right, self.size, self.valsize
437    C = right[node]
438    D = left[C]
439    size[D] = size[node]
440    valsize[D] = valsize[node]
441    if left[D] == 0:
442      size[node] -= size[C]
443      valsize[node] -= valsize[C]
444      size[C] -= 1
445      valsize[C] -= self.val[D]
446    else:
447      size[node] -= size[C] - size[left[D]]
448      valsize[node] -= valsize[C] - valsize[left[D]]
449      size[C] -= size[left[D]] + 1
450      valsize[C] -= valsize[left[D]] + self.val[D]
451    left[C] = right[D]
452    right[D] = C
453    right[node] = left[D]
454    left[D] = node
455    self._update_balance(D)
456    return D

461  def _kth_elm_tree(self, k: int) -> Tuple[T, int]:
462    left, right, vals, size = self.left, self.right, self.val, self.size
463    if k < 0:
464      k += self.len_elm()
465    assert 0 <= k < self.len_elm()
466    node = self.root
467    while True:
468      t = size[left[node]]
469      if t == k:
470        return self.key[node], vals[node]
471      if t > k:
472        node = left[node]
473      else:
474        node = right[node]
475        k -= t + 1

477  def _discard(self, node: int, path: List[int], di: int) -> bool:
478    left, right, keys, vals = self.left, self.right, self.key, self.val
479    balance, size, valsize = self.balance, self.size, self.valsize
480    fdi = 0
481    if left[node] and right[node]:
482      path.append(node)
483      di <<= 1
484      di |= 1
485      lmax = left[node]
486      while right[lmax]:
487        path.append(lmax)
488        di <<= 1
489        fdi <<= 1
490        fdi |= 1
491        lmax = right[lmax]
492      lmax_val = vals[lmax]
493      keys[node] = keys[lmax]
494      vals[node] = lmax_val
495      node = lmax
496    cnode = right[node] if left[node] == 0 else left[node]
497    if path:
498      if di & 1:
499        left[path[-1]] = cnode
500      else:
501        right[path[-1]] = cnode
502    else:
503      self.root = cnode
504      return True
505    while path:
506      new_node = 0
507      pnode = path.pop()
508      balance[pnode] -= 1 if di & 1 else -1
509      size[pnode] -= 1
510      valsize[pnode] -= lmax_val if fdi & 1 else 1
511      di >>= 1
512      fdi >>= 1
513      if balance[pnode] == 2:
514        new_node = self._rotate_LR(pnode) if balance[left[pnode]] < 0 else self._rotate_L(pnode)
515      elif balance[pnode] == -2:
516        new_node = self._rotate_RL(pnode) if balance[right[pnode]]> 0 else self._rotate_R(pnode)
517      elif balance[pnode] != 0:
518        break
519      if new_node:
520        if not path:
521          self.root = new_node
522          return
523        if di & 1:
524          left[path[-1]] = new_node
525        else:
526          right[path[-1]] = new_node
527        if balance[new_node] != 0:
528          break
529    while path:
530      pnode = path.pop()
531      size[pnode] -= 1
532      valsize[pnode] -= lmax_val if fdi & 1 else 1
533      fdi >>= 1
534    return True

536  def discard(self, key: T, val: int=1) -> bool:
537    keys, vals, left, right, valsize = self.key, self.val, self.left, self.right, self.valsize
538    path = []
539    di = 0
540    node = self.root
541    while node:
542      if key == keys[node]:
543        break
544      path.append(node)
545      di <<= 1
546      if key < keys[node]:
547        di |= 1
548        node = left[node]
549      else:
550        node = right[node]
551    else:
552      return False
553    if val > vals[node]:
554      val = vals[node] - 1
555      vals[node] -= val
556      valsize[node] -= val
557      for p in path:
558        valsize[p] -= val
559    if vals[node] == 1:
560      self._discard(node, path, di)
561    else:
562      vals[node] -= val
563      valsize[node] -= val
564      for p in path:
565        valsize[p] -= val
566    return True

576  def add(self, key: T, val: int=1) -> None:
577    if self.root == 0:
578      self.root = self._make_node(key, val)
579      return
580    left, right, keys, valsize = self.left, self.right, self.key, self.valsize
581    size, balance = self.size, self.balance
582    node = self.root
583    di = 0
584    path = []
585    while node:
586      if key == keys[node]:
587        self.val[node] += val
588        valsize[node] += val
589        for p in path:
590          valsize[p] += val
591        return
592      path.append(node)
593      di <<= 1
594      if key < keys[node]:
595        di |= 1
596        node = left[node]
597      else:
598        node = right[node]
599    if di & 1:
600      left[path[-1]] = self._make_node(key, val)
601    else:
602      right[path[-1]] = self._make_node(key, val)
603    new_node = 0
604    while path:
605      node = path.pop()
606      size[node] += 1
607      valsize[node] += val
608      balance[node] += 1 if di & 1 else -1
609      di >>= 1
610      if balance[node] == 0:
611        break
612      if balance[node] == 2:
613        new_node = self._rotate_LR(node) if balance[left[node]] < 0 else self._rotate_L(node)
614        break
615      elif balance[node] == -2:
616        new_node = self._rotate_RL(node) if balance[right[node]]> 0 else self._rotate_R(node)
617        break
618    if new_node:
619      if path:
620        if di & 1:
621          left[path[-1]] = new_node
622        else:
623          right[path[-1]] = new_node
624      else:
625        self.root = new_node
626    for p in path:
627      size[p] += 1
628      valsize[p] += val

700  def pop(self, k: int=-1) -> T:
701    left, right, vals, valsize = self.left, self.right, self.val, self.valsize
702    keys = self.key
703    node = self.root
704    if k < 0:
705      k += valsize[node]
706    path = []
707    if k == valsize[node]-1:
708      while right[node]:
709        path.append(node)
710        node = right[node]
711      x = keys[node]
712      if vals[node] == 1:
713        self._discard(node, path, 0)
714      else:
715        vals[node] -= 1
716        valsize[node] -= 1
717        for p in path:
718          valsize[p] -= 1
719      return x
720    di = 0
721    while True:
722      t = vals[node] + valsize[left[node]]
723      if t-vals[node] <= k < t:
724        x = keys[node]
725        break
726      path.append(node)
727      di <<= 1
728      if t > k:
729        di |= 1
730        node = left[node]
731      else:
732        node = right[node]
733        k -= t
734    if vals[node] == 1:
735      self._discard(node, path, di)
736    else:
737      vals[node] -= 1
738      valsize[node] -= 1
739      for p in path:
740        valsize[p] -= 1
741    return x
