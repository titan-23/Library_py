 102  def _build(self, a: Iterable[int]) -> None:
 103    key, bit_len, left, right, size, balance, total = self.key, self.bit_len, self.left, self.right, self.size, self.balance, self.total
 104    _popcount = AVLTreeBitVector._popcount
 105    def rec(lr: int) -> int:
 106      l, r = lr>>bit, lr&msk
 107      mid = (l + r) >> 1
 108      hl, hr = 0, 0
 109      if l != mid:
 110        le = rec(l<<bit|mid)
 111        left[mid], hl = le>>bit, le&msk
 112        size[mid] += size[left[mid]]
 113        total[mid] += total[left[mid]]
 114      if mid + 1 != r:
 115        ri = rec((mid+1)<<bit|r)
 116        right[mid], hr = ri>>bit, ri&msk
 117        size[mid] += size[right[mid]]
 118        total[mid] += total[right[mid]]
 119      balance[mid] = hl - hr
 120      return mid<<bit|(max(hl, hr)+1)
 121    if not isinstance(a, Sequence):
 122      a = list(a)
 123    n = len(a)
 124    bit = n.bit_length() + 2
 125    msk = (1 << bit) - 1
 126    end = self.end
 127    self.reserve(n)
 128    i = 0
 129    indx = end
 130    for i in range(0, n, titan_pylib_AVLTreeBitVector_W):
 131      j = 0
 132      v = 0
 133      while j < titan_pylib_AVLTreeBitVector_W and i + j < n:
 134        v <<= 1
 135        v |= a[i+j]
 136        j += 1
 137      key[indx] = v
 138      bit_len[indx] = j
 139      size[indx] = j
 140      total[indx] = _popcount(v)
 141      indx += 1
 142    self.end = indx
 143    self.root = rec(end<<bit|self.end)>>bit

 226  def _pref(self, r: int) -> int:
 227    left, right, bit_len, size, key, total = self.left, self.right, self.bit_len, self.size, self.key, self.total
 228    node = self.root
 229    s = 0
 230    while r > 0:
 231      t = size[left[node]] + bit_len[node]
 232      if t - bit_len[node] < r <= t:
 233        r -= size[left[node]]
 234        s += total[left[node]] + AVLTreeBitVector._popcount(key[node] >> (bit_len[node] - r))
 235        break
 236      if t > r:
 237        node = left[node]
 238      else:
 239        s += total[left[node]] + AVLTreeBitVector._popcount(key[node])
 240        node = right[node]
 241        r -= t
 242    return s

 262  def insert(self, k: int, key: int) -> None:
 263    """``k`` 番目に ``v`` を挿入します。
 264
 265    :math:`O(\\log{n})` です。
 266
 267    Args:
 268      k (int): 挿入位置のインデックスです。
 269      key (int): 挿入する値です。 ``0`` または ``1`` である必要があります。
 270    """
 271    if self.root == 0:
 272      self.root = self._make_node(key, 1)
 273      return
 274    left, right, size, bit_len, balance, keys, total = self.left, self.right, self.size, self.bit_len, self.balance, self.key, self.total
 275    node = self.root
 276    path = []
 277    d = 0
 278    while node:
 279      t = size[left[node]] + bit_len[node]
 280      if t - bit_len[node] <= k <= t:
 281        break
 282      d <<= 1
 283      size[node] += 1
 284      total[node] += key
 285      path.append(node)
 286      node = left[node] if t > k else right[node]
 287      if t > k:
 288        d |= 1
 289      else:
 290        k -= t
 291    k -= size[left[node]]
 292    if bit_len[node] < titan_pylib_AVLTreeBitVector_W:
 293      v = keys[node]
 294      bl = bit_len[node] - k
 295      keys[node] = (((v >> bl) << 1 | key) << bl) | (v & ((1<<bl)-1))
 296      bit_len[node] += 1
 297      size[node] += 1
 298      total[node] += key
 299      return
 300    path.append(node)
 301    size[node] += 1
 302    total[node] += key
 303    v = keys[node]
 304    bl = titan_pylib_AVLTreeBitVector_W - k
 305    v = (((v >> bl) << 1 | key) << bl) | (v & ((1<<bl)-1))
 306    left_key = v >> titan_pylib_AVLTreeBitVector_W
 307    left_key_popcount = left_key & 1
 308    keys[node] = v & ((1 << titan_pylib_AVLTreeBitVector_W) - 1)
 309    node = left[node]
 310    d <<= 1
 311    d |= 1
 312    if not node:
 313      if bit_len[path[-1]] < titan_pylib_AVLTreeBitVector_W:
 314        bit_len[path[-1]] += 1
 315        keys[path[-1]] = (keys[path[-1]] << 1) | left_key
 316        return
 317      else:
 318        left[path[-1]] = self._make_node(left_key, 1)
 319    else:
 320      path.append(node)
 321      size[node] += 1
 322      total[node] += left_key_popcount
 323      d <<= 1
 324      while right[node]:
 325        node = right[node]
 326        path.append(node)
 327        size[node] += 1
 328        total[node] += left_key_popcount
 329        d <<= 1
 330      if bit_len[node] < titan_pylib_AVLTreeBitVector_W:
 331        bit_len[node] += 1
 332        keys[node] = (keys[node] << 1) | left_key
 333        return
 334      else:
 335        right[node] = self._make_node(left_key, 1)
 336    new_node = 0
 337    while path:
 338      node = path.pop()
 339      balance[node] += 1 if d & 1 else -1
 340      d >>= 1
 341      if balance[node] == 0:
 342        break
 343      if balance[node] == 2:
 344        new_node = self._rotate_LR(node) if balance[left[node]] == -1 else self._rotate_L(node)
 345        break
 346      elif balance[node] == -2:
 347        new_node = self._rotate_RL(node) if balance[right[node]] == 1 else self._rotate_R(node)
 348        break
 349    if new_node:
 350      if path:
 351        if d & 1:
 352          left[path[-1]] = new_node
 353        else:
 354          right[path[-1]] = new_node
 355      else:
 356        self.root = new_node

 358  def _pop_under(self, path: List[int], d: int, node: int, res: int) -> None:
 359    left, right, size, bit_len, balance, keys, total = self.left, self.right, self.size, self.bit_len, self.balance, self.key, self.total
 360    fd, lmax_total, lmax_bit_len = 0, 0, 0
 361    if left[node] and right[node]:
 362      path.append(node)
 363      d <<= 1
 364      d |= 1
 365      lmax = left[node]
 366      while right[lmax]:
 367        path.append(lmax)
 368        d <<= 1
 369        fd <<= 1
 370        fd |= 1
 371        lmax = right[lmax]
 372      lmax_total = AVLTreeBitVector._popcount(keys[lmax])
 373      lmax_bit_len = bit_len[lmax]
 374      keys[node] = keys[lmax]
 375      bit_len[node] = lmax_bit_len
 376      node = lmax
 377    cnode = right[node] if left[node] == 0 else left[node]
 378    if path:
 379      if d & 1:
 380        left[path[-1]] = cnode
 381      else:
 382        right[path[-1]] = cnode
 383    else:
 384      self.root = cnode
 385      return
 386    while path:
 387      new_node = 0
 388      node = path.pop()
 389      balance[node] -= 1 if d & 1 else -1
 390      size[node] -= lmax_bit_len if fd & 1 else 1
 391      total[node] -= lmax_total if fd & 1 else res
 392      d >>= 1
 393      fd >>= 1
 394      if balance[node] == 2:
 395        new_node = self._rotate_LR(node) if balance[left[node]] < 0 else self._rotate_L(node)
 396      elif balance[node] == -2:
 397        new_node = self._rotate_RL(node) if balance[right[node]] > 0 else self._rotate_R(node)
 398      elif balance[node] != 0:
 399        break
 400      if new_node:
 401        if not path:
 402          self.root = new_node
 403          return
 404        if d & 1:
 405          left[path[-1]] = new_node
 406        else:
 407          right[path[-1]] = new_node
 408        if balance[new_node] != 0:
 409          break
 410    while path:
 411      node = path.pop()
 412      size[node] -= lmax_bit_len if fd & 1 else 1
 413      total[node] -= lmax_total if fd & 1 else res
 414      fd >>= 1

 416  def pop(self, k: int) -> int:
 417    """``k`` 番目の要素を削除し、その値を返します。
 418
 419    :math:`O(\\log{n})` です。
 420
 421    Args:
 422      k (int): 削除位置のインデックスです。
 423    """
 424    assert 0 <= k < len(self)
 425    left, right, size = self.left, self.right, self.size
 426    bit_len, keys, total = self.bit_len, self.key, self.total
 427    node = self.root
 428    d = 0
 429    path = []
 430    while node:
 431      t = size[left[node]] + bit_len[node]
 432      if t - bit_len[node] <= k < t:
 433        break
 434      path.append(node)
 435      node = left[node] if t > k else right[node]
 436      d <<= 1
 437      if t > k:
 438        d |= 1
 439      else:
 440        k -= t
 441    k -= size[left[node]]
 442    v = keys[node]
 443    res = v >> (bit_len[node] - k - 1) & 1
 444    if bit_len[node] == 1:
 445      self._pop_under(path, d, node, res)
 446      return res
 447    keys[node] = ((v >> (bit_len[node]-k)) << ((bit_len[node]-k-1))) | (v & ((1<<(bit_len[node]-k-1))-1))
 448    bit_len[node] -= 1
 449    size[node] -= 1
 450    total[node] -= res
 451    for p in path:
 452      size[p] -= 1
 453      total[p] -= res
 454    return res

 529  def select0(self, k: int) -> int:
 530    """``k`` 番目の ``0`` のインデックスを返します。
 531
 532    :math:`O(\\log{n}^2)` です。
 533    """
 534    if k < 0 or self.rank0(len(self)) <= k:
 535      return -1
 536    l, r = 0, len(self)
 537    while r - l > 1:
 538      m = (l + r) >> 1
 539      if m - self._pref(m) > k:
 540        r = m
 541      else:
 542        l = m
 543    return l

 605    v = keys[node]
 606    bl = titan_pylib_AVLTreeBitVector_W - k
 607    v = (((v >> bl) << 1 | key) << bl) | (v & ((1<<bl)-1))
 608    left_key = v >> titan_pylib_AVLTreeBitVector_W
 609    left_key_popcount = left_key & 1
 610    keys[node] = v & ((1 << titan_pylib_AVLTreeBitVector_W) - 1)
 611    node = left[node]
 612    d <<= 1
 613    d |= 1
 614    if not node:
 615      if bit_len[path[-1]] < titan_pylib_AVLTreeBitVector_W:
 616        bit_len[path[-1]] += 1
 617        keys[path[-1]] = (keys[path[-1]] << 1) | left_key
 618        return s
 619      else:
 620        left[path[-1]] = self._make_node(left_key, 1)
 621    else:
 622      path.append(node)
 623      size[node] += 1
 624      total[node] += left_key_popcount
 625      d <<= 1
 626      while right[node]:
 627        node = right[node]
 628        path.append(node)
 629        size[node] += 1
 630        total[node] += left_key_popcount
 631        d <<= 1
 632      if bit_len[node] < titan_pylib_AVLTreeBitVector_W:
 633        bit_len[node] += 1
 634        keys[node] = (keys[node] << 1) | left_key
 635        return s
 636      else:
 637        right[node] = self._make_node(left_key, 1)
 638    new_node = 0
 639    while path:
 640      node = path.pop()
 641      balance[node] += 1 if d & 1 else -1
 642      d >>= 1
 643      if balance[node] == 0:
 644        break
 645      if balance[node] == 2:
 646        new_node = self._rotate_LR(node) if balance[left[node]] == -1 else self._rotate_L(node)
 647        break
 648      elif balance[node] == -2:
 649        new_node = self._rotate_RL(node) if balance[right[node]] == 1 else self._rotate_R(node)
 650        break
 651    if new_node:
 652      if path:
 653        if d & 1:
 654          left[path[-1]] = new_node
 655        else:
 656          right[path[-1]] = new_node
 657      else:
 658        self.root = new_node
 659    return s

 661  def _access_pop_and_rank1(self, k: int) -> int:
 662    assert 0 <= k < len(self)
 663    left, right, size = self.left, self.right, self.size
 664    bit_len, keys, total = self.bit_len, self.key, self.total
 665    s = 0
 666    node = self.root
 667    d = 0
 668    path = []
 669    while node:
 670      t = size[left[node]] + bit_len[node]
 671      if t - bit_len[node] <= k < t:
 672        break
 673      if t <= k:
 674        s += total[left[node]] + AVLTreeBitVector._popcount(keys[node])
 675      path.append(node)
 676      node = left[node] if t > k else right[node]
 677      d <<= 1
 678      if t > k:
 679        d |= 1
 680      else:
 681        k -= t
 682    k -= size[left[node]]
 683    s += total[left[node]] + AVLTreeBitVector._popcount(keys[node] >> (bit_len[node] - k))
 684    v = keys[node]
 685    res = v >> (bit_len[node] - k - 1) & 1
 686    if bit_len[node] == 1:
 687      self._pop_under(path, d, node, res)
 688      return s << 1 | res
 689    keys[node] = ((v >> (bit_len[node]-k)) << ((bit_len[node]-k-1))) | (v & ((1<<(bit_len[node]-k-1))-1))
 690    bit_len[node] -= 1
 691    size[node] -= 1
 692    total[node] -= res
 693    for p in path:
 694      size[p] -= 1
 695      total[p] -= res
 696    return s << 1 | res

 749  def __init__(self, n: int):
 750    """長さ ``n`` の ``BitVector`` です。
 751
 752    bit を保持するのに ``array[I]`` を使用します。
 753    ``block_size= n / 32`` として、使用bitは ``32*block_size=2n bit`` です。
 754
 755    累積和を保持するのに同様の ``array[I]`` を使用します。
 756    32bitごとの和を保存しています。同様に使用bitは ``2n bit`` です。
 757    """
 758    assert 0 <= n < 4294967295
 759    self.N = n
 760    self.block_size = (n + 31) >> 5
 761    # bit数 32*n/32 * 2 = 2n bit
 762    b = bytes(4*(self.block_size+1))
 763    self.bit = array('I', b)
 764    self.acc = array('I', b)

 766  @staticmethod
 767  def _popcount(x: int) -> int:
 768    x = x - ((x >> 1) & 0x55555555)
 769    x = (x & 0x33333333) + ((x >> 2) & 0x33333333)
 770    x = x + (x >> 4) & 0x0f0f0f0f
 771    x += x >> 8
 772    x += x >> 16
 773    return x & 0x0000007f

 908  def __init__(self, sigma: int, a: Sequence[int]=[]):
 909    """``[0, sigma)`` の整数列を管理する ``WaveletMatrix`` を構築します。
 910    :math:`O(n\\log{\\sigma})` です。
 911
 912    Args:
 913      sigma (int): 扱う整数の上限です。
 914      a (Sequence[int], optional): 構築する配列です。
 915    """
 916    self.sigma: int = sigma
 917    self.log: int = (sigma-1).bit_length()
 918    self.mid: array[int] = array('I', bytes(4*self.log))
 919    self.size: int = len(a)
 920    self.v: List[BitVector] = [BitVector(self.size) for _ in range(self.log)]
 921    self._build(a)
 922
 923  def _build(self, a: Sequence[int]) -> None:
 924    # 列 a から wm を構築する
 925    for bit in range(self.log-1, -1, -1):
 926      # bit目の0/1に応じてvを構築 + aを安定ソート
 927      v = self.v[bit]
 928      zero, one = [], []
 929      for i, e in enumerate(a):
 930        if e >> bit & 1:
 931          v.set(i)
 932          one.append(e)
 933        else:
 934          zero.append(e)
 935      v.build()
 936      self.mid[bit] = len(zero)  # 境界をmid[bit]に保持
 937      a = zero + one

 939  def access(self, k: int) -> int:
 940    """k番目の値を返します。
 941    :math:`O(\\log{\\sigma})` です。
 942
 943    Args:
 944      k (int): インデックスです。
 945    """
 946    assert -self.size <= k < self.size, \
 947        f'IndexError: {self.__class__.__name__}.access({k}), size={self.size}'
 948    if k < 0:
 949      k += self.size
 950    s = 0  # 答え
 951    for bit in range(self.log-1, -1, -1):
 952      if self.v[bit].access(k):
 953        # k番目が立ってたら、
 954        # kまでの1とすべての0が次のk
 955        s |= 1 << bit
 956        k = self.v[bit].rank1(k) + self.mid[bit]
 957      else:
 958        # kまでの0が次のk
 959        k = self.v[bit].rank0(k)
 960    return s

 966  def rank(self, r: int, x: int) -> int:
 967    """``a[0, r)`` に含まれる ``x`` の個数を返します。
 968    :math:`O(\\log{\\sigma})` です。
 969    """
 970    assert 0 <= r <= self.size, f'IndexError: {self.__class__.__name__}.rank(), r={r}, size={self.size}'
 971    assert 0 <= x < 1<<self.log, f'ValueError: {self.__class__.__name__}.rank(), x={x}, LIM={1<<self.log}'
 972    l = 0
 973    mid = self.mid
 974    for bit in range(self.log-1, -1, -1):
 975      # 位置 r より左に x が何個あるか
 976      # x の bit 目で場合分け
 977      if x >> bit & 1:
 978        # 立ってたら、次のl, rは以下
 979        l = self.v[bit].rank1(l) + mid[bit]
 980        r = self.v[bit].rank1(r) + mid[bit]
 981      else:
 982        # そうでなければ次のl, rは以下
 983        l = self.v[bit].rank0(l)

 988    """``k`` 番目の ``v`` のインデックスを返します。
 989    :math:`O(\\log{\\sigma})` です。
 990    """
 991    assert 0 <= k < self.size, f'IndexError: {self.__class__.__name__}.select({k}, {x}), k={k}, size={self.size}'
 992    assert 0 <= x < 1<<self.log, f'ValueError: {self.__class__.__name__}.select({k}, {x}), x={x}, LIM={1<<self.log}'
 993    # x の開始位置 s を探す
 994    s = 0
 995    for bit in range(self.log-1, -1, -1):
 996      if x >> bit & 1:
 997        s = self.v[bit].rank0(self.size) + self.v[bit].rank1(s)
 998      else:

1024        r = r - r0 + mid[bit]

1041  def topk(self, l: int, r: int, k: int) -> List[Tuple[int, int]]:
1042    """``a[l, r)`` の中で、要素を出現回数が多い順にその頻度とともに ``k`` 個返します。
1043    :math:`O(\\min(r-l, \\sigam) \\log(\\sigam))` です。
1044
1045    Note:
1046      :math:`\\sigma` が大きい場合、計算量に注意です。
1047
1048    Returns:
1049      List[Tuple[int, int]]: ``(要素, 頻度)`` を要素とする配列です。
1050    """
1051    assert 0 <= l <= r <= self.size, f'IndexError: {self.__class__.__name__}.topk({l}, {r}, {k}), size={self.size}'
1052    assert 0 <= k < r-l, f'IndexError: {self.__class__.__name__}.topk({l}, {r}, {k}), wrong k'
1053    # heap[-length, x, l, bit]
1054    hq: List[Tuple[int, int, int, int]] = [(-(r-l), 0, l, self.log-1)]
1055    ans = []
1056    while hq:
1057      length, x, l, bit = heappop(hq)
1058      length = -length
1059      if bit == -1:
1060        ans.append((x, length))
1061        k -= 1
1062        if k == 0:
1063          break
1064      else:
1065        r = l + length
1066        l0 = self.v[bit].rank0(l)
1067        r0 = self.v[bit].rank0(r)
1068        if l0 < r0:
1069          heappush(hq, (-(r0-l0), x, l0, bit-1))
1070        l1 = self.v[bit].rank1(l) + self.mid[bit]
1071        r1 = self.v[bit].rank1(r) + self.mid[bit]
1072        if l1 < r1:
1073          heappush(hq, (-(r1-l1), x|(1<<bit), l1, bit-1))
1074    return ans

1084  def _range_freq(self, l: int, r: int, x: int) -> int:
1085    '''a[l, r) で x 未満の要素の数を返す'''
1086    ans = 0
1087    for bit in range(self.log-1, -1, -1):
1088      l0, r0 = self.v[bit].rank0(l), self.v[bit].rank0(r)
1089      if x >> bit & 1:
1090        # bit が立ってたら、区間の 0 の個数を答えに加算し、新たな区間は 1 のみ
1091        ans += r0 - l0
1092        # 1 が次の bit 列でどこに行くか
1093        l += self.mid[bit] - l0
1094        r += self.mid[bit] - r0
1095      else:
1096        # 0 が次の bit 列でどこに行くか
1097        l, r = l0, r0
1098    return ans

1155    self.log: int = (sigma-1).bit_length()
1156    self.v: List[AVLTreeBitVector] = [AVLTreeBitVector()] * self.log

1161  def _build(self, a: Sequence[int]) -> None:
1162    v = array('B', bytes(self.size))
1163    for bit in range(self.log-1, -1, -1):
1164      # bit目の0/1に応じてvを構築 + aを安定ソート
1165      zero, one = [], []
1166      for i, e in enumerate(a):

1168          v[i] = 1

1170        else:
1171          v[i] = 0
1172          zero.append(e)
