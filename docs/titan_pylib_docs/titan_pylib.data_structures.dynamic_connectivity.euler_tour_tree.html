  7class EulerTourTree(Generic[T, F]):
  8  """``Euler Tour Tree`` です。部分木クエリの強さに定評があります。
  9
 10  森です。各連結成分(木)は独立なのでそれごとに見ていきます。
 11
 12  基本戦術はオイラーツアー(Euler tour technique)です。これはググってください。
 13  実は、ある木をオイラーツアーした列と、その木に対して 根の変更 / 辺の追加 / 辺の削除 をした木のオイラーツアーした列との差分は多くはありません。実際に紙に書くとよいでしょう。これをうまいこと管理します。
 14
 15  列を平衡二分木で管理します。ここで、オイラーツアーの列はでは頂点ではなく辺の列とします。例えば、
 16  `[0, 1, 0, 2, 0]`
 17  の頂点列は辺列
 18  `[{0, 0}, {0, 1}, {1, 1}, {1, 0}, {0, 2}, {2, 2}, {2, 0}]`
 19  などとなります。
 20  また、補助データ構造として辺からノードのポインタをたどれる辞書を保持します。
 21  各処理の流れは以下のようになります。
 22
 23  - 根の変更: `reroot(v)`
 24    - 辺 `{v, v}` の頂点の直前で `split` し、それを順に `A, B` とする
 25    - `B` と `A` をこの順にマージする
 26  - 辺の追加: `link(u, v)`
 27    - `reroot(u); reroot(v)`
 28    - `u, v` が属する木(のオイラーツアーした辺の列)をそれぞれ `E1, E2` とする
 29    - `E1, [{u, v}], E2, [{v, u}]` をこの順にマージする
 30  - 辺の削除: `cut(u, v)`
 31    - `reroot(v); reroot(u)`
 32    - `{u, v}, {v, u}` で `split` してできたものを順に `A, B, C` とする。ただし、 `{u, v}` は `A` に含まれ、 `{v, u}` は `C` に含まれる。
 33    - `A` の末尾と `C` の先頭を削除し、 `A` と `C` をこの順にマージする
 34  - 連結性判定: `same(u, v)`
 35    - `u, v` を `splay` して、 `u` の親が `None` じゃなければ連結。 `u == v` のときは別途処理をする。
 36
 37  計算量は、オイラーツアーの管理と辺→ノードの管理に赤黒木を使えば最悪 `O(logN)` です。この実装ではsplay木とハッシュテーブルで管理するので償却 `O(logN)` +期待 `O(1)` です(もとのグラフの頂点数に対してオイラーツアーすると `(元の頂点数)+(辺数)*2?` の頂点ができるので、たしかに `O(元のグラフの頂点数)` ではありますが、定数倍がバカです)。
 38  オイラーツアーがあれば、部分木クエリは簡単に処理できます。

 86          stack.append(to)
 87          to = next(to)

 96  def build(self, G: List[List[int]]) -> None:
 97    """隣接リスト `G` をもとにして、辺を張ります。
 98
 99    :math:`O(n)` です。
100
101    Args:
102      G (List[List[int]]): 隣接リストです。
103
104    Note:
105      ``build`` メソッドを使用する場合は他のメソッドより前に使用しなければなりません。
106    """
107    n, ptr_vertex, ptr_edge, e, id = self.n, self.ptr_vertex, self.ptr_edge, self.e, self.id
108    seen = [0] * n
109    Node = EulerTourTree.Node
110
111    @EulerTourTree.antirec
112    def dfs(v: int, p: int=-1) -> Generator:
113      a.append(v*n+v)
114      for x in G[v]:
115        if x == p:
116          continue
117        a.append(v*n+x)
118        yield dfs(x, v)
119        a.append(x*n+v)
120      yield
121
122    @EulerTourTree.antirec
123    def rec(l: int, r: int) -> Generator:
124      mid = (l + r) >> 1
125      u, v = divmod(a[mid], n)
126      node = ptr_vertex[u] if u == v else Node(e, id)
127      if u == v:
128        seen[u] = 1
129      else:
130        ptr_edge[u*n+v] = node
131      if l != mid:
132        node.left = yield rec(l, mid)
133        node.left.par = node
134      if mid + 1 != r:
135        node.right = yield rec(mid+1, r)
136        node.right.par = node
137      self._update(node)
138      yield node
139
140    for root in range(self.n):
141      if seen[root]:
142        continue
143      a: List[int] = []
144      dfs(root)
145      rec(0, len(a))

188  def _splay(self, node: Node) -> None:
189    self._propagate(node)
190    while node.par is not None and node.par.par is not None:
191      pnode = node.par
192      gnode = pnode.par
193      self._propagate(gnode)
194      self._propagate(pnode)
195      self._propagate(node)
196      node.par = gnode.par
197      if (gnode.left is pnode) == (pnode.left is node):
198        if pnode.left is node:
199          tmp1 = node.right
200          pnode.left = tmp1
201          node.right = pnode
202          pnode.par = node
203          tmp2 = pnode.right

207        else:

209          pnode.right = tmp1
210          node.left = pnode
211          pnode.par = node
212          tmp2 = pnode.left
213          gnode.right = tmp2
214          pnode.left = gnode
215          gnode.par = pnode
216        if tmp1:
217          tmp1.par = pnode
218        if tmp2:
219          tmp2.par = gnode
220      else:
221        if pnode.left is node:
222          tmp1 = node.right
223          pnode.left = tmp1
224          node.right = pnode
225          tmp2 = node.left
226          gnode.right = tmp2
227          node.left = gnode
228          pnode.par = node
229          gnode.par = node
230        else:
231          tmp1 = node.left
232          pnode.right = tmp1
233          node.left = pnode
234          tmp2 = node.right
235          gnode.left = tmp2
236          node.right = gnode
237          pnode.par = node

239        if tmp1:

242          tmp2.par = gnode

244      self._update(pnode)

308  def link(self, u: int, v: int) -> None:
309    """辺 ``{u, v}`` を追加します。
310    :math:`O(\\log{n})` です。
311
312    Note:
313      ``u`` と ``v`` が同じ連結成分であってはいけません。
314    """
315    # add edge{u, v}
316    self.reroot(u)
317    self.reroot(v)
318    assert u*self.n+v not in self.ptr_edge, f'EulerTourTree.link(), {(u, v)} in ptr_edge'
319    assert v*self.n+u not in self.ptr_edge, f'EulerTourTree.link(), {(v, u)} in ptr_edge'
320    uv_node = EulerTourTree.Node(self.e, self.id)
321    vu_node = EulerTourTree.Node(self.e, self.id)
322    self.ptr_edge[u*self.n+v] = uv_node
323    self.ptr_edge[v*self.n+u] = vu_node
324    u_node = self.ptr_vertex[u]
325    v_node = self.ptr_vertex[v]
326    self._merge(u_node, uv_node)
327    self._merge(uv_node, v_node)
328    self._merge(v_node, vu_node)
329    self._group_numbers -= 1

331  def cut(self, u: int, v: int) -> None:
332    """辺 ``{u, v}`` を削除します。
333    :math:`O(\\log{n})` です。
334
335    Note:
336      辺 ``{u, v}`` が存在してなければいけません。
337    """
338    # erace edge{u, v}
339    self.reroot(v)
340    self.reroot(u)
341    assert u*self.n+v in self.ptr_edge, f'EulerTourTree.cut(), {(u, v)} not in ptr_edge'
342    assert v*self.n+u in self.ptr_edge, f'EulerTourTree.cut(), {(v, u)} not in ptr_edge'
343    uv_node = self.ptr_edge.pop(u*self.n+v)
344    vu_node = self.ptr_edge.pop(v*self.n+u)
345    a, _ = self._split_left(uv_node)
346    _, c = self._split_right(vu_node)
347    a = self._pop(a)
348    c = self._popleft(c)
349    self._merge(a, c)
350    self._group_numbers += 1

352  def merge(self, u: int, v: int) -> bool:
353    """
354    頂点 ``u`` と ``v`` が同じ連結成分にいる場合はなにもせず ``False`` を返します。

356    :math:`O(\\log{n})` です。

359      return False

361    return True

417  def subtree_apply(self, v: int, p: int, f: F) -> None:
418    """頂点 ``v`` を根としたときの部分木に ``f`` を作用します。
419
420    ``v`` の親は ``p`` です。
421    ``v`` の親が存在しないときは ``p=-1`` として下さい。
422
423    :math:`O(\\log{n})` です。
424
425    Args:
426      v (int): 根です。
427      p (int): ``v`` の親です。
428      f (F): 作用素です。
429    """
430    if p == -1:
431      v_node = self.ptr_vertex[v]
432      self._splay(v_node)
433      v_node.key = self.mapping(f, v_node.key)
434      v_node.data = self.mapping(f, v_node.data)
435      v_node.lazy = self.composition(f, v_node.lazy)
436      return
437    self.reroot(v)
438    self.reroot(p)
439    assert p*self.n+v in self.ptr_edge, f'EulerTourTree.subtree_apply(), {(p, v)} not in ptr_edge'
440    assert v*self.n+p in self.ptr_edge, f'EulerTourTree.subtree_apply(), {(v, p)} not in ptr_edge'
441    v_node = self.ptr_vertex[v]
442    a, b = self._split_right(self.ptr_edge[p*self.n+v])
443    b, d = self._split_left(self.ptr_edge[v*self.n+p])
444    self._splay(v_node)
445    v_node.key = self.mapping(f, v_node.key)
446    v_node.data = self.mapping(f, v_node.data)
447    v_node.lazy = self.composition(f, v_node.lazy)
448    self._propagate(v_node)
449    self._merge(a, b)
450    self._merge(b, d)

452  def subtree_sum(self, v: int, p: int) -> T:
453    """頂点 ``v`` を根としたときの部分木の総和を返します。
454
455    ``v`` の親は ``p`` です。
456    ``v`` の親が存在しないときは ``p=-1`` として下さい。
457
458    :math:`O(\\log{n})` です。
459
460    Args:
461      v (int): 根です。
462      p (int): ``v`` の親です。
463    """
464    if p == -1:
465      v_node = self.ptr_vertex[v]
466      self._splay(v_node)
467      return v_node.data
468    self.reroot(v)
469    self.reroot(p)
470    assert p*self.n+v in self.ptr_edge, f'EulerTourTree.subtree_sum(), {(p, v)} not in ptr_edge'
471    assert v*self.n+p in self.ptr_edge, f'EulerTourTree.subtree_sum(), {(v, p)} not in ptr_edge'

473    a, b = self._split_right(self.ptr_edge[p*self.n+v])
474    b, d = self._split_left(self.ptr_edge[v*self.n+p])
475    self._splay(v_node)
476    res = v_node.data
477    self._merge(a, b)
478    self._merge(b, d)

class EulerTourTree(n_or_a: int | Iterable[T], op: Callable[[T, T], T], mapping: Callable[[F, T], T], composition: Callable[[F, F], F], e: T, id: F)[ソース]

ベースクラス: Generic[T, F]

Euler Tour Tree です。部分木クエリの強さに定評があります。

森です。各連結成分(木)は独立なのでそれごとに見ていきます。

基本戦術はオイラーツアー(Euler tour technique)です。これはググってください。 実は、ある木をオイラーツアーした列と、その木に対して 根の変更 / 辺の追加 / 辺の削除 をした木のオイラーツアーした列との差分は多くはありません。実際に紙に書くとよいでしょう。これをうまいこと管理します。

列を平衡二分木で管理します。ここで、オイラーツアーの列はでは頂点ではなく辺の列とします。例えば、 [0, 1, 0, 2, 0] の頂点列は辺列 [{0, 0}, {0, 1}, {1, 1}, {1, 0}, {0, 2}, {2, 2}, {2, 0}] などとなります。 また、補助データ構造として辺からノードのポインタをたどれる辞書を保持します。 各処理の流れは以下のようになります。

  • 根の変更: reroot(v) - 辺 {v, v} の頂点の直前で split し、それを順に A, B とする - B と A をこの順にマージする

  • 辺の追加: link(u, v) - reroot(u); reroot(v) - u, v が属する木(のオイラーツアーした辺の列)をそれぞれ E1, E2 とする - E1, [{u, v}], E2, [{v, u}] をこの順にマージする

  • 辺の削除: cut(u, v) - reroot(v); reroot(u) - {u, v}, {v, u} で split してできたものを順に A, B, C とする。ただし、 {u, v} は A に含まれ、 {v, u} は C に含まれる。 - A の末尾と C の先頭を削除し、 A と C をこの順にマージする

  • 連結性判定: same(u, v) - u, v を splay して、 u の親が None じゃなければ連結。 u == v のときは別途処理をする。

計算量は、オイラーツアーの管理と辺→ノードの管理に赤黒木を使えば最悪 O(logN) です。この実装ではsplay木とハッシュテーブルで管理するので償却 O(logN) +期待 O(1) です(もとのグラフの頂点数に対してオイラーツアーすると (元の頂点数)+(辺数)*2? の頂点ができるので、たしかに O(元のグラフの頂点数) ではありますが、定数倍がバカです)。 オイラーツアーがあれば、部分木クエリは簡単に処理できます。
