392  def _build(self, a: Sequence[T]) -> None:
393    Node = RedBlackTreeSet.Node
394    def rec(l: int, r: int, d: int) -> Node:
395      mid = (l + r) >> 1
396      node = Node(a[mid])
397      node.col = int((not flag and d&1) or (flag and d > 1 and not d&1))
398      if l != mid:
399        node.left = rec(l, mid, d+1)
400        node.left.par = node
401      if mid+1 != r:
402        node.right = rec(mid+1, r, d+1)
403        node.right.par = node
404      return node
405    a = BSTSetNodeBase[T, RedBlackTreeSet.Node].sort_unique(a)
406    flag = len(a).bit_length() & 1
407    self.node = rec(0, len(a), 0)
408    self.min_node = self.node._min()
409    self.max_node = self.node._max()

463  def add(self, key: T) -> bool:
464    if not self.node:
465      node = RedBlackTreeSet.Node(key)
466      self.node = node
467      self.min_node = node
468      self.max_node = node

470      return True
471    pnode = RedBlackTreeSet.NIL
472    node = self.node
473    while node:
474      pnode = node
475      if key == node.key:
476        return False
477      node = node.left if key < node.key else node.right
478    self.size += 1
479    z = RedBlackTreeSet.Node(key)
480    if key < self.min_node.key:
481      self.min_node = z
482    if key > self.max_node.key:
483      self.max_node = z
484    z.par = pnode
485    if not pnode:
486      self.node = z
487    elif key < pnode.key:
488      pnode.left = z
489    else:
490      pnode.right = z

493      g = z.par.par
494      if z.par is g.left:
495        y = g.right

501        else:
502          if z is z.par.right:
503            z = z.par
504            self._rotate_left(z)
505          z.par.col = 0
506          g.col = 1
507          self._rotate_right(g)
508          break
509      else:
510        y = g.left
511        if y.col:
512          z.par.col = 0
513          y.col = 0
514          g.col = 1
515          z = g
516        else:
517          if z is z.par.left:

527  def discard_iter(self, node: Node) -> None:
528    """``node`` を削除します。
529
530    償却 :math:`O(1)` らしいです。
531
532    Args:
533      node (Node): 削除する ``node`` です。
534    """
535    assert isinstance(node, RedBlackTreeSet.Node)
536    self.size -= 1
537    if node.key == self.min_node.key:
538      self.min_node = node._next()
539    if node.key == self.max_node.key:
540      self.max_node = node._prev()
541    y = node
542    y_col = y.col
543    if not node.left:
544      x = node.right
545      self._transplant(node, node.right)
546    elif not node.right:
547      x = node.left
548      self._transplant(node, node.left)
549    else:
550      y = self._get_min(node.right)
551      y_col = y.col
552      x = y.right
553      if y.par is node:
554        x.par = y
555      else:
556        self._transplant(y, y.right)
557        y.right = node.right
558        y.right.par = y
559      self._transplant(node, y)
560      y.left = node.left
561      y.left.par = y
562      y.col = node.col
563    if y_col:
564      return
565    while x is not self.node and not x.col:
566      if x is x.par.left:
567        y = x.par
568        w = y.right
569        if w.col:
570          w.col = 0
571          y.col = 1
572          self._rotate_left(y)
573          w = y.right
574        if not (w.left.col or w.right.col):
575          w.col = 1
576          x = y
577        else:
578          if not w.right.col:
579            w.left.col = 0
580            w.col = 1
581            self._rotate_right(w)
582            w = y.right
583          w.col = y.col
584          y.col = 0
585          w.right.col = 0
586          self._rotate_left(x.par)
587          x = self.node
588      else:
589        y = x.par
590        w = y.left
591        if w.col:
592          w.col = 0
593          y.col = 1
594          self._rotate_right(y)
595          w = y.left
596        if not (w.right.col or w.left.col):
597          w.col = 1
598          x = y
599        else:
600          if not w.left.col:
601            w.right.col = 0
602            w.col = 1
603            self._rotate_left(w)
604            w = y.left
605          w.col = y.col
606          y.col = 0
607          w.left.col = 0
608          self._rotate_right(y)
609          x = self.node
610    x.col = 0
