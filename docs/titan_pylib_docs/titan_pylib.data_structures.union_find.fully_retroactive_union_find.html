 30    """
 31    各引数は遅延セグ木のアレです。よしなに。

 34      op (Callable[[T, T], T]): 非可換でも構いません。

 37    self.mapping = mapping
 38    self.composition = composition
 39    self.e = e
 40    self.id = id

 56    return (self.arr[node<<2|2] == self.n) or not (self.arr[self.arr[node<<2|2]<<2] == node or self.arr[self.arr[node<<2|2]<<2|1] == node)

 58  def _propagate(self, node: int) -> None:
 59    if node == self.n: return
 60    arr = self.arr
 61    if arr[node<<2|3]:
 62      self.data[node<<1], self.data[node<<1|1] = self.data[node<<1|1], self.data[node<<1]
 63      arr[node<<2|3] = 0
 64      ln, rn = arr[node<<2], arr[node<<2|1]
 65      arr[node<<2] = rn
 66      arr[node<<2|1] = ln
 67      arr[ln<<2|3] ^= 1
 68      arr[rn<<2|3] ^= 1
 69    if self.lazy[node] == self.id:
 70      return
 71    lazy, data, key = self.lazy, self.data, self.key
 72    nlazy = lazy[node]
 73    lnode, rnode = arr[node<<2], arr[node<<2|1]
 74    if lnode != self.n:
 75      data[lnode<<1] = self.mapping(nlazy, data[lnode<<1])
 76      data[lnode<<1|1] = self.mapping(nlazy, data[lnode<<1|1])
 77      key[lnode] = self.mapping(nlazy, key[lnode])
 78      lazy[lnode] = nlazy if lazy[lnode] == self.id else self.composition(nlazy, lazy[lnode])
 79    if rnode != self.n:
 80      data[rnode<<1] = self.mapping(nlazy, data[rnode<<1])
 81      data[rnode<<1|1] = self.mapping(nlazy, data[rnode<<1|1])
 82      key[rnode] = self.mapping(nlazy, key[rnode])
 83      lazy[rnode] = nlazy if lazy[rnode] == self.id else self.composition(nlazy, lazy[rnode])
 84    lazy[node] = self.id

 86  def _update(self, node: int) -> None:
 87    if node == self.n: return
 88    ln, rn = self.arr[node<<2], self.arr[node<<2|1]
 89    self._propagate(ln)
 90    self._propagate(rn)
 91    self.data[node<<1] = self.op(self.op(self.data[ln<<1], self.key[node]), self.data[rn<<1])
 92    self.data[node<<1|1] = self.op(self.op(self.data[rn<<1|1], self.key[node]), self.data[ln<<1|1])
 93    self.size[node] = 1 + self.size[ln] + self.size[rn]

 95  def _update_triple(self, x: int, y: int, z: int) -> None:
 96    data, key, arr, size = self.data, self.key, self.arr, self.size
 97    lx, rx = arr[x<<2], arr[x<<2|1]
 98    ly, ry = arr[y<<2], arr[y<<2|1]
 99    self._propagate(lx)
100    self._propagate(rx)
101    self._propagate(ly)
102    self._propagate(ry)
103    data[z<<1] = data[x<<1]
104    data[x<<1] = self.op(self.op(data[lx<<1], key[x]), data[rx<<1])
105    data[y<<1] = self.op(self.op(data[ly<<1], key[y]), data[ry<<1])
106    data[z<<1|1] = data[x<<1|1]
107    data[x<<1|1] = self.op(self.op(data[rx<<1|1], key[x]), data[lx<<1|1])
108    data[y<<1|1] = self.op(self.op(data[ry<<1|1], key[y]), data[ly<<1|1])
109    size[z] = size[x]
110    size[x] = 1 + size[lx] + size[rx]
111    size[y] = 1 + size[ly] + size[ry]

113  def _update_double(self, x: int, y: int) -> None:

116    self._propagate(lx)
117    self._propagate(rx)
118    data[y<<1] = data[x<<1]
119    data[x<<1] = self.op(self.op(data[lx<<1], key[x]), data[rx<<1])
120    data[y<<1|1] = data[x<<1|1]
121    data[x<<1|1] = self.op(self.op(data[rx<<1|1], key[x]), data[lx<<1|1])
122    size[y] = size[x]
123    size[x] = 1 + size[lx] + size[rx]

125  def _splay(self, node: int) -> None:
126    # splayを抜けた後、nodeは遅延伝播済みにするようにする
127    # (splay後のnodeのleft,rightにアクセスしやすいと非常にラクなはず)
128    if node == self.n: return
129    _propagate, _is_root, _update_triple = self._propagate, self._is_root, self._update_triple
130    _propagate(node)
131    if _is_root(node): return
132    arr = self.arr
133    pnode = arr[node<<2|2]
134    while not _is_root(pnode):
135      gnode = arr[pnode<<2|2]
136      _propagate(gnode)
137      _propagate(pnode)
138      _propagate(node)
139      f = arr[pnode<<2] == node
140      g = (arr[gnode<<2|f] == pnode) ^ (arr[pnode<<2|f] == node)
141      nnode = (node if g else pnode) << 2 | f ^ g
142      arr[pnode<<2|f^1] = arr[node<<2|f]
143      arr[gnode<<2|f^g^1] = arr[nnode]
144      arr[node<<2|f] = pnode
145      arr[nnode] = gnode
146      arr[node<<2|2] = arr[gnode<<2|2]
147      arr[gnode<<2|2] = nnode>>2
148      arr[arr[pnode<<2|f^1]<<2|2] = pnode
149      arr[arr[gnode<<2|f^g^1]<<2|2] = gnode
150      arr[pnode<<2|2] = node
151      _update_triple(gnode, pnode, node)
152      pnode = arr[node<<2|2]
153      if arr[pnode<<2] == gnode:
154        arr[pnode<<2] = node
155      elif arr[pnode<<2|1] == gnode:
156        arr[pnode<<2|1] = node
157      else:
158        return
159    _propagate(pnode)
160    _propagate(node)
161    f = arr[pnode<<2] == node
162    arr[pnode<<2|f^1] = arr[node<<2|f]
163    arr[node<<2|f] = pnode
164    arr[arr[pnode<<2|f^1]<<2|2] = pnode
165    arr[node<<2|2] = arr[pnode<<2|2]
166    arr[pnode<<2|2] = node
167    self._update_double(pnode, node)

169  def expose(self, v: int) -> int:
170    """`v` が属する木において、その木を管理しているsplay木の根からvまでのパスを作ります。
171
172    償却 :math:`O(\\log{n})` です。
173    """
174    arr, n, _splay, _update = self.arr, self.n, self._splay, self._update
175    pre = v
176    while arr[v<<2|2] != n:
177      _splay(v)
178      arr[v<<2|1] = n
179      _update(v)
180      if arr[v<<2|2] == n:
181        break
182      pre = arr[v<<2|2]
183      _splay(pre)
184      arr[pre<<2|1] = v
185      _update(pre)
186    arr[v<<2|1] = n
187    _update(v)
188    return pre

200  def link(self, c: int, p: int) -> None:
201    """辺 ``{c -> p}`` を追加します。
202
203    償却 :math:`O(\\log{n})` です。

206      ``c`` は元の木の根でなければならないです。

209    self.expose(c)

271  def merge(self, u: int, v: int) -> bool:
272    """
273    ``u``, ``v`` が同じ連結成分なら ``False`` を返します。
274    そうでなければ辺 ``{u -> v}`` を追加して ``True`` を返します。
275
276    償却 :math:`O(\\log{n})` です。
277    """
278
279    if self.same(u, v): return False
280    self.evert(u)
281    self.expose(v)
282    self.arr[u<<2|2] = v
283    self.arr[v<<2|1] = u
284    self._update(v)
285    self.group_cnt -= 1
286    return True

308  def path_apply(self, u: int, v: int, f: F) -> None:
309    """``u`` から ``v`` へのパスに ``f`` を作用させます。
310
311    償却 :math:`O(\\log{n})` です。
312    """
313    self.evert(u)
314    self.expose(v)
315    self.key[v] = self.mapping(f, self.key[v])
316    self.data[v<<1] = self.mapping(f, self.data[v<<1])
317    self.data[v<<1|1] = self.mapping(f, self.data[v<<1|1])
318    self.lazy[v] = f if self.lazy[v] == self.id else self.composition(f, self.lazy[v])
319    self._propagate(v)

330  def path_kth_elm(self, s: int, t: int, k: int) -> int:
331    '''``u`` から ``v`` へ ``k`` 個進んだ頂点を返します。
332    存在しないときは ``-1`` を返します。
333
334    償却 :math:`O(\\log{n})` です。
335    '''
336    self.evert(s)
337    self.expose(t)
338    if self.size[t] <= k:
339      return -1
340    size, arr = self.size, self.arr
341    while True:
342      self._propagate(t)
343      s = size[arr[t<<2]]
344      if s == k:
345        self._splay(t)
346        return t
347      t = arr[t<<2|(s<k)]
348      if s < k:
349        k -= s + 1

377  def __init__(self, n: int, m: int) -> None:
378    """頂点数 ``n`` 、クエリ列の長さ ``m`` の ``FullyRetroactiveUnionFind`` を作ります。
379
380    ここで、クエリは `unite` のみです。
381
382    :math:`O(n+m)` です。
383
384    Args:
385      n (int): 頂点数です。
386      m (int): クエリ列の長さです。
387    """
388    m += 1
389    self.n: int = n
390    self.edge: List[Tuple[int, int, int]] = [()] * m
391    self.node_pool: Set[int] = set(range(n, n+m))
392    self.lct: LinkCutTree[int, None] = LinkCutTree(n+m,
393                                                   op=lambda s, t: s if s > t else t,
394                                                   mapping=lambda f, s: -1,
395                                                   composition=lambda f, g: None,
396                                                   e=-1,
397                                                   id=None)
398
399  def unite(self, u: int, v: int, t: int) -> None:
400    """時刻 ``t`` のクエリを ``unite(u, v)`` にします。
401
402    償却 :math:`O(\\log{(n+m)})` です。
403
404    Args:
405      u (int): 集合の要素です。
406      v (int): 集合の要素です。
407      t (int): 時刻です。
408
409    Note:
410      ``disconnect`` を使用する場合、 ``u``, ``v`` が連結されていてはいけません。
411    """
412    node = self.node_pool.pop()
413    self.edge[t] = (u, v, node)
414    self.lct[node] = t
415    self.lct.merge(u, node)
416    self.lct.merge(node, v)
417
418  def disconnect(self, t: int) -> None:
419    """時刻 ``t`` の連結クエリをなくして、そのクエリの2頂点を非連結にします。
420
421    償却 :math:`O(\\log{(n+m)})` です。
422
423    Args:
424      t (int): 時刻です。
425
426    Note:
427      時刻 ``t`` のクエリは連結クエリでないといけません。
428    """
429    assert self.edge[t] is not None
430    u, v, node = self.edge[t]
431    self.node_pool.add(node)
432    self.edge[t] = None
433    self.lct.split(u, node)
434    self.lct.split(node, v)
435
436  def same(self, u: int, v: int, t: int) -> bool:
437    """時刻 ``t`` で ``u``, ``v`` の連結判定をします。
438
439    償却 :math:`O(\\log{(n+m)})` です。
440
441    Args:
442      u (int): 集合の要素です。
443      v (int): 集合の要素です。
444      t (int): 時刻です。
445
446    Returns:
447      bool:
448    """
449    if not self.lct.same(u, v):
450      return False
451    return self.lct.path_prod(u, v) <= t
