 47  def _build(self, a: List[T]) -> None:
 48    Node = ReversibleLazyAVLTree.Node
 49    id = self.id
 50    def sort(l: int, r: int) -> Node:
 51      mid = (l + r) >> 1
 52      node = Node(a[mid], id)
 53      if l != mid:
 54        node.left = sort(l, mid)
 55      if mid+1 != r:
 56        node.right = sort(mid+1, r)
 57      self._update(node)
 58      return node

 61  def _propagate(self, node: Node) -> None:
 62    l, r = node.left, node.right
 63    if node.rev:
 64      node.data, node.rdata = node.rdata, node.data
 65      node.left, node.right = r, l
 66      if l:
 67        l.rev ^= 1
 68      if r:
 69        r.rev ^= 1
 70      node.rev = 0
 71    if node.lazy != self.id:
 72      lazy = node.lazy
 73      if l:
 74        l.data = self.mapping(lazy, l.data)
 75        l.rdata = self.mapping(lazy, l.rdata)
 76        l.key = self.mapping(lazy, l.key)
 77        l.lazy = lazy if l.lazy == self.id else self.composition(lazy, l.lazy)
 78      if r:

 85  def _update(self, node: Node) -> None:
 86    node.size = 1
 87    node.data = node.key
 88    node.rdata = node.key

156  def _merge_with_root(self, l: Node, root: Node, r: Node) -> Node:
157    diff = (0 if r is None else -r.height) if l is None else (l.height if r is None else l.height-r.height)
158    if diff > 1:
159      self._propagate(l)
160      l.right = self._merge_with_root(l.right, root, r)

187  def _pop_max(self, node: Node) -> Tuple[Node, Node]:
188    self._propagate(node)
189    path = []
190    mx = node
191    while node.right:
192      path.append(node)

197    for _ in range(len(path)-1):

240  def apply(self, l: int, r: int, f: F) -> None:
241    if l >= r or (not self.root): return
242    stack = [(self.root), (self.root, 0, len(self))]
243    while stack:
244      if isinstance(stack[-1], tuple):
245        node, left, right = stack.pop()
246        if right <= l or r <= left: continue
247        self._propagate(node)
248        if l <= left and right < r:
249          node.key = self.mapping(f, node.key)
250          node.data = self.mapping(f, node.data)
251          node.rdata = self.mapping(f, node.rdata)
252          node.lazy = f if node.lazy == self.id else self.composition(f, node.lazy)
253        else:
254          lsize = node.left.size if node.left else 0
255          stack.append(node)
256          if node.left:
257            stack.append((node.left, left, left+lsize))
258          if l <= left+lsize < r:
259            node.key = self.mapping(f, node.key)
260          if node.right:
261            stack.append((node.right, left+lsize+1, right))
262      else:
263        self._update(stack.pop())

285  def prod(self, l: int, r: int) -> T:
286    if l >= r or (not self.root): return self.e
287    def dfs(node: ReversibleLazyAVLTree.Node, left: int, right: int) -> T:
288      if right <= l or r <= left:
289        return self.e

291      if l <= left and right < r:
292        return node.data
