 62  def _popcount(x: int) -> int:
 63    x = x - ((x >> 1) & 0x55555555)
 64    x = (x & 0x33333333) + ((x >> 2) & 0x33333333)
 65    x = x + (x >> 4) & 0x0f0f0f0f
 66    x += x >> 8
 67    x += x >> 16
 68    return x & 0x0000007f

103  def _build(self, a: Iterable[int]) -> None:
104    key, bit_len, left, right, size, balance, total = self.key, self.bit_len, self.left, self.right, self.size, self.balance, self.total
105    _popcount = AVLTreeBitVector._popcount
106    def rec(lr: int) -> int:
107      l, r = lr>>bit, lr&msk
108      mid = (l + r) >> 1
109      hl, hr = 0, 0
110      if l != mid:
111        le = rec(l<<bit|mid)
112        left[mid], hl = le>>bit, le&msk
113        size[mid] += size[left[mid]]
114        total[mid] += total[left[mid]]
115      if mid + 1 != r:
116        ri = rec((mid+1)<<bit|r)
117        right[mid], hr = ri>>bit, ri&msk
118        size[mid] += size[right[mid]]
119        total[mid] += total[right[mid]]
120      balance[mid] = hl - hr
121      return mid<<bit|(max(hl, hr)+1)
122    if not isinstance(a, Sequence):
123      a = list(a)
124    n = len(a)
125    bit = n.bit_length() + 2
126    msk = (1 << bit) - 1
127    end = self.end
128    self.reserve(n)
129    i = 0
130    indx = end
131    for i in range(0, n, titan_pylib_AVLTreeBitVector_W):
132      j = 0
133      v = 0
134      while j < titan_pylib_AVLTreeBitVector_W and i + j < n:
135        v <<= 1
136        v |= a[i+j]
137        j += 1
138      key[indx] = v
139      bit_len[indx] = j
140      size[indx] = j
141      total[indx] = _popcount(v)
142      indx += 1
143    self.end = indx
144    self.root = rec(end<<bit|self.end)>>bit

228    left, right, bit_len, size, key, total = self.left, self.right, self.bit_len, self.size, self.key, self.total

231    while r > 0:

233      if t - bit_len[node] < r <= t:
234        r -= size[left[node]]

238        node = left[node]

266

268      k (int): 挿入位置のインデックスです。

272      self.root = self._make_node(key, 1)

380        left[path[-1]] = cnode

382        right[path[-1]] = cnode
383    else:
384      self.root = cnode
385      return

387      new_node = 0
388      node = path.pop()
389      balance[node] -= 1 if d & 1 else -1
390      size[node] -= lmax_bit_len if fd & 1 else 1
391      total[node] -= lmax_total if fd & 1 else res
392      d >>= 1
393      fd >>= 1
394      if balance[node] == 2:
395        new_node = self._rotate_LR(node) if balance[left[node]] < 0 else self._rotate_L(node)
396      elif balance[node] == -2:

398      elif balance[node] != 0:
399        break

401        if not path:
402          self.root = new_node
403          return
404        if d & 1:

406        else:

409          break

416  def pop(self, k: int) -> int:
417    """``k`` 番目の要素を削除し、その値を返します。
418    :math:`O(\\log{n})` です。
419
420    Args:
421      k (int): 削除位置のインデックスです。
422    """
423    assert 0 <= k < len(self)
424    left, right, size = self.left, self.right, self.size
425    bit_len, keys, total = self.bit_len, self.key, self.total
426    node = self.root
427    d = 0
428    path = []
429    while node:
430      t = size[left[node]] + bit_len[node]
431      if t - bit_len[node] <= k < t:
432        break
433      path.append(node)
434      node = left[node] if t > k else right[node]
435      d <<= 1
436      if t > k:
437        d |= 1
438      else:
439        k -= t
440    k -= size[left[node]]
441    v = keys[node]
442    res = v >> (bit_len[node] - k - 1) & 1
443    if bit_len[node] == 1:
444      self._pop_under(path, d, node, res)
445      return res
446    keys[node] = ((v >> (bit_len[node]-k)) << ((bit_len[node]-k-1))) | (v & ((1<<(bit_len[node]-k-1))-1))
447    bit_len[node] -= 1
448    size[node] -= 1
449    total[node] -= res
450    for p in path:
451      size[p] -= 1
452      total[p] -= res
453    return res

599    path.append(node)
600    size[node] += 1
601    total[node] += key
602    v = keys[node]
603    bl = titan_pylib_AVLTreeBitVector_W - k
604    v = (((v >> bl) << 1 | key) << bl) | (v & ((1<<bl)-1))
605    left_key = v >> titan_pylib_AVLTreeBitVector_W
606    left_key_popcount = left_key & 1
607    keys[node] = v & ((1 << titan_pylib_AVLTreeBitVector_W) - 1)
608    node = left[node]
609    d <<= 1
610    d |= 1
611    if not node:
612      if bit_len[path[-1]] < titan_pylib_AVLTreeBitVector_W:
613        bit_len[path[-1]] += 1
614        keys[path[-1]] = (keys[path[-1]] << 1) | left_key
615        return s
616      else:
617        left[path[-1]] = self._make_node(left_key, 1)
618    else:
619      path.append(node)
620      size[node] += 1
621      total[node] += left_key_popcount
622      d <<= 1
623      while right[node]:
624        node = right[node]
625        path.append(node)
626        size[node] += 1
627        total[node] += left_key_popcount
628        d <<= 1
629      if bit_len[node] < titan_pylib_AVLTreeBitVector_W:
630        bit_len[node] += 1
631        keys[node] = (keys[node] << 1) | left_key
632        return s
633      else:
634        right[node] = self._make_node(left_key, 1)
635    new_node = 0
636    while path:
637      node = path.pop()
638      balance[node] += 1 if d & 1 else -1
639      d >>= 1
640      if balance[node] == 0:
641        break
642      if balance[node] == 2:
643        new_node = self._rotate_LR(node) if balance[left[node]] == -1 else self._rotate_L(node)
644        break
645      elif balance[node] == -2:
646        new_node = self._rotate_RL(node) if balance[right[node]] == 1 else self._rotate_R(node)
647        break
648    if new_node:
649      if path:
650        if d & 1:
651          left[path[-1]] = new_node
652        else:
653          right[path[-1]] = new_node
654      else:
655        self.root = new_node
656    return s
657
658  def _access_pop_and_rank1(self, k: int) -> int:
659    assert 0 <= k < len(self)
660    left, right, size = self.left, self.right, self.size
661    bit_len, keys, total = self.bit_len, self.key, self.total
662    s = 0
663    node = self.root
664    d = 0
665    path = []
666    while node:
667      t = size[left[node]] + bit_len[node]
668      if t - bit_len[node] <= k < t:
669        break
670      if t <= k:
671        s += total[left[node]] + AVLTreeBitVector._popcount(keys[node])
672      path.append(node)
673      node = left[node] if t > k else right[node]
674      d <<= 1
675      if t > k:
676        d |= 1
677      else:
678        k -= t
679    k -= size[left[node]]
680    s += total[left[node]] + AVLTreeBitVector._popcount(keys[node] >> (bit_len[node] - k))
681    v = keys[node]
682    res = v >> (bit_len[node] - k - 1) & 1
683    if bit_len[node] == 1:
684      self._pop_under(path, d, node, res)
685      return s << 1 | res
686    keys[node] = ((v >> (bit_len[node]-k)) << ((bit_len[node]-k-1))) | (v & ((1<<(bit_len[node]-k-1))-1))
687    bit_len[node] -= 1
688    size[node] -= 1
689    total[node] -= res
690    for p in path:
691      size[p] -= 1
692      total[p] -= res
693    return s << 1 | res

695  def __getitem__(self, k: int) -> int:
696    """``k`` 番目の要素を返します。
697    :math:`O(\\log{n})` です。
698    """
699    assert 0 <= k < len(self)
700    left, right, bit_len, size, key = self.left, self.right, self.bit_len, self.size, self.key
701    node = self.root
702    while True:
703      t = size[left[node]] + bit_len[node]
704      if t - bit_len[node] <= k < t:
705        k -= size[left[node]]
706        return key[node] >> (bit_len[node] - k - 1) & 1
707      if t > k:
708        node = left[node]
709      else:
710        node = right[node]
711        k -= t
712
713  def __setitem__(self, k: int, v: int) -> None:
714    """``k`` 番目の要素を ``v`` に更新します。
715    :math:`O(\\log{n})` です。
716    """
717    left, right, bit_len, size, key, total = self.left, self.right, self.bit_len, self.size, self.key, self.total
718    assert v == 0 or v == 1, 'ValueError'
719    node = self.root
720    path = []
721    while True:
722      t = size[left[node]] + bit_len[node]
723      path.append(node)
724      if t - bit_len[node] <= k < t:
725        k -= size[left[node]]
726        if v:
727          key[node] |= 1 << k
728        else:
729          key[node] &= ~(1 << k)
730        break
731      elif t > k:
732        node = left[node]
733      else:
734        node = right[node]
735        k -= t
736    while path:
737      node = path.pop()
738      total[node] = AVLTreeBitVector._popcount(key[node]) + total[left[node]] + total[right[node]]

ベースクラス: BitVectorInterface

AVL木で書かれたビットベクトルです。簡潔でもなんでもありません。

bit列を管理するわけですが、各節点は 1~32 bit を持つようにしています。 これにより、最大 32 倍高速化が行えます。(16~32bitとするといいんだろうけど)
