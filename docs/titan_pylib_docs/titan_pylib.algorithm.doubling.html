 1from typing import Callable, TypeVar, Generic
 2T = TypeVar('T')
 3
 4class Doubling(Generic[T]):
 5  """ダブリングテーブルを構築します。
 6  :math:`O(n\\log{lim})` です。
 7
 8  Args:
 9    Generic (T): 状態の型です。
10    n (int): テーブルサイズです。
11    lim (int): クエリの最大数です。
12    move_to (Callable[[T], T]): 遷移関数です。 ``u`` から ``v`` へ遷移します。
13  """
14
15  def __init__(self, n: int, lim: int, move_to: Callable[[T], T]) -> None:
16    self.move_to = move_to
17    self.n = n
18    self.lim = lim
19    self.log = lim.bit_length()
20    db = [[0]*self.n for _ in range(self.log+1)]
21    for i in range(self.n):
22      db[0][i] = move_to(i)
23    for k in range(self.log):
24      for i in range(self.n):
25        db[k+1][i] = db[k][db[k][i]]
26    self.db = db
27
28  def kth(self, start: T, k: int) -> T:
29    """``start`` から ``k`` 個進んだ状態を返します。
30    :math:`O(\\log{k})` です。
31
32    Args:
33      start (T): スタートの状態です。
34      k (int): 遷移関数を適用する回数です。
35    """
36    now = start
37    for i in range(self.log):
38      if k & 1:
39        now = self.db[i][now]
40      k >>= 1
41    return now
