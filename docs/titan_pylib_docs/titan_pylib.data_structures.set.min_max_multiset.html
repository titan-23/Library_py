 13class DoubleEndedHeap(Generic[T]):
 14  """
 15  - [両端優先度付きキューのInterval-Heap実装](https://natsugiri.hatenablog.com/entry/2016/10/10/035445)
 16  - [Double-ended priority queue(wikipedia)](https://en.wikipedia.org/wiki/Double-ended_priority_queue)
 17  """
 18
 19  def __init__(self, a: Iterable[T]=[]):
 20    """
 21    :math:`O(n)` です。
 22
 23    Args:
 24      a (Iterable[T], optional): 構築する配列です。
 25    """
 26    self._data = list(a)
 27    self._heapify()
 28
 29  def _heapify(self) -> None:
 30    n = len(self._data)
 31    for i in range(n-1, -1, -1):
 32      if i & 1 and self._data[i-1] < self._data[i]:
 33        self._data[i-1], self._data[i] = self._data[i], self._data[i-1]
 34      k = self._down(i)
 35      self._up(k, i)
 36
 37  def add(self, key: T) -> None:
 38    """``key`` を1つ追加します。
 39
 40    :math:`O(\\log{n})` です。
 41    """
 42    self._data.append(key)
 43    self._up(len(self._data)-1)
 44
 45  def pop_min(self) -> T:
 46    """最小の要素を **削除して** 返します。
 47
 48    :math:`O(\\log{n})` です。
 49    """
 50    if len(self._data) < 3:
 51      res = self._data.pop()
 52    else:
 53      self._data[1], self._data[-1] = self._data[-1], self._data[1]
 54      res = self._data.pop()
 55      k = self._down(1)
 56      self._up(k)
 57    return res
 58
 59  def pop_max(self) -> T:
 60    """最大の要素を **削除して** 返します。
 61
 62    :math:`O(\\log{n})` です。
 63    """
 64    if len(self._data) < 2:
 65      res = self._data.pop()

 69      self._up(self._down(0))

 95  def _down(self, k: int) -> int:
 96    n = len(self._data)
 97    if k & 1:
 98      while k<<1|1 < n:
 99        c = 2*k+3
100        if n <= c or self._data[c-2] < self._data[c]:
101          c -= 2
102        if c < n and self._data[c] < self._data[k]:
103          self._data[k], self._data[c] = self._data[c], self._data[k]
104          k = c
105        else:
106          break
107    else:
108      while 2*k+2 < n:
109        c = 2*k+4
110        if n <= c or self._data[c] < self._data[c-2]:
111          c -= 2
112        if c < n and self._data[k] < self._data[c]:
113          self._data[k], self._data[c] = self._data[c], self._data[k]
114          k = c
115        else:
116          break
117    return k

119  def _up(self, k: int, root: int=1) -> int:
120    if (k|1) < len(self._data) and self._data[k&~1] < self._data[k|1]:
121      self._data[k&~1], self._data[k|1] = self._data[k|1], self._data[k&~1]
122      k ^= 1
123    while root < k:
124      p = self._parent(k)
125      if not self._data[p] < self._data[k]:
126        break
127      self._data[p], self._data[k] = self._data[k], self._data[p]
128      k = p
129    while root < k:
130      p = self._parent(k) | 1
131      if not self._data[k] < self._data[p]:
132        break
133      self._data[p], self._data[k] = self._data[k], self._data[p]
134      k = p
135    return k

149class MinMaxMultiset(Generic[T]):
150
151  def __init__(self, a: Iterable[T]=[]):
152    a = list(a)
153    data = {}
154    for x in a:
155      if x in data:
156        data[x] += 1
157      else:
158        data[x] = 1
159    self.data = data
160    self.heap = DoubleEndedHeap(a)
161    self.len = len(a)
162
163  def add(self, key: T, val: int=1) -> None:
164    if val == 0: return
165    self.heap.add(key)
166    if key in self.data:
167      self.data[key] += val
168    else:
169      self.data[key] = val
170    self.len += val
171
172  def discard(self, key: T, val: int=1) -> bool:
173    if key not in self.data: return False
174    cnt = self.data[key]
175    if val < cnt:
176      self.len -= val
177      self.data[key] -= val
178    else:
179      self.len -= cnt
180      del self.data[key]
181    return True
182
183  def pop_min(self) -> T:
184    while True:
185      v = self.heap.get_min()
186      if v in self.data:
187        if self.data[v] == 1:
188          self.heap.pop_min()
189          del self.data[v]
190        else:
191          self.data[v] -= 1
192        self.len -= 1
193        return v
194      self.heap.pop_min()
195
196  def pop_max(self) -> T:
197    while True:
198      v = self.heap.get_max()
199      if v in self.data:
200        self.len -= 1
201        if self.data[v] == 1:
202          self.heap.pop_max()
203          del self.data[v]
204        else:
205          self.data[v] -= 1
206        return v
207      self.heap.pop_max()
208
209  def get_min(self) -> T:
210    while True:
211      v = self.heap.get_min()
212      if v in self.data:
213        return v
214      else:
215        self.heap.pop_min()

217  def get_max(self) -> T:
218    while True:
