 65class SegmentTree(SegmentTreeInterface, Generic[T]):
 66  """セグ木です。非再帰です。
 67  """
 68
 69  def __init__(self,
 70               n_or_a: Union[int, Iterable[T]],
 71               op: Callable[[T, T], T],
 72               e: T) -> None:
 73    """``SegmentTree`` を構築します。
 74    :math:`O(n)` です。
 75
 76    Args:
 77      n_or_a (Union[int, Iterable[T]]): ``n: int`` のとき、 ``e`` を初期値として長さ ``n`` の ``SegmentTree`` を構築します。
 78                                        ``a: Iterable[T]`` のとき、 ``a`` から ``SegmentTree`` を構築します。
 79      op (Callable[[T, T], T]): 2項演算の関数です。
 80      e (T): 単位元です。
 81    """
 82    self._op = op
 83    self._e = e
 84    if isinstance(n_or_a, int):
 85      self._n = n_or_a
 86      self._log = (self._n - 1).bit_length()
 87      self._size = 1 << self._log
 88      self._data = [e] * (self._size << 1)
 89    else:
 90      n_or_a = list(n_or_a)
 91      self._n = len(n_or_a)
 92      self._log = (self._n - 1).bit_length()
 93      self._size = 1 << self._log
 94      _data = [e] * (self._size << 1)
 95      _data[self._size:self._size+self._n] = n_or_a
 96      for i in range(self._size-1, 0, -1):
 97        _data[i] = op(_data[i<<1], _data[i<<1|1])
 98      self._data = _data

100  def set(self, k: int, v: T) -> None:
101    """一点更新です。
102    :math:`O(\\log{n})` です。
103
104    Args:
105      k (int): 更新するインデックスです。
106      v (T): 更新する値です。
107
108    制約:
109      :math:`-n \\leq n \\leq k < n`
110    """
111    assert -self._n <= k < self._n, \
112        f'IndexError: {self.__class__.__name__}.set({k}, {v}), n={self._n}'
113    if k < 0:
114      k += self._n
115    k += self._size
116    self._data[k] = v
117    for _ in range(self._log):
118      k >>= 1
119      self._data[k] = self._op(self._data[k<<1], self._data[k<<1|1])

131    assert -self._n <= k < self._n, \

133    if k < 0:
134      k += self._n
135    return self._data[k+self._size]

137  def prod(self, l: int, r: int) -> T:
138    """区間 ``[l, r)`` の総積を返します。
139    :math:`O(\\log{n})` です。
140
141    Args:
142      l (int): インデックスです。
143      r (int): インデックスです。
144
145    制約:
146      :math:`0 \\leq l \\leq r \\leq n`
147    """
148    assert 0 <= l <= r <= self._n, \
149        f'IndexError: {self.__class__.__name__}.prod({l}, {r})'
150    l += self._size
151    r += self._size
152    lres = self._e
153    rres = self._e
154    while l < r:
155      if l & 1:
156        lres = self._op(lres, self._data[l])
157        l += 1
158      if r & 1:
159        rres = self._op(self._data[r^1], rres)
160      l >>= 1
161      r >>= 1
162    return self._op(lres, rres)

170  def max_right(self, l: int, f: Callable[[T], bool]) -> int:
171    '''Find the largest index R s.t. f([l, R)) == True. / O(\\log{n})'''
172    assert 0 <= l <= self._n, \
173        f'IndexError: {self.__class__.__name__}.max_right({l}, f) index out of range'
174    assert f(self._e), \
175        f'{self.__class__.__name__}.max_right({l}, f), f({self._e}) must be true.'
176    if l == self._n:
177      return self._n
178    l += self._size
179    s = self._e
180    while True:
181      while l & 1 == 0:
182        l >>= 1
183      if not f(self._op(s, self._data[l])):
184        while l < self._size:
185          l <<= 1
186          if f(self._op(s, self._data[l])):
187            s = self._op(s, self._data[l])
188            l |= 1
189        return l - self._size
190      s = self._op(s, self._data[l])
191      l += 1
192      if l & -l == l:
193        break
194    return self._n
195
196  def min_left(self, r: int, f: Callable[[T], bool]) -> int:
197    '''Find the smallest index L s.t. f([L, r)) == True. / O(\\log{n})'''
198    assert 0 <= r <= self._n, \
199        f'IndexError: {self.__class__.__name__}.min_left({r}, f) index out of range'
200    assert f(self._e), \
201        f'{self.__class__.__name__}.min_left({r}, f), f({self._e}) must be true.'
202    if r == 0:
203      return 0
204    r += self._size
205    s = self._e
206    while True:
207      r -= 1
208      while r > 1 and r & 1:
209        r >>= 1
210      if not f(self._op(self._data[r], s)):
211        while r < self._size:
212          r = r << 1 | 1
213          if f(self._op(self._data[r], s)):
214            s = self._op(self._data[r], s)
215            r ^= 1
216        return r + 1 - self._size
217      s = self._op(self._data[r], s)
218      if r & -r == r:
219        break
220    return 0

260class HashStringBase():
261  """HashStringのベースクラスです。
262  """
263
264  def __init__(self, n: int, base: int=-1, seed: Optional[int]=None) -> None:
265    """
266    :math:`O(n)` です。
267
268    Args:
269        n (int): 文字列の長さの上限です。
270        base (int, optional): Defaults to -1.
271        seed (Optional[int], optional): Defaults to None.
272    """
273    random.seed(seed)
274    base = random.randint(37, 10**9) if base < 0 else base
275    powb = [1] * (n+1)
276    invb = [1] * (n+1)
277    invbpow = pow(base, -1, _titan_pylib_HashString_MOD)
278    for i in range(1, n+1):
279      powb[i] = HashStringBase.get_mul(powb[i-1], base)
280      invb[i] = HashStringBase.get_mul(invb[i-1], invbpow)
281    self.n = n
282    self.powb = powb
283    self.invb = invb
284
285  @staticmethod
286  def get_mul(a: int, b: int) -> int:
287    au = a >> 31
288    ad = a & _titan_pylib_HashString_MASK31
289    bu = b >> 31
290    bd = b & _titan_pylib_HashString_MASK31
291    mid = ad * bu + au * bd
292    midu = mid >> 30
293    midd = mid & _titan_pylib_HashString_MASK30
294    return HashStringBase.get_mod(au * bu * 2 + midu + (midd << 31) + ad * bd)
295
296  @staticmethod
297  def get_mod(x: int) -> int:
298    xu = x >> 61
299    xd = x & _titan_pylib_HashString_MASK61
300    res = xu + xd
301    if res >= _titan_pylib_HashString_MOD:
302      res -= _titan_pylib_HashString_MOD
303    return res
304
305  def unite(self, h1: int, h2: int, k: int) -> int:
306    # len(h2) == k
307    # h1 <- h2
308    return self.get_mod(self.get_mul(h1, self.powb[k]) + h2)

310class HashString():
311
312  def __init__(self, hsb: HashStringBase, s: str, update: bool=False) -> None:
313    """ロリハを構築します。
314
315    :math:`O(n)` です。
316
317    Args:
318        hsb (HashStringBase): ベースクラスです。
319        s (str): ロリハを構築する文字列です。
320        update (bool, optional): ``update=True`` のとき、1点更新が可能になります。
321    """
322    n = len(s)
323    data = [0] * n
324    acc = [0] * (n+1)
325    powb = hsb.powb
326    for i, c in enumerate(s):
327      data[i] = hsb.get_mul(powb[n-i-1], _titan_pylib_HashString_DIC[c])
328      acc[i+1] = hsb.get_mod(acc[i] + data[i])
329    self.hsb = hsb
330    self.n = n
331    self.acc = acc
332    self.used_seg = False
333    if update:
334      self.seg = SegmentTree(data, lambda s, t: (s+t)%_titan_pylib_HashString_MOD, 0)
335
336  def get(self, l: int, r: int) -> int:
337    """``s[l, r)`` のハッシュ値を返します。
338
339    1点更新処理後は :math:`O(\\log{n})` 、そうでなければ :math:`O(1)` です。
340
341    Args:
342      l (int): インデックスです。
343      r (int): インデックスです。
344
345    Returns:
346      int: ハッシュ値です。
347    """
348    if self.used_seg:
349      return self.hsb.get_mul(self.seg.prod(l, r), self.hsb.invb[self.n-r])
350    return self.hsb.get_mul(self.hsb.get_mod(self.acc[r]-self.acc[l]), self.hsb.invb[self.n-r])

383  def get_lcp(self) -> List[int]:
384    """lcp配列を返します。
385
386    :math:`O(n\\log{n})` です。
387    """
388    a = [0] * self.n
389    memo = [-1] * (self.n+1)
390    for i in range(self.n):
391      ok, ng = 0, self.n-i+1
392      while ng - ok > 1:
393        mid = (ok + ng) >> 1
394        if memo[mid] == -1:
395          memo[mid] = self.get(0, mid)
396        if memo[mid] == self.get(i, i+mid):
397          ok = mid
398        else:
399          ng = mid
400      a[i] = ok
401    return a

415def merge_sort(a: Iterable[T], key: Callable[[T, T], bool]=lambda s, t: s < t) -> List[T]:
416  """マージソートです。
417
418  非破壊的です。
419  最悪 :math:`O(n\\log{n})` 時間です。
420
421  Args:
422    a (Iterable[T]): ソートする列です。
423    key (Callable[[T, T], bool], optional): 比較関数 `key` にしたがって比較演算をします。
424                                            (第1引数)<(第2引数) のとき、 ``True`` を返すようにしてください。
425  """
426  def _sort(a: List[T]) -> List[T]:
427    n = len(a)
428    if n <= 1:
429      return a
430    if n == 2:
431      if not key(a[0], a[1]):
432        a[0], a[1] = a[1], a[0]
433      return a
434    left = _sort(a[:n//2])
435    right = _sort(a[n//2:])
436    res = newlist_hint(n)
437    i, j, l, r = 0, 0, len(left), len(right)
438    while i < l and j < r:
439      if key(left[i], right[j]):
440        res.append(left[i])
441        i += 1
442      else:
443        res.append(right[j])
444        j += 1
445    for i in range(i, l):
446      res.append(left[i])
447    for j in range(j, r):
448      res.append(right[j])
449    return res
450  return _sort(list(a))

454def get_suffix_array(s: str, hs: HashString) -> List[int]:
455  """suffix_arrayを求めます。
456
457  ロリハで大小比較をするため、比較関数に :math:`O(logn)` 、
458  ソートが :math:`O(nlogn*(比較関数の計算慮))` で、
459  全体 :math:`O(nlog^2n)` です。
460
461  Args:
462      s (str): 文字列です。
463      hs (HashString): ``HashString`` です。
464
465  Returns:
466      List[int]: suffix_array です。
467  """
468  def cmp(u: int, v: int) -> bool:
469    ok, ng = 0, min(n-u, n-v)
470    while ng - ok > 1:
471      mid = (ok + ng) >> 1
472      if hs.get(u, u+mid) == hs.get(v, v+mid):
473        ok = mid
474      else:
475        ng = mid
476    return s[u+ok] < s[v+ok]
477  n = len(s)
478  return merge_sort(range(n), key=cmp)
