 11import math
 12from bisect import bisect_left, bisect_right
 13from typing import Generic, Iterable, Iterator, List, Tuple, TypeVar, Optional
 14T = TypeVar('T', bound=SupportsLessThan)
 15
 16class SortedSet(Generic[T]):
 17
 18  BUCKET_RATIO = 50
 19  REBUILD_RATIO = 170
 20
 21  def _build(self, a: Optional[List[T]] = None) -> None:
 22    "Evenly divide `a` into buckets."
 23    if a is None: a = list(self)
 24    size = len(a)
 25    bucket_size = int(math.ceil(math.sqrt(size / self.BUCKET_RATIO)))
 26    self.a = [a[size * i // bucket_size : size * (i + 1) // bucket_size] for i in range(bucket_size)]
 27  
 28  def __init__(self, a: Iterable[T] = []) -> None:
 29    "Make a new SortedSet from iterable. / O(N) if sorted and unique / O(N log N)"
 30    a = list(a)
 31    self.size = len(a)
 32    if not all(a[i] < a[i + 1] for i in range(len(a) - 1)):
 33      a = sorted(set(a))
 34    self._build(a)
 35
 36  def __iter__(self) -> Iterator[T]:
 37    for i in self.a:
 38      for j in i: yield j
 39
 40  def __reversed__(self) -> Iterator[T]:
 41    for i in reversed(self.a):
 42      for j in reversed(i): yield j
 43  
 44  def __eq__(self, other) -> bool:
 45    return list(self) == list(other)
 46  
 47  def __len__(self) -> int:
 48    return self.size
 49  
 50  def __repr__(self) -> str:
 51    return "SortedSet" + str(self.a)
 52  
 53  def __str__(self) -> str:
 54    s = str(list(self))
 55    return "{" + s[1 : len(s) - 1] + "}"

 57  def _position(self, x: T) -> Tuple[List[T], int]:
 58    "Find the bucket and position which x should be inserted. self must not be empty."
 59    for a in self.a:
 60      if x <= a[-1]: break
 61    return (a, bisect_left(a, x))
 62
 63  def __contains__(self, x: T) -> bool:
 64    if self.size == 0: return False
 65    a, i = self._position(x)
 66    return i != len(a) and a[i] == x
 67
 68  def add(self, x: T) -> bool:
 69    "Add an element and return True if added. / O(√N)"
 70    if self.size == 0:
 71      self.a = [[x]]
 72      self.size = 1
 73      return True
 74    a, i = self._position(x)
 75    if i != len(a) and a[i] == x: return False
 76    a.insert(i, x)
 77    self.size += 1
 78    if len(a) > len(self.a) * self.REBUILD_RATIO:
 79      self._build()
 80    return True
 81  
 82  def _pop(self, a: List[T], i: int) -> T:
 83    ans = a.pop(i)
 84    self.size -= 1
 85    if not a: self._build()
 86    return ans
 87
 88  def discard(self, x: T) -> bool:
 89    "Remove an element and return True if removed. / O(√N)"
 90    if self.size == 0: return False
 91    a, i = self._position(x)
 92    if i == len(a) or a[i] != x: return False
 93    self._pop(a, i)
 94    return True

 96  def lt(self, x: T) -> Optional[T]:
 97    "Find the largest element < x, or None if it doesn't exist."
 98    for a in reversed(self.a):
 99      if a[0] < x:
100        return a[bisect_left(a, x) - 1]
101
102  def le(self, x: T) -> Optional[T]:
103    "Find the largest element <= x, or None if it doesn't exist."
104    for a in reversed(self.a):
105      if a[0] <= x:
106        return a[bisect_right(a, x) - 1]
107
108  def gt(self, x: T) -> Optional[T]:
109    "Find the smallest element > x, or None if it doesn't exist."
110    for a in self.a:

113
114  def ge(self, x: T) -> Optional[T]:
115    "Find the smallest element >= x, or None if it doesn't exist."
116    for a in self.a:

120  def __getitem__(self, i: int) -> T:
121    "Return the i-th element."
122    if i < 0:
123      for a in reversed(self.a):
124        i += len(a)
125        if i >= 0: return a[i]
126    else:
127      for a in self.a:
128        if i < len(a): return a[i]
129        i -= len(a)
130    raise IndexError

132  def pop(self, i: int = -1) -> T:
133    "Pop and return the i-th element."
134    if i < 0:
135      for a in reversed(self.a):
136        i += len(a)
137        if i >= 0: return self._pop(a, i)
138    else:
139      for a in self.a:
140        if i < len(a): return self._pop(a, i)
141        i -= len(a)
142    raise IndexError

157    "Count the number of elements < x."
158    ans = 0

162      ans += len(a)
