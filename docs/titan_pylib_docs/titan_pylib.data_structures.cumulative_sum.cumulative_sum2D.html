10
11    :math:`O(hw)` です。
12
13    Args:
14      h (int): 行数です。

18    acc = [0] * ((h+1)*(w+1))
19    for ij in range(h*w):
20      i, j = divmod(ij, w)
21      acc[(i+1)*(w+1)+j+1] = acc[i*(w+1)+j+1] + acc[(i+1)*(w+1)+j] - acc[i*(w+1)+j] + a[i][j]
22    self.h = h
23    self.w = w
24    self.acc = acc
